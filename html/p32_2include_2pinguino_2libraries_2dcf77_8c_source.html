68 /*******************************************************************************
69  * Appends a received bit at the DCF code chain
70  ******************************************************************************/
71 void DCF77_appendSignal(u8 signal)
72 {
73 // serial1printf("BuffPntr: %02d ", BuffPntr);
74  if ((signal > 0) && (BuffPntr < 32))
75  Buff1 = Buff1 | (1 << BuffPntr);
76  if ((signal > 0) && (BuffPntr > 31))
77  Buff2 = Buff2 | (1 << (BuffPntr - 32));
78 // serial1printf("Append Buff: %08x %08x \r\n", DCF77.Buff1, DCF77.Buff2);
79  BuffPntr++;
80  if (BuffPntr > 59)
81  {
82  BuffPntr = 0;
83  Buff1 = 0;
84  Buff2 = 0;
85  }
86 }

88 /*******************************************************************************
89 * Calculate even parity
90 *******************************************************************************/
91 u8 Parity_Even (u32 Bits)
92 {
93  u8 i;
94  u8 p = 0;
95  for (i = 0; i < 32; i++)
96  if ((Bits & (1 << i)) > 0)
97  p = !p;
98  return p;

101 /*******************************************************************************
102  * Decodes DCF code after the 59 minute gap
103  ******************************************************************************/
104 void Decode_Buffer(void)
105 {
106  u32 Code;
107  u8 Par, Par1, Par2, Par3;
108 // Disp_Input_Binary(Buff1, Buff2);
109 
110  if (BuffPntr == 59)
111  {
112  // Minutes
113  Code = (Buff1 >> 21) & 0x7F;
114  Par = Parity_Even(Code);
115  Par1 = (Buff1 >> 28) & 1;
116 // serial1printf("Par= %d Par1= %d\r\n", Par, Par1);
117  if (Par == Par1)
118  {
119  Par1 = 11;
120  DCF77.minutes = bcd2bin(Code);
121 // serial1printf("mm= %02d \r\n", DCF77.mm);
122  }
123 
124  // Hours
125  Code = ((Buff1 >> 29) & 0x7) | ((Buff2 & 0x7) << 3);
126  Par = Parity_Even(Code);
127  Par2 = (Buff2 >> 3) & 1;
128 // serial1printf("Par= %d Par2= %d\r\n", Par, Par2);
129  if (Par == Par2)
130  {
131  Par2 = 12;
132  DCF77.hours = bcd2bin(Code);
133 // serial1printf("hh= %02d \r\n", DCF77.hh);
134  }
135 
136  Code = (Buff2 >> 4) & 0x3FFFFF;
137  Par = Parity_Even(Code);
138  Par3 = (Buff2 >> 26) & 1;
139 // serial1printf("Par= %d Par3= %d\r\n", Par, Par3);
140  if (Par == Par3)
141  {
142  Par3 = 13;
143  Code = (Buff2 >> 10) & 0x7;
144  DCF77.dayofweek = bcd2bin(Code);
145 // serial1printf("dw= %02d \r\n", DCF77.dw);
146 
147  Code = (Buff2 >> 4) & 0x3F;
148  DCF77.dayofmonth = bcd2bin(Code);
149 // serial1printf("da= %02d \r\n", DCF77.da);
150 
151  Code = (Buff2 >> 13) & 0x1F;
152  DCF77.month = bcd2bin(Code);
153 // serial1printf("mo= %02d \r\n", DCF77.mo);
154 
155  Code = (Buff2 >> 18) & 0xFF;
156  DCF77.year = bcd2bin(Code);
157 // serial1printf("yr= %02d \r\n", DCF77.yr);
158  }
159 // Part = Par1 + Par2 + Par3;
160 // serial1printf("Part= %d\r\n", Part);
161  if (Par1 + Par2 + Par3 == 36)
162  {
163  DCF77.seconds = 0;
164  Nosync = 0;
165 // serial1printf("Tim= %d:%02d \r\n", DCF77.hh, DCF77.mm);
166 // serial1printf("Dat=: %02d-%02d-%02d %1d\r\n", DCF77.da, DCF77.mo, DCF77.yr, DCF77.dw);
167  RTClock = DCF77;
168  }
169  }
170  // Reset Inputbuffer
171  BuffPntr = 0;
172  Buff1 = 0;
173  Buff2 = 0;
174 }

176 /*******************************************************************************
177  * Called every 20 ms by ISR
178  * Builds DCF code by calling DCF77_appendsignal
179  ******************************************************************************/
180 void DCF77_scanSignal(void)
181 {
182  u16 LengthPulse;
183  u16 CurrFlank;
184  u8 CurrSignal ;
185  CurrSignal = digitalread(SignalPin);
186  if (CurrSignal != PrevSignal)
187  {
188  if (CurrSignal == 1)
189  {
190 // digitalwrite(GREENLED, HIGH);
191  /* At a raising flank */
192  CurrFlank = Cntr_20ms;
193  if ((CurrFlank - PrevFlank) > DCF_Sync_1500ms)
194  Decode_Buffer();
195  PrevFlank = CurrFlank;
196  }
197  else
198  {
199 // digitalwrite(GREENLED, LOW);
200  /* At a falling flank */
201  LengthPulse = Cntr_20ms - PrevFlank;
202  if (LengthPulse < DCF_Discr_150ms)
203  DCF77_appendSignal(0);
204  else
205  DCF77_appendSignal(1);
206  }
207  PrevSignal = CurrSignal;
208  }
209 }

211 /*******************************************************************************
212 * Update Time/Date record (every second by ISR)
213 *******************************************************************************/
214 void Update_Time (void)
215 {
216  u8 Mnth[12] = {31, 29, 31, 30, 31, 30, 31, 31, 30, 31, 30, 31};
217 
218  if (RTClock.seconds == 0)
219  RTClock.nosync = Nosync;
220  RTClock.seconds++; // Seconds
221  if (RTClock.seconds > 59)
222  {
223  Nosync++;
224  RTClock.seconds = 0;
225  RTClock.minutes++; // Minutes
226  if (RTClock.minutes > 59)
227  {
228  RTClock.minutes = 0;
229  RTClock.hours++; // Hours
230  if (RTClock.hours > 23)
231  {
232  RTClock.hours = 0;
233  RTClock.dayofweek++; // Day of Week
234  if (RTClock.dayofweek > 7)
235  RTClock.dayofweek = 1;
236  RTClock.dayofmonth++; // Date
237  if (RTClock.dayofmonth > Mnth[RTClock.month-1])
238  {
239  RTClock.dayofmonth = 1;
240  RTClock.month++; // Month
241  if (RTClock.month > 12)
242  {
243  RTClock.month = 1;
244  RTClock.year++; // Year
245  if (RTClock.year > 99)
246  RTClock.year = 0;
247  }
248  }
249  }
250  }
251  }

269 /* Timer1 Configuration
270  The timer clock prescale (TCKPS) is 1:64
271  The TMR1 Count register increments on every PBCLK clock cycle
272  PBCLK = CPUCLK / 2 = 40MHz => TMR1 inc. every 64/40MHZ = 1,6 us
273  20 ms = 20000 us => 20000 / 1.6 = 12500 cycles
274 */
275  IntConfigureSystem(INT_SYSTEM_CONFIG_MULT_VECTOR);
276  T1CON = 0x20; // prescaler 1:64, internal peripheral clock
277  TMR1 = 0; // clear timer register
278  #if defined(PIC32_PINGUINO_220)||defined(PINGUINO32MX250)||defined(PINGUINO32MX220)
279  PR1 = 6250; // load period register
280  #else
281  PR1 = 12500; // load period register
282  #endif
283  IPC1SET = 7; // select interrupt priority and sub-priority
284  IFS0CLR = 1 << INT_TIMER1_VECTOR; // clear interrupt flag
285  IEC0SET = 1 << INT_TIMER1_VECTOR; // enable timer 1 interrupt
286  T1CONSET = 0x8000; // start timer 1
287 }
