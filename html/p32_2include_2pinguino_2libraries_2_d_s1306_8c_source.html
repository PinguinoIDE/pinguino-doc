251 // Encodes an hour byte using selected default format for write, honoring "ALL" (for alarms)
252 unsigned char DS1306_encodeHourByte(unsigned char hour24, unsigned char hour12, char ampm)
253 {
254  if ((DS1306_writeHours24 && ((hour24 & DS1306_ANY))) || (!DS1306_writeHours24 && (hour12 & DS1306_ANY))) {
255  return DS1306_ANY;
256  } else {
257  if (DS1306_writeHours24) {
258  // Writing 24 hour format times
259  return DS1306_encodeBCD7M(hour24, 0x3F);
260  } else {
261  return (DS1306_encodeBCD7M(hour12, 0x1F) | 0x40 | (ampm == 'P' ? 0x20 : 0x00));
262  }
263  }
264 }

266 // Decodes an hour byte, honoring "ALL" (for alarms)
267 void DS1306_decodeHourByte(unsigned char hourByte, unsigned char *hour24, unsigned char *hour12, char *ampm)
268 {
269  if (hourByte & DS1306_ANY) {
270  *hour12 = DS1306_ANY;
271  *hour24 = DS1306_ANY;
272  *ampm = 0;
273  } else {
274  if (hourByte & 0x40) {
275  // 12 hour mode
276  *hour12 = DS1306_decodeBCD7M(hourByte, 0x1F);
277  if (hourByte & 0x20) {
278  // PM
279  if (*hour12 == 12) {
280  *hour24 = 12;
281  } else {
282  *hour24 = 12 + *hour12;
283  }
284  *ampm = 'P';
285  } else {
286  // AM
287  *hour24 = (*hour12 == 12) ? 0 : *hour12;
288  *ampm = 'A';
289  }
290  } else {
291  // 24 hour mode
292  *hour24 = DS1306_decodeBCD7M(hourByte, 0x3F);
293  if (*hour24 == 0) {
294  *hour12 = 12;
295  *ampm = 'A';
296  } else if (*hour24 == 12) {
297  *hour12 = 12;
298  *ampm = 'P';
299  } else if (*hour24 < 12) {
300  *ampm = 'A';
301  *hour12 = *hour24;
302  } else {
303  *ampm = 'P';
304  *hour12 = *hour24 - 12;
305  }
306  }
307  }
308 }

421 // Enable trickle charging
422 // Must provide number of diodes (1 or 2)
423 // and KOhm resistance (2, 4 or 8)
424 // Any other values will fail (false returned, no changes made)
425 BOOL DS1306_enableTrickleCharge(unsigned char numDiodes, unsigned char kRes)
426 {
427  if (numDiodes < 1 || numDiodes > 2) return false;
428 
429  unsigned char byte = 0xA0 | (numDiodes << 2);
430  switch(kRes) {
431  case 2 : byte |= 0x01; break;
432  case 4 : byte |= 0x02; break;
433  case 8 : byte |= 0x03; break;
434  default : return false;
435  }
436 
437  DS1306_writen(DS1306_TCR, &byte, 1);
438 }

447 // Retrieve trickle charging state
448 // Returns true (Trickle enabled), false (Trickle disabled)
449 // When enabled, numDiodes and kRes will be set to (1, 2) and (2, 4, 8) respectively
450 // When disabled numDiodes and kRes will be set to 0
451 BOOL DS1306_getTrickleChargeState(unsigned char *numDiodes, unsigned char *kRes)
452 {
453  unsigned char byte;
454  DS1306_readn(DS1306_TCR, &byte, 1);
455 
456  *numDiodes = 0;
457  *kRes = 0;
458 
459  if ((byte & 0xF0) != 0xA0) return false; // TCR disabled
460 
461  switch(byte & 0x03) {
462  case 0x01 : *kRes = 2; break;
463  case 0x02 : *kRes = 4; break;
464  case 0x03 : *kRes = 8; break;
465  default : return false;
466  }
467 
468  // TCR is enabled if we got this far
469  *numDiodes = (byte & 0x0C) >> 2;
470 
471  return true;
472 }
