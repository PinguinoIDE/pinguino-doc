97  sets row_offsets for a single line display so that

136 void _lcd_printNumber(u32 n, u8 base) {
137  u8 buf[8 * sizeof(long)]; // Assumes 8-bit chars.
138  u32 i = 0;
139 
140  if (n == 0) {
141  _lcd_write('0');
142  return;
143  }
144 
145  while (n > 0) {
146  buf[i++] = n % base;
147  n /= base;
148  }
149 
150  for (; i > 0; i--)
151  _lcd_write((char) (buf[i - 1] < 10 ?
152  '0' + buf[i - 1] :
153  'A' + buf[i - 1] - 10));
154 }

157 void _lcd_printFloat(float number, u8 digits) {
158  u8 i, toPrint;
159  u32 int_part;
160  float rounding, remainder;
161 
162  // Handle negative numbers
163  if (number < 0.0) {
164  _lcd_write('-');
165  number = -number;
166  }
167 
168  // Round correctly so that print(1.999, 2) prints as "2.00"
169  rounding = 0.5;
170  for (i=0; i<digits; ++i)
171  rounding /= 10.0;
172 
173  number += rounding;
174 
175  // Extract the integer part of the number and print it
176  int_part = (u32)number;
177  remainder = number - (float)int_part;
178  _lcd_printNumber(int_part, 10);
179 
180  // Print the decimal point, but only if there are digits beyond
181  if (digits > 0)
182  _lcd_write('.');
183 
184  // Extract digits from the remainder one at a time
185  while (digits-- > 0) {
186  remainder *= 10.0;
187  toPrint = (u16)remainder; //Integer part without use of math.h lib,
188  //I think better! (Fazzi)
189  _lcd_printNumber(toPrint, 10);
190  remainder -= toPrint;
191  }
192 }

269 void _lcd_begin(u8 lines, u8 dotsize) {
270  if (lines > 1) {
271  _displayfunction |= LCD_2LINE;
272  }
273  _numlines = lines;
274  _currline = 0;
275 
276  // Some one line displays can select 10 pixel high font
277  if ((dotsize != 0) && (lines == 1)) {
278  _displayfunction |= LCD_5x10DOTS;
279  }
280 
281  Delayus(15); //Pinguino needs it? long delay on startup time!
282  // Now we pull both RS and R/W low to begin commands
283  digitalwrite(_rs_pin, LOW);
284  digitalwrite(_enable_pin, LOW);
285 
286  //put the LCD into 4 bit or 8 bit mode
287  if (! (_displayfunction & LCD_8BITMODE)) {
288  // this is according to the hitachi HD44780 datasheet
289  // figure 24, pg 46
290 
291  // we start in 8bit mode, try to set 4 bit mode
292  _lcd_write4bits(0x03);
293  Delayus(4500); // wait min 4.1ms
294 
295  // second try
296  _lcd_write4bits(0x03);
297  Delayus(4500); // wait min 4.1ms
298 
299  // third go!
300  _lcd_write4bits(0x03);
301  Delayus(150);
302 
303  // finally, set to 8-bit interface
304  _lcd_write4bits(0x02);
305  } else {
306  // this is according to the hitachi HD44780 datasheet
307  // page 45 figure 23
308 
309  // Send function set command sequence
310  _lcd_command(LCD_FUNCTIONSET | _displayfunction);
311  Delayus(4500); // wait more than 4.1ms
312 
313  // second try
314  _lcd_command(LCD_FUNCTIONSET | _displayfunction);
315  Delayus(150);
316 
317  // third go
318  _lcd_command(LCD_FUNCTIONSET | _displayfunction);
319  }
320 
321  // finally, set # lines, font size, etc.
322  _lcd_command(LCD_FUNCTIONSET | _displayfunction);
323 
324  // turn the display on with no cursor or blinking default
325  _displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
326  _lcd_display();
327 
328  // clear it off
329  _lcd_clear();
330 
331  // Initialize to default text direction (for romance languages)
332  _displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
333  // set the entry mode
334  _lcd_command(LCD_ENTRYMODESET | _displaymode);
335 
336 }
