233 /* ----------------------------------------------------------------------------
234  ---------- Ecriture d'un quartet (mode 4 bits) dans le LCD
235  ----------------------------------------------------------------------------
236  Envoie d'un quartet vers les pins :
237  - D4 a D7 du LCD
238  - P4 a P7 du PCF8574
239  NB : quartet est en fait un 8 bits dont seuls les 4 bits de poids fort nous interessent
240  @param quartet = 4 bits a envoyer au LCD
241  @param mode = LCD Command (LCD_CMD) or Data (LCD_DATA) mode
242  --------------------------------------------------------------------------*/
243 
244 static void lcdi2c_send4(u8 quartet, u8 mode)
245 {
246  // D7 D6 D5 D4 EN RW RS BL
247  PCF8574_data.val = quartet; // x x x x 0 0 0 0
248  //LCD_EN = LOW; // x x x x 0 0 0 0
249  LCD_RW = LCD_WRITE; // x x x x 0 0 0 0
250  LCD_RS = mode; // x x x x 0 0 0/1 0
251  LCD_BL = Backlight; // x x x x 0 0 0/1 0/1
252 
254 
255  //LCD_EN = LOW;
256  //I2C_send(I2C1, PCF8574_address, PCF8574_data.val);
257  //Delayus(1);
258 
259  LCD_EN = HIGH;
260  I2C_send(I2C1, PCF8574_address, PCF8574_data.val);
261  Delayus(1);// E Pulse Width > 300ns
262 
263  LCD_EN = LOW;
264  I2C_send(I2C1, PCF8574_address, PCF8574_data.val);
265  Delayus(1);// E Enable Cycle > (300 + 200) = 500ns
266 }

268 /* ----------------------------------------------------------------------------
269  ---------- Ecriture d'un octet dans le LCD en mode 4 bits
270  ----------------------------------------------------------------------------
271  Les données sont écrites en envoyant séquentiellement :
272  1/ les quatre bits de poids fort
273  2/ les quatre bits de poids faible
274  NB : les poids sont stockes dans les quatre bits de poids fort
275  qui correspondent aux pins D4 a D7 du LCD ou du PCF8574
276  @param octet = octet a envoyer au LCD
277  @param mode = LCD Command (LCD_CMD) or Data (LCD_DATA) mode
278  --------------------------------------------------------------------------*/
279 
280 static void lcdi2c_send8(u8 octet, u8 mode)
281 {
282  lcdi2c_send4(octet & LCD_MASK, mode); // envoie les "Upper 4 bits"
283  lcdi2c_send4((octet << 4) & LCD_MASK, mode); // envoie les "Lower 4 bits"
284  Delayus(46); // Wait for instruction excution time (more than 46us)
285 }

492 /* ----------------------------------------------------------------------------
493  ---------- Définit un caractère personnalisé de 8x8 points.
494  ----------------------------------------------------------------------------
495  d'après Nabil Al-HOSSRI <http://nalhossri.free.fr>
496  ----------------------------------------------------------------------------
497  Le LCD utilisé admet au maximum 8 caractères spéciaux.
498  char_code : code du caractère à définir (0 <= char_code <= 7)
499  lcdi2c_newchar(car3, 0); Définit le caractère 'é' à l'adresse 0 de la mémoire CG RAM
500  lcdi2c_newchar(car8, 1); Définit le caractère 'è' à l'adresse 1.
501  --------------------------------------------------------------------------*/
502 
503 void lcdi2c_newchar(const u8 *c, u8 char_code)
504 {
505  u8 i, a;
506 
507  // les caractères sont logés dans le CGRAM du LCD à partir de l'adresse 0x40.
508  a = (char_code << 3) | LCD_ADRESS_CGRAM;
509  for(i=0; i<8; i++)
510  {
511  lcdi2c_send8(a, LCD_CMD);
512  lcdi2c_send8(c[i], LCD_DATA);
513  a++;
514  };
515 }

543 /* ----------------------------------------------------------------------------
544  ---------- Initialisation du LCD
545  ----------------------------------------------------------------------------
546  This function must be called before any other function.
547  No need to wait between 2 commands because i2c bus is quite slow.
548  cf. Microchip AN587 Interfacing PICmicro® MCUs to an LCD Module
549  ----------------------------------------------------------------------------
550  pcf8574 adress format is [0 1 0 0 A2 A1 A0 0]
551  --------------------------------------------------------------------------*/
552 
553 void lcdi2c_init(u8 numcol, u8 numline, u8 i2c_address)
554 {
555  numcolmax = numcol - 1;
556  numlinemax = numline - 1;
557  PCF8574_address = 0b01001110 | i2c_address;
558  PCF8574_data.val = 0;
559 
560  //I2C_init(I2C1, I2C_MASTER_MODE, I2C_100KHZ);
561  //I2C_init(I2C1, I2C_MASTER_MODE, I2C_400KHZ);
562  I2C_init(I2C1, I2C_MASTER_MODE, I2C_1MHZ);
563 
564  Delayms(15); // Wait more than 15 ms after VDD rises to 4.5V
565  lcdi2c_send4(0x30, LCD_CMD); // 0x30 - Mode 8 bits
566  Delayms(5); // Wait for more than 4.1 ms
567  lcdi2c_send4(0x30, LCD_CMD); // 0x30 - Mode 8 bits
568  Delayus(100); // Wait more than 100 μs
569  lcdi2c_send4(0x30, LCD_CMD); // 0x30 - Mode 8 bits
570  Delayus(100); // Wait more than 100 μs
571  lcdi2c_send4(0x20, LCD_CMD); // 0x20 - Mode 4 bits
572  lcdi2c_send8(LCD_SYSTEM_SET_4BITS, LCD_CMD);// 0x28 - Mode 4 bits - 2 Lignes - 5x8
573  lcdi2c_send8(LCD_ENTRY_MODE_SET, LCD_CMD); // 0x06 - Increment + Display not shifted (Déplacement automatique du curseur)

577  lcdi2c_noBlink();

579  lcdi2c_noAutoscroll();
580  lcdi2c_setCursor(0, 0);
581 */

583  lcdi2c_home();
