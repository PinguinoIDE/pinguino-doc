19 // The Ip checksum is calculated over the ip header only starting
20 // with the header length field and a total length of 20 bytes
21 // unitl ip.dst
22 // You must set the IP checksum field to zero before you start
23 // the calculation.
24 // len for ip is 20.
25 //
26 // For UDP/TCP we do not make up the required pseudo header. Instead we
27 // use the ip.src and ip.dst fields of the real packet:
28 // The udp checksum calculation starts with the ip.src field
29 // Ip.src=4bytes,Ip.dst=4 bytes,Udp header=8bytes + data length=16+len
30 // You must set the upd checksum field to zero before you start
31 // the calculation.
32 // len for udp is: 8 + 8 + data length
33 //
34 // For more information on how this algorithm works see:
35 // http://www.netfor2.com/checksum.html
36 // http://www.msc.uky.edu/ken/cs471/notes/chap3.htm
37 // The RFC has also a C code example: http://www.faqs.org/rfcs/rfc1071.html
38 uint16_t checksum(uint8_t *buf, uint16_t len,uint8_t type){
39  // type 0=ip
40  // 1=udp
41  // 2=tcp
42  uint32_t sum = 0;
43 
44  if(type==1){
45  sum+=IP_PROTO_UDP_V; // protocol udp
46  // the length here is the length of udp (data+header len)
47  // =length - IP addr length
48  sum+=len-8; // = real udp len
49  }
50  if(type==2){
51  sum+=IP_PROTO_TCP_V;
52  // the length here is the length of tcp (data+header len)
53  // =length - IP addr length
54  sum+=len-8; // = real tcp len
55  }
56  // build the sum of 16bit words
57  while(len >1){
58  sum += 0xFFFF & (*buf<<8|*(buf+1));
59  buf+=2;
60  len-=2;
61  }
62  // if there is a byte left then add it (padded with zero)
63  if (len){
64  sum += (0xFF & *buf)<<8;
65  }
66  // now calculate the sum over the bytes in the sum
67  // until the result is only 16bit long
68  while (sum>>16){
69  sum = (sum & 0xFFFF)+(sum >> 16);
70  }
71  // build 1's complement:
72  return( (uint16_t) sum ^ 0xFFFF);
73 }
