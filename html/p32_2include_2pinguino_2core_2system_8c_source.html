260 void SystemClocksCalcCpuClockSettings(SystemClocksSettings *s,
261  uint32_t cpuFrequency)
262 {
263  if (cpuFrequency <= CPUCOREMAXFREQUENCY)
264  {
265  #ifdef EXTENDED_TEST_CASE
266  s->error = Error_None;
267  #endif
268  }
269  else
270  {
271  cpuFrequency = CPUCOREMAXFREQUENCY;
272 
273  #ifdef EXTENDED_TEST_CASE
274  s->error = Error_RequestedFrequencyTooHigh;
275  #endif
276  }
277 
278  uint8_t pllOutputDivIndex;
279  for (pllOutputDivIndex = 0;
280  pllOutputDivIndex < pllOutputDivsCount;
281  pllOutputDivIndex++)
282  {
283  uint8_t pllMulIndex;
284  for (pllMulIndex = 0; pllMulIndex < pllMulsCount; pllMulIndex++)
285  {
286  if ( (cpuFrequency
287  * pllInputDivs[s->PLLIDIV]
288  * pllOutputDivs[pllOutputDivIndex])
289  == (CRYSTALFREQUENCY * pllMuls[pllMulIndex]) )
290  {
291  s->PLLODIV = pllOutputDivIndex;
292  s->PLLMULT = pllMulIndex;
293  // Match found
294  return;
295  }
296  }
297  }
298 
299  /*
300  * No combination found -> try to get max CPU frequency. This
301  * depends from pllInputDiv.
302  *
303  * The selected values will result in the max CPU frequency, if
304  * the pllInputDiv is 1 or 2.
305  *
306  * With pllInputDiv > 2 we cannot reach the max frequency.
307  */
308  if (pllInputDivs[s->PLLIDIV] >= 2)
309  {
310  s->PLLODIV = PLLODIV1; // /1
311 
312  #ifdef EXTENDED_TEST_CASE
313  s->error = Error_NoCombinationFound;
314  #endif
315  }
316  else
317  {
318  s->PLLODIV = PLLODIV2; // /2
319 
320  #ifdef EXTENDED_TEST_CASE
321  s->error = Error_NoCombinationFound;
322  #endif
323  }
324 
325  s->PLLMULT = PLLMULT20; // x20
326 }

339 void SystemClocksCalcPeripheralClockSettings(SystemClocksSettings *s,
340  uint32_t peripheralFrequency)
341 {
342  #ifdef EXTENDED_TEST_CASE
343  s->error = Error_None;
344  #endif
345 
346  if (peripheralFrequency > CPUCOREMAXFREQUENCY)
347  {
348  peripheralFrequency = CPUCOREMAXFREQUENCY;
349 
350  #ifdef EXTENDED_TEST_CASE
351  s->error = Error_RequestedFrequencyTooHigh;
352  #endif
353  }
354 
355  const uint32_t cpuFrequency = SystemClocksGetCpuFrequency(s);
356 
357  uint8_t i;
358  for (i = 0; i < pbDivsCount; i++)
359  {
360  if (cpuFrequency == peripheralFrequency * pbDivs[i])
361  {
362  s->PBDIV = i;
363  // Match found
364  return;
365  }
366  }
367 
368  // No match: Use default value
369  s->PBDIV = PBDIV1; // /1
370  #ifdef EXTENDED_TEST_CASE
371  s->error = Error_NoCombinationFound;
372  #endif
373 }

388 void SystemClocksWriteSettings(const SystemClocksSettings *s)
389 {
390  SystemUnlock();
391 
402  /*
403  * bit 10-8 NOSC<2:0>: New Oscillator Selection bits
404  * 011 = Primary Oscillator with PLL module (XTPLL, HSPLL or ECPLL)
405  */
406  OSCCONbits.NOSC = POSCPLL;
407 
408  /*
409  * bit 29-27 PLLODIV<2:0>: Output Divider for PLL
410  */
411  OSCCONbits.PLLODIV = s->PLLODIV;
412 
413  /*
414  * bit 18-16 PLLMULT<2:0>: PLL Multiplier bits
415  */
416  OSCCONbits.PLLMULT = s->PLLMULT;
417 
418  /*
419  * bit 0 OSWEN: Oscillator Switch Enable bit
420  * 1 = Initiate an osc switch to selection specified by NOSC2:NOSC0 bits
421  * 0 = Oscillator switch is complete
422  */
423  OSCCONbits.OSWEN = 1;
424  // Busy wait until osc switch has been completed
425  while (OSCCONbits.OSWEN == 1)
426  {
427  asm("nop");
428  }
429 
430  /*
431  * bit 20-19 PBDIV<1:0>: Peripheral Bus Clock Divisor
432  */
433  OSCCONbits.PBDIV = s->PBDIV;
434 
435  // Set wait states
436  #if defined(PIC32_PINGUINO_220)||defined(PINGUINO32MX250)||defined(PINGUINO32MX220)
437  // TODO
438  //PMMODEbits.WAITB = 0b00; // Data wait of 1 TPB
439  #else
440  CHECON = (SystemClocksGetCpuFrequency(s) / 20) - 1; // FlashWaitStates
441  #endif
442 
443  //
444  SystemLock();
445 }
