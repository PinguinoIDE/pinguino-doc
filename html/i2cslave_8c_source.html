68 void i2c_hw_slave_write_i2c(u8 what)
69  // wait 'til buffer is empty
70 { u8 dosend;

174 void I2C_interrupt()
175 {
176 u8 tmpstat;
177 u8 _trash;
178 u8 rcv;
179 LA7=1;
180  if (! PIR1bits.SSPIF)
181  return;
182 
183  PIR1bits.SSPIF = 0;
184  tmpstat = SSPSTAT;
185 
186  // check for overflow
187  if (SSPCON1bits.SSPOV)
188  {
189  _trash = i2c_hw_slave_read_i2c();
190  _trash = i2c_hw_slave_read_i2c();
191  SSPCON1bits.SSPOV = 0;
192  }
193 
194  // start & stop handlers are necessary
195 
196  if ((tmpstat & 0b00001111) == 0b00001000)
197  {
198  i2c_hw_slave_on_start();
199  return;
200  }
201  if ((tmpstat & 0b00110000) == 0b00110000)
202  {
203  i2c_hw_slave_on_stop();
204  return;
205  }
206 
207  // mask out unimportant bit
208  tmpstat = tmpstat & 0b00101101;
209 
210  // State machine for PIC18f
211 
212  // slightly modified statemachine for PIC18F devices
213  // see AN734 revision B for additional details
214 
215  // state 1: write operation, last byte is address, buffer full
216  // byte is an address, it we get here, we just know master
217  // wants to talk to us, in a new write operation...
218  // and we also know address is recognized (BF is set, see spec)
219  // anyway, we must read buffer to reset BF bit
220  // (buffer actually contains previously detected address, that's
221  // why it's usually ignored)
222  //
223  // => call i2c_hw_slave_on_state_1(u8)
224  if (!(tmpstat ^ 0b00001001))
225  {
226  _trash = i2c_hw_slave_read_i2c();
227  i2c_hw_slave_on_state_1(_trash);
228  }
229  // state 2: write operation, last byte is data, buffer full
230  // master wants to write, it sends a byte
231  //
232  // => call i2c_hw_slave_on_state_2(u8)
233  //
234  // (note: at a write-only sequence, fast masters give a stop (clearing bit3)
235  // before we can read the register. The or statment is handles this)
236  else if (!((tmpstat | 0b00001000) ^ 0b00101001))
237  {
238  rcv = i2c_hw_slave_read_i2c();
239  i2c_hw_slave_on_state_2(rcv);
240  }
241  // state 3: read operation, last byte is address, buffer empty
242  // master wants to get a value from us
243  //
244  // => call i2c_hw_slave_on_state_3()
245  // Mask BF flag for newer statemachines
246  else if (!((tmpstat & 0b11111110) ^ 0b00001100))
247  {
248  // clear BF flag by clearing SSPBUF, required for PIC18f devices
249  // with a new version of the i2c statemachine (see also AN734)
250  _trash = i2c_hw_slave_read_i2c();
251  i2c_hw_slave_on_state_3();
252 
253  }
254  else if (!((tmpstat & 0b11111011) ^ 0b00101000))
255  {
256 
257  // check CKP bit to distinguish between state 4 and 5
258  if ( SSPCON1bits.CKP == 0 )
259 
260  // state 4: read operation, last byte is data, buffer empty
261  // master still wants to get a value from us
262  //
263  // => call i2c_hw_slave_on_state_4()
264  // Don't check R/W flag, use CKP bit instead
265  i2c_hw_slave_on_state_4();
266 
267  else
268 
269  // state 5: nack
270  // master doesn't want to talk with us anymore
271  //
272  // => call i2c_hw_slave_on_state_5()
273  // AN734 does not talk about setting CKP, whereas spec says
274  // it must be set. Some people say it can be error prone.
275  // SSPCON1bits.CKP = 1
276  i2c_hw_slave_on_state_5();
277 
278  }
279  // state not recognized... Error !
280  else
281  // something went wrong, that is, XOR operations did not match
282  // SSPSTAT bits
283  i2c_hw_slave_on_error();
284
