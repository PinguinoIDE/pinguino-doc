51 //IOCON Flag
52 //Unimplemented: Read as ‘0’.
53 #define MCP23017_B_NotUse 1
54 //INTPOL: This bit sets the polarity of the INT output pin.
55 //1 = Active-high.
56 //0 = Active-low.
57 #define MCP23017_B_INTPOL 2
58 //ODR: This bit configures the INT pin as an open-drain output.
59 //1 = Open-drain output (overrides the INTPOL bit).
60 //0 = Active driver output (INTPOL bit sets the polarity).
61 #define MCP23017_B_ODR 4
62 //HAEN: Hardware Address Enable bit (MCP23S17 only).
63 //Address pins are always enabled on MCP23017.
64 //1 = Enables the MCP23017 address pins.
65 //0 = Disables the MCP23017 address pins.
66 #define MCP23017_B_HAEN 8
67 //DISSLW: Slew Rate control bit for SDA output.
68 //1 = Slew rate disabled.
69 //0 = Slew rate enabled.
70 #define MCP23017_B_DISSLW 16
71 //SEQOP: Sequential Operation mode bit.
72 //1 = Sequential operation disabled, address pointer does not increment.
73 //0 = Sequential operation enabled, address pointer increments.
74 #define MCP23017_B_SEQOP 32
75 //MIRROR: INT Pins Mirror bit
76 //1 = The INT pins are internally connected
77 //0 = The INT pins are not connected. INTA is associated with PortA and INTB is associated with PortB
78 #define MCP23017_B_MIRROR 64
79 //BANK: Controls how the registers are addressed
80 //1 = The registers associated with each port are separated into different banks
81 //0 = The registers are in the same bank (addresses are sequential)

636 /***********************************************************************************
637 BOOL MCP23017_pinMode(u8 Pin, u8 OutIn,MCP23017 * Mcp)
638 
639 Define if a pin is an Inpur or a Output
640 
641 Pin = Pin on which operate (from 1 to 16)
642 OutIn = (INPUT = Input Pin, OUTPUT = Output Pin)
643 Mcp = Pointer to a user declared MCP23017 structure
644 
645 RETURN FALSE on Error
646 ***********************************************************************************/
647 BOOL MCP23017_pinMode(u8 Pin, u8 OutIn,MCP23017 * Mcp)
648 {
649  Pin--;
650  if (Pin > 15)
651  return(FALSE);
652  if (Pin < 8)
653  {
654  if (OutIn == INPUT)
655  Mcp->IODIR.B[0] |= 1 << Pin;
656  else
657  Mcp->IODIR.B[0] &= ~(1 << Pin);
658  if (MCP23017_write_B(Mcp,MCP23017_IODIRA,Mcp->IODIR.B[0]) == FALSE)
659  return(FALSE);
660  }
661  else
662  {
663  Pin -= 8;
664  if (OutIn == INPUT)
665  Mcp->IODIR.B[1] |= 1 << Pin;
666  else
667  Mcp->IODIR.B[1] &= ~(1 << Pin);
668  if (MCP23017_write_B(Mcp,MCP23017_IODIRB,Mcp->IODIR.B[1]) == FALSE)
669  return(FALSE);
670  }
671  return (TRUE);
