196 void SPI_init(u32 word,u32 speed)
197 {
198  unsigned char rData;
199  u32 Fpb;
200  u16 clk;
201  Fpb = GetPeripheralClock();
202  // 1. Disable the SPI interrupts in the respective IEC0/1 register.
203  IntDisable(INTFAULT);
204  IntDisable(INTTXDONE);
205  IntDisable(INTRXDONE);
206  // 2. Stop and reset the SPI module by clearing the ON bit.
207  SPICONCLR = 0x8000; // bit 15
208  // 3. Clear the receive buffer.
209  rData = BUFFER;
210  // 4. Set speed rate
211  clk = (Fpb/(2*speed))-1;
212  CLKSPD = clk;
213  // 5. Swtich on SPI module with Options added
214  SPICONSET = 0x8000 | word;
215 }

278 u32 SPI_setClock(u32 speed)
279 {
280  u32 Fpb;
281  u16 clk;

283  Fpb = GetPeripheralClock();
284  if (speed > (Fpb / 2))

366 void SPI_begin()
367 {
368  unsigned char rData;
369 
370  // 1. Disable the SPI interrupts in the respective IEC0/1 register.
371  IntDisable(INTFAULT);
372  IntDisable(INTTXDONE);
373  IntDisable(INTRXDONE);
374 
375  // 2. Stop and reset the SPI module by clearing the ON bit.
376  SPICONF = 0;
377 
378  // 3. Clear the receive buffer.
379  rData = BUFFER;
380 
381  // 4. Clear the ENHBUF bit (SPIxCON<16>) if using Standard Buffer mode.
382  // This bit can only be written when the ON bit = 0
383  // SPIENHBUF = 0; // not available on all devices
384 
385  // 5. If SPI interrupts are not going to be used, skip this step and
386  // continue to step 6. Otherwise the following additional steps are performed:
387  // a) Clear the SPIx interrupt flags/events in the respective IFS0/1 register.
388 /*
389  IntClearFlag(INTFAULT);
390  IntClearFlag(INTTXDONE);
391  IntClearFlag(INTRXDONE);
392 
393  // b) Set the SPIx interrupt enable bits in the respective IEC0/1 register.
394  IntEnable(INTFAULT);
395  IntEnable(INTTXDONE);
396  IntEnable(INTRXDONE);
397 
398  // c) Write the SPIx interrupt priority and subpriority bits in the respective IPC5/7 register.
399  IntSetVectorPriority(INTVECTOR, 3, 1);
400 */
401  // 6. Write the Baud Rate register, SPIxBRG.
402  CLKSPD = this_clock; // Default SPI_PBCLOCK_DIV64
403 
404  // 7. Clear the SPIROV bit (SPIxSTAT<6>).
405  STATUS = 0; // clear the Overflow
406 
407  // 8. Write the desired settings to the SPIxCON register.
408 
409  switch (this_mode)
410  {
411  case SPI_MODE0: // SPI bus mode 0,0
412  CKP = 0;
413  CKE = 0; // data transmitted on falling edge
414  break;
415  case SPI_MODE1: // default SPI bus mode 0,1
416  CKP = 0;
417  CKE = 1; // data transmitted on falling edge
418  break;
419  case SPI_MODE2: // SPI bus mode 1,0
420  CKP = 1; // clock idle state high
421  CKE = 0; // data transmitted on rising edge
422  break;
423  case SPI_MODE3: // SPI bus mode 1,1
424  CKP = 1; // clock idle state high
425  CKE = 1; // data transmitted on falling edge
426  break;
427  }
428 
429  SMP = this_phase; // Input data sampled at end of middle of data output time
430  MSTEN = this_role; // Master or Slave mode
431 
432  // The SSx pin is not driven by the SPI Master.
433  // User have to drive the bit himself and pulse it before the SPI
434  // transmission takes place.
435  // He can therefore use any other I/O pin for that purpose,
436  // but we provide one ;-)
437 
438  if (this_role != SPI_SLAVE)
439  {
440  #if defined(PINGUINO32MX250) || defined(PINGUINO32MX220)
441  TRISBCLR = 1<<7; // RB7 is defined as SS1 pin (cf. io.c)
442  #endif
443  }
444 
445  // 9. Enable SPI operation by setting the ON bit (SPIxCON<15>).
446  SPICONSET = 0x8000; // Set bit 15
447 }
