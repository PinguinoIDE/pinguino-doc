35 /* Compute() **********************************************************************
36 * This, as they say, is where the magic happens. this function should be called
37 * every time "void loop()" executes. the function will decide for itself whether a new
38 * pid Output needs to be computed
39 **********************************************************************************/
40 void PID_Compute()
41 {
42  unsigned long now, timeChange;
43  double input, error, dInput, output;
44 
45  if(!inAuto) return;
46 
47  now = millis();
48  timeChange = (now - lastTime);
49 
50  if(timeChange>=SampleTime)
51  {
52  /*Compute all the working error variables*/
53  input = *myInput;
54  error = *mySetpoint - input;
55  ITerm+= (ki * error);
56  if(ITerm > outMax) ITerm= outMax;
57  else if(ITerm < outMin) ITerm= outMin;
58  dInput = (input - lastInput);
59 
60  /*Compute PID Output*/
61  output = kp * error + ITerm- kd * dInput;
62 
63  if(output > outMax) output = outMax;
64  else if(output < outMin) output = outMin;
65  *myOutput = output;
66 
67  /*Remember some variables for next time*/
68  lastInput = input;
69  lastTime = now;
70  }
71 }

115 /* SetOutputLimits(...)****************************************************
116 * This function will be used far more often than SetInputLimits. while

166 /* SetControllerDirection(...)*************************************************
167 * The PID will either be connected to a DIRECT acting process (+Output leads
168 * to +Input) or a REVERSE acting process(+Output leads to -Input.) we need to
