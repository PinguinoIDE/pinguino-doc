188 #if defined(RTCCSETALARMTIME) || defined(RTCCSETALARMTIMEDATE)
189 void RTCC_SetAlarmTime(u32 alTime)
190 {
191  rtccTime t0;
192  u8 dummy;
193 
194  t0.l = alTime;
195  RTCC_SetWriteEnable();
196  RTCC_Wait();
197  ALRMCFG = 0x00; // clear the ALRMEN, CHIME, AMASK and ARPT;
198  ALRMRPT = 0; // clear Alarm repeat counter
199  ALRMCFGbits.ALRMPTR1 = 0;
200  ALRMCFGbits.ALRMPTR0 = 1;
201  // ALRMPTR = 01 = Weekday/Hours
202  ALRMVALL = t0.hours;
203  dummy = ALRMVALH; // dummy read of RTCVALH to auto-decrement RTCPTR
204  // ALRMPTR = 00 = Minutes/Seconds
205  ALRMVALL = t0.seconds;
206  ALRMVALH = t0.minutes;
207  RTCC_SetWriteDisable();
208 }
209 #endif
