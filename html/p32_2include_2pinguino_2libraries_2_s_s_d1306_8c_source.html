497 {

504  // Dummy byte (???)

506  // Start page
507  SSD1306_sendCommand(start & DISPLAY_ROW_MASK);
508 
509  // Time interval between each scroll step

643 void SSD1306_printChar(u8 c)
644 {
645  u8 b;
646 
647  while (SSD1306.cursor.x >= (DISPLAY_WIDTH / SSD1306.font.width))
648  {
649  SSD1306.cursor.x -= (DISPLAY_WIDTH / SSD1306.font.width);
650  SSD1306.cursor.y++;
651  }
652 
653  while (SSD1306.cursor.y > 7)
654  {
655  SSD1306_scrollUp();
656  }
657 
658  switch (c)
659  {
660  case '\n':
661  SSD1306.cursor.y++;
662  break;
663 
664  case '\r':
665  SSD1306.cursor.x = 0;
666  break;
667 
668  case '\t':
669  SSD1306.cursor.x = (SSD1306.cursor.x + 4) % 4;
670  break;
671 
672  default:
673  for (b = 0; b < SSD1306.font.width; b++)
674  {
675  SSD1306_buffer[SSD1306.cursor.x * SSD1306.font.width + SSD1306.cursor.y * 128 + b] =
676  SSD1306.font.address[2 + (c - 32) * SSD1306.font.width + b];
677  }
678 
679  SSD1306.cursor.x++;
680  }
681 }
