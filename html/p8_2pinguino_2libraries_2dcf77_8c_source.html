57 /*
58  Timer Configuration
59  ex. 1 tick every TMR_OVERLOAD = 20ms
60  The timer clock prescale (TCKPS) is 1:8
61  The TMR Count register increments on every FOSC/4 cycle
62  FOSC/4 = 48/4 = 12 MHz => TMR inc. every 8*1/12 MHZ = 2/3 us
63  20 ms = 20.000 us => 20.000 / (2 / 3) = 20 * 1500 = 30.000 cycles = 0x7530
64 */

70 #define TMR_ONESEC ((1000/TMR_OVERLOAD) - 1)

123 /*******************************************************************************
124 * Calculate even parity
125 *******************************************************************************/
126 
127 u8 Parity_Even (u32 Bits)
128 {
129  u8 i, p = 0;
130 
131  for (i = 0; i < 32; i++)
132  if ((Bits & (1 << i)) > 0)
133  p = !p;
134 
135  return p;
136 }

142 void Decode_Buffer(void)
143 {
144  u32 Code;
145  u8 Par, Par1, Par2, Par3;
146  // Disp_Input_Binary(Buff1, Buff2);
147 
148  if (BuffPntr == 59)
149  {
150  // Minutes
151  Code = (Buff1 >> 21) & 0x7F;
152  Par = Parity_Even(Code);
153  Par1 = (Buff1 >> 28) & 1;
154 // debug("Par= %d Par1= %d\r\n", Par, Par1);
155  if (Par == Par1)
156  {
157  Par1 = 11;
158  DCF77.minutes = bcd2bin(Code);
159 // debug("mm= %02d \r\n", DCF77.mm);
160  }
161 
162  // Hours
163  Code = ((Buff1 >> 29) & 0x7) | ((Buff2 & 0x7) << 3);
164  Par = Parity_Even(Code);
165  Par2 = (Buff2 >> 3) & 1;
166 // debug("Par= %d Par2= %d\r\n", Par, Par2);
167  if (Par == Par2)
168  {
169  Par2 = 12;
170  DCF77.hours = bcd2bin(Code);
171 // debug("hh= %02d \r\n", DCF77.hh);
172  }
173 
174  Code = (Buff2 >> 4) & 0x3FFFFF;
175  Par = Parity_Even(Code);
176  Par3 = (Buff2 >> 26) & 1;
177 // debug("Par= %d Par3= %d\r\n", Par, Par3);
178  if (Par == Par3)
179  {
180  Par3 = 13;
181  Code = (Buff2 >> 10) & 0x7;
182  DCF77.dayofweek = bcd2bin(Code);
183 // debug("dw= %02d \r\n", DCF77.dw);
184 
185  Code = (Buff2 >> 4) & 0x3F;
186  DCF77.dayofmonth = bcd2bin(Code);
187 // debug("da= %02d \r\n", DCF77.da);
188 
189  Code = (Buff2 >> 13) & 0x1F;
190  DCF77.month = bcd2bin(Code);
191 // debug("mo= %02d \r\n", DCF77.mo);
192 
193  Code = (Buff2 >> 18) & 0xFF;
194  DCF77.year = bcd2bin(Code);
195 // debug("yr= %02d \r\n", DCF77.yr);
196  }

200  {
201  DCF77.seconds = 0;

222 /*******************************************************************************
223  * Called every 20 ms by ISR
224  * Builds DCF code by calling DCF77_appendsignal
225  ******************************************************************************/
226 
227 void DCF77_scanSignal(void)
228 {
229  u16 LengthPulse;
230  u16 CurrFlank;
231  u8 CurrSignal;
232 
233  CurrSignal = digitalread(SignalPin);
234  if (CurrSignal != PrevSignal)
235  {
236  if (CurrSignal == 1)
237  {
238 // digitalwrite(USERLED, HIGH);
239  /* At a raising flank */
240  CurrFlank = Cntr_tmr;
241  if ((CurrFlank - PrevFlank) > DCF_Sync_1500ms)
242  Decode_Buffer();
243  PrevFlank = CurrFlank;
244  }
245  else
246  {
247 // digitalwrite(USERLED, LOW);
248  /* At a falling flank */
249  LengthPulse = Cntr_tmr - PrevFlank;
250  if (LengthPulse < DCF_Discr_150ms)
251  DCF77_appendSignal(0);
252  else
253  DCF77_appendSignal(1);
254  }
255  PrevSignal = CurrSignal;
256  }
257 }

263 void Update_Time (void)

275  RTClock.seconds = 0;
276 
277  RTClock.minutes++; // Minutes
278  if (RTClock.minutes > 59)
279  {
280  RTClock.minutes = 0;
281 
282  RTClock.hours++; // Hours
283  if (RTClock.hours > 23)
284  {
285  RTClock.hours = 0;
286 
287  RTClock.dayofweek++; // Day of Week
288  if (RTClock.dayofweek > 7)
289  RTClock.dayofweek = 1;
290 
291  RTClock.dayofmonth++; // Date
292  if (RTClock.dayofmonth > Mnth[RTClock.month-1])
293  {
294  RTClock.dayofmonth = 1;
295 
296  RTClock.month++; // Month
297  if (RTClock.month > 12)
298  {
299  RTClock.month = 1;
300 
301  RTClock.year++; // Year
302  if (RTClock.year > 99)
303  RTClock.year = 0;
