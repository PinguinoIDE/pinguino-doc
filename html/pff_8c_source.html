73 static CLUST get_fat ( /* 1:IO error, Else:Cluster status */
74  CLUST clst /* Cluster# to get the link information */
75 )
76 {
77  u16 wc, bc, ofs;
78  u8 buf[4];
79  FATFS *fs = FatFs;
80 
81  if (clst < 2 || clst >= fs->max_clust) /* Range check */
82  return 1;
83 
84  switch (fs->fs_type) {
85  case FS_FAT12 :
86  bc = (u16)clst; bc += bc / 2;
87  ofs = bc % 512; bc /= 512;
88  if (ofs != 511) {
89  if (disk_readp(buf, fs->fatbase + bc, ofs, 2)) break;
90  } else {
91  if (disk_readp(buf, fs->fatbase + bc, 511, 1)) break;
92  if (disk_readp(buf+1, fs->fatbase + bc + 1, 0, 1)) break;
93  }
94  wc = LD_WORD(buf);
95  return (clst & 1) ? (wc >> 4) : (wc & 0xFFF);
96 
97  case FS_FAT16 :
98  if (disk_readp(buf, fs->fatbase + clst / 256, (u16)(((u16)clst % 256) * 2), 2)) break;
99  return LD_WORD(buf);
100 #if _FS_FAT32
101  case FS_FAT32 :
102  if (disk_readp(buf, fs->fatbase + clst / 128, (u16)(((u16)clst % 128) * 4), 4)) break;
103  return LD_DWORD(buf) & 0x0FFFFFFF;
104 #endif
105  }
106 
107  return 1; /* An error occured at the disk I/O layer */
108 }

163 static FRESULT dir_next ( /* FR_OK:Succeeded, FR_NO_FILE:End of table */
164  DIR *dj /* Pointer to directory object */
165 )
166 {
167  CLUST clst;
168  u16 i;
169  FATFS *fs = FatFs;
170 
171  i = dj->index + 1;
172  if (!i || !dj->sect) /* Report EOT when index has reached 65535 */
173  return FR_NO_FILE;
174 
175  if (!(i & (16-1))) { /* Sector changed? */
176  dj->sect++; /* Next sector */
177 
178  if (dj->clust == 0) { /* Static table */
179  if (i >= fs->n_rootdir) /* Report EOT when end of table */
180  return FR_NO_FILE;
181  }
182  else { /* Dynamic table */
183  if (((i / 16) & (fs->csize-1)) == 0) { /* Cluster changed? */
184  clst = get_fat(dj->clust); /* Get next cluster */
185  if (clst <= 1) return FR_DISK_ERR;
186  if (clst >= fs->max_clust) /* When it reached end of dynamic table */
187  return FR_NO_FILE; /* Report EOT */
188  dj->clust = clst; /* Initialize data for new cluster */
189  dj->sect = clust2sect(clst);
190  }
191  }
192  }
193 
194  dj->index = i;
195 
196  return FR_OK;
197 }

380 static FRESULT follow_path ( /* FR_OK(0): successful, !=0: error code */
381  DIR *dj, /* Directory object to return last directory and found object */
382  const char *path /* Full-path string to find a file or directory */
383 )

390 
391  while (*path == ' ') path++; /* Skip leading spaces */
392  if (*path == '/') path++; /* Strip heading separator */
393  dj->sclust = 0; /* Set start directory (always root dir) */
394 
395 // memcpypgm2ram(t, path, 15);
396 
397  if ((u8)*path <= ' ') { /* Null path means the root directory */
398  res = dir_rewind(dj);
399  FatFs->buf[0] = 0;

401  } else { /* Follow path */
402  for (;;) {
403  res = create_name(dj, &path); /* Get a segment */
404  if (res != FR_OK) break;

408  res = FR_NO_PATH;
409  break;
410  }
411  if (*(dj->fn+11)) break; /* Last segment match. Function completed. */
412  dir = FatFs->buf; /* There is next segment. Follow the sub directory */
413  if (!(dir[DIR_Attr] & AM_DIR)) { /* Cannot follow because it is a file */
414  res = FR_NO_PATH; break;
415  }
416  dj->sclust =
417 #if _FS_FAT32
418  ((u32)LD_WORD(dir+DIR_FstClusHI) << 16) |
419 #endif
420  LD_WORD(dir+DIR_FstClusLO);
421  }
422  }

434 static u8 check_fs ( /* 0:The FAT boot record, 1:Valid boot record but not an FAT, 2:Not a boot record, 3:Error */

470 FRESULT pf_mount (
471  FATFS *fs /* Pointer to new file system object (NULL: Unmount) */
472 )
473 {
474  u8 fmt, buf[36];
475  u32 bsect, fsize, tsect, mclst;
476  DRESULT Dresult;
477  Dresult = disk_initialize();
478  if(Dresult != RES_OK)
479  return FR_DISK_ERR;
480  FatFs = 0;
481  if (!fs) return FR_OK; /* Unregister fs object */
482 
483  if (disk_initialize() & STA_NOINIT){ /* Check if the drive is ready or not */
484  return FR_NOT_READY;
485  }
486  /* Search FAT partition on the drive */
487  bsect = 0;
488  fmt = check_fs(buf, bsect); /* Check sector 0 as an SFD format */
489  if (fmt == 1) { /* Not an FAT boot record, it may be FDISK format */
490  /* Check a partition listed in top of the partition table */
491  if (disk_readp(buf, bsect, MBR_Table, 16)) { /* 1st partition entry */
492  fmt = 3;
493  } else {
494  if (buf[4]) { /* Is the partition existing? */
495  bsect = LD_DWORD(&buf[8]); /* Partition offset in LBA */
496  fmt = check_fs(buf, bsect); /* Check the partition */
497  }
498  }
499  }
500  if (fmt == 3) return FR_DISK_ERR;
501  if (fmt) return FR_NO_FILESYSTEM; /* No valid FAT patition is found */
502  /* Initialize the file system object */
503  if (disk_readp(buf, bsect, 13, sizeof(buf))) return FR_DISK_ERR;
504 
505  fsize = LD_WORD(buf+BPB_FATSz16-13); /* Number of sectors per FAT */
506  if (!fsize) fsize = LD_DWORD(buf+BPB_FATSz32-13);
507 
508  fsize *= buf[BPB_NumFATs-13]; /* Number of sectors in FAT area */
509  fs->fatbase = bsect + LD_WORD(buf+BPB_RsvdSecCnt-13); /* FAT start sector (lba) */
510  fs->csize = buf[BPB_SecPerClus-13]; /* Number of sectors per cluster */
511  fs->n_rootdir = LD_WORD(buf+BPB_RootEntCnt-13); /* Nmuber of root directory entries */
512  tsect = LD_WORD(buf+BPB_TotSec16-13); /* Number of sectors on the file system */
513  if (!tsect) tsect = LD_DWORD(buf+BPB_TotSec32-13);
514  mclst = (tsect /* Last cluster# + 1 */
515  - LD_WORD(buf+BPB_RsvdSecCnt-13) - fsize - fs->n_rootdir / 16
516  ) / fs->csize + 2;
517  fs->max_clust = (CLUST)mclst;
518 
519  fmt = FS_FAT12; /* Determine the FAT sub type */
520  if (mclst >= 0xFF7) fmt = FS_FAT16; /* Number of clusters >= 0xFF5 */
521  if (mclst >= 0xFFF7) /* Number of clusters >= 0xFFF5 */
522 #if _FS_FAT32
523  fmt = FS_FAT32;
524 #else
525  return FR_NO_FILESYSTEM;
526 #endif
527 
528  fs->fs_type = fmt; /* FAT sub-type */
529 #if _FS_FAT32
530  if (fmt == FS_FAT32)
531  fs->dirbase = LD_DWORD(buf+(BPB_RootClus-13)); /* Root directory start cluster */
532  else
533 #endif
534  fs->dirbase = fs->fatbase + fsize; /* Root directory start sector (lba) */
535  fs->database = fs->fatbase + fsize + fs->n_rootdir / 16; /* Data start sector (lba) */
536 
537  fs->flag = 0;
538  FatFs = fs;
539 
540  return FR_OK;
541 }

610

613  if ((fs->fptr / 512 % fs->csize) == 0) { /* On the cluster boundary? */

629  btr -= rcnt; *br += rcnt;
630  }

636  return FR_DISK_ERR;

674  while (btw) { /* Repeat until all data transferred */
675  if (((u16)fs->fptr % 512) == 0) { /* On the sector boundary? */
676  if ((fs->fptr / 512 % fs->csize) == 0) { /* On the cluster boundary? */
677  clst = (fs->fptr == 0) ? /* On the top of the file? */
678  fs->org_clust : get_fat(fs->curr_clust);
679  if (clst <= 1) goto fw_abort;
680  fs->curr_clust = clst; /* Update current cluster */
681  fs->csect = 0; /* Reset sector offset in the cluster */
682  }
683  sect = clust2sect(fs->curr_clust); /* Get current sector */
684  if (!sect) goto fw_abort;
685  fs->dsect = sect + fs->csect++;
686  if (disk_writep(0, fs->dsect)) goto fw_abort; /* Initiate a sector write operation */
687  fs->flag |= FA__WIP;
688  }
689  wcnt = 512 - ((u16)fs->fptr % 512); /* Number of bytes to write to the sector */
690  if (wcnt > btw) wcnt = btw;
691  if (disk_writep(p, wcnt)) goto fw_abort; /* Send data to the sector */
692  fs->fptr += wcnt; p += wcnt; /* Update pointers and counters */
693  btw -= wcnt; *bw += wcnt;
694  if (((u16)fs->fptr % 512) == 0) {
695  if (disk_writep(0, 0)) goto fw_abort; /* Finalize the currtent secter write operation */
696  fs->flag &= ~FA__WIP;
697  }
698  }
699 
700  return FR_OK;
701 
702 fw_abort:
703  fs->flag = 0;
704  return FR_DISK_ERR;

734  fs->fptr = (ifptr - 1) & ~(bcs - 1); /* start from the current cluster */
735  ofs -= fs->fptr;
736  clst = fs->curr_clust;
737  } else { /* When seek to back cluster, */
738  clst = fs->org_clust; /* start from the first cluster */
739  fs->curr_clust = clst;

744  fs->curr_clust = clst;

746  ofs -= bcs;
747  }
748  fs->fptr += ofs;

751  fs->csect = (u8)(ofs / 512); /* Sector offset in the cluster */
752  if (ofs % 512)
