32 uint8_t enc28j60ReadOp(uint8_t op, uint8_t address)
33 {
34 uint8_t received_byte;
35 
36  CSACTIVE;
37  // issue read command
38  SPI_write(op | (address & ADDR_MASK));
39  // read data

42  if(address & 0x80)

46  // release CS

159 void enc28j60Init(u8* macaddr)
160 {
161  // initialize I/O
162  // ss as output:
163  pinmode(ENC28J60_CONTROL_CS, OUTPUT);
164  CSPASSIVE; // ss=0
165  // init SPI module
166  SPI_init();
167 
168 
169  // perform system reset
170  enc28j60WriteOp(ENC28J60_SOFT_RESET, 0, ENC28J60_SOFT_RESET);
171  Delayms(50);
172  // check CLKRDY bit to see if reset is complete
173  // The CLKRDY does not work. See Rev. B4 Silicon Errata point. Just wait.
174  //while(!(enc28j60Read(ESTAT) & ESTAT_CLKRDY));
175  // do bank 0 stuff
176  // initialize receive buffer
177  // 16-bit transfers, must write low byte first
178  // set receive buffer start address
179  NextPacketPtr = RXSTART_INIT;
180  // Rx start
181  enc28j60Write(ERXSTL, RXSTART_INIT&0xFF);
182  enc28j60Write(ERXSTH, RXSTART_INIT>>8);
183  // set receive pointer address
184  enc28j60Write(ERXRDPTL, RXSTART_INIT&0xFF);
185  enc28j60Write(ERXRDPTH, RXSTART_INIT>>8);
186  // RX end
187  enc28j60Write(ERXNDL, RXSTOP_INIT&0xFF);
188  enc28j60Write(ERXNDH, RXSTOP_INIT>>8);
189  // TX start
190  enc28j60Write(ETXSTL, TXSTART_INIT&0xFF);
191  enc28j60Write(ETXSTH, TXSTART_INIT>>8);
192  // TX end
193  enc28j60Write(ETXNDL, TXSTOP_INIT&0xFF);
194  enc28j60Write(ETXNDH, TXSTOP_INIT>>8);
195  // do bank 1 stuff, packet filter:
196  // For broadcast packets we allow only ARP packtets
197  // All other packets should be unicast only for our mac (MAADR)
198  //
199  // The pattern to match on is therefore
200  // Type ETH.DST
201  // ARP BROADCAST
202  // 06 08 -- ff ff ff ff ff ff -> ip checksum for theses bytes=f7f9
203  // in binary these poitions are:11 0000 0011 1111
204  // This is hex 303F->EPMM0=0x3f,EPMM1=0x30
205  //enc28j60Write(ERXFCON, ERXFCON_UCEN|ERXFCON_CRCEN|ERXFCON_PMEN);
206  enc28j60Write(ERXFCON, ERXFCON_CRCEN);
207  enc28j60Write(EPMM0, 0x3f);
208  enc28j60Write(EPMM1, 0x30);
209  enc28j60Write(EPMCSL, 0xf9);
210  enc28j60Write(EPMCSH, 0xf7);
211  //
212  //
213  // do bank 2 stuff
214  // enable MAC receive
215  enc28j60Write(MACON1, MACON1_MARXEN|MACON1_TXPAUS|MACON1_RXPAUS);
216  // bring MAC out of reset
217  enc28j60Write(MACON2, 0x00);
218  // enable automatic padding to 60bytes and CRC operations
219  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, MACON3, MACON3_PADCFG0|MACON3_TXCRCEN|MACON3_FRMLNEN);
220  // set inter-frame gap (non-back-to-back)
221  enc28j60Write(MAIPGL, 0x12);
222  enc28j60Write(MAIPGH, 0x0C);
223  // set inter-frame gap (back-to-back)
224  enc28j60Write(MABBIPG, 0x12);
225  // Set the maximum packet size which the controller will accept
226  // Do not send packets longer than MAX_FRAMELEN:
227  enc28j60Write(MAMXFLL, MAX_FRAMELEN&0xFF);
228  enc28j60Write(MAMXFLH, MAX_FRAMELEN>>8);
229  // do bank 3 stuff
230  // write MAC address
231  // NOTE: MAC address in ENC28J60 is byte-backward
232  enc28j60Write(MAADR5, macaddr[0]);
233  enc28j60Write(MAADR4, macaddr[1]);
234  enc28j60Write(MAADR3, macaddr[2]);
235  enc28j60Write(MAADR2, macaddr[3]);
236  enc28j60Write(MAADR1, macaddr[4]);
237  enc28j60Write(MAADR0, macaddr[5]);
238  // no loopback of transmitted frames
239  enc28j60PhyWrite(PHCON2, PHCON2_HDLDIS);
240  // switch to bank 0
241  enc28j60SetBank(ECON1);
242  // enable interrutps
243  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, EIE, 0);
244  //enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, EIE, EIE_INTIE|EIE_PKTIE);
245  // enable packet reception
246  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, ECON1_RXEN);
247 }

262 void enc28j60PacketSend(uint16_t len, uint8_t* packet)
263 {
264  // Check no transmit in progress
265  while (enc28j60ReadOp(ENC28J60_READ_CTRL_REG, ECON1) & ECON1_TXRTS)
266  {
267  // Reset the transmit logic problem. See Rev. B4 Silicon Errata point 12.
268  //if( (enc28j60Read(EIR) & EIR_TXERIF) )
269  {
270  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, ECON1_TXRST);
271  enc28j60WriteOp(ENC28J60_BIT_FIELD_CLR, ECON1, ECON1_TXRST);
272  }
273  }
274 
275  // Set the write pointer to start of transmit buffer area
276  enc28j60Write(EWRPTL, TXSTART_INIT&0xFF);
277  enc28j60Write(EWRPTH, TXSTART_INIT>>8);
278  // Set the TXND pointer to correspond to the packet size given
279  enc28j60Write(ETXNDL, (TXSTART_INIT+len)&0xFF);
280  enc28j60Write(ETXNDH, (TXSTART_INIT+len)>>8);
281  // write per-packet control byte (0x00 means use macon3 settings)
282  enc28j60WriteOp(ENC28J60_WRITE_BUF_MEM, 0, 0x00);
283  // copy the packet into the transmit buffer
284  enc28j60WriteBuffer(len, packet);
285  // send the contents of the transmit buffer onto the network
286  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON1, ECON1_TXRTS);
287 }

298 // Gets a packet from the network receive buffer, if one is available.
299 // The packet will by headed by an ethernet header.
300 // maxlen The maximum acceptable length of a retrieved packet.
301 // packet Pointer where packet data should be stored.
302 // Returns: Packet length in bytes if a packet was retrieved, zero otherwise.
303 uint16_t enc28j60PacketReceive(uint16_t maxlen, uint8_t* packet)
304 {
305  uint16_t rxstat;
306  uint16_t len;
307  // check if a packet has been received and buffered
308  //if( !(enc28j60Read(EIR) & EIR_PKTIF) ){
309  // The above does not work. See Rev. B4 Silicon Errata point 6.
310  if( enc28j60Read(EPKTCNT) ==0 ){
311  return(0);
312  }
313 
314  // Set the read pointer to the start of the received packet
315  enc28j60Write(ERDPTL, (NextPacketPtr &0xFF));
316  enc28j60Write(ERDPTH, (NextPacketPtr)>>8);
317  // read the next packet pointer
318  NextPacketPtr = enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0);
319  NextPacketPtr |= enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0)<<8;
320  // read the packet length (see datasheet page 43)
321  len = enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0);
322  len |= enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0)<<8;
323  len-=4; //remove the CRC count
324  // read the receive status (see datasheet page 43)
325  rxstat = enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0);
326  rxstat |= ((uint16_t)enc28j60ReadOp(ENC28J60_READ_BUF_MEM, 0))<<8;
327  // limit retrieve length
328  if (len>maxlen-1){
329  len=maxlen-1;
330  }
331  // check CRC and symbol errors (see datasheet page 44, table 7-3):
332  // The ERXFCON.CRCEN is set by default. Normally we should not
333  // need to check this.
334  if ((rxstat & 0x80)==0){
335  // invalid
336  len=0;
337  }else{
338  // copy the packet from the receive buffer
339  enc28j60ReadBuffer(len, packet);
340  }
341  // Move the RX read pointer to the start of the next received packet
342  // This frees the memory we just read out
343  enc28j60Write(ERXRDPTL, (NextPacketPtr &0xFF));
344  enc28j60Write(ERXRDPTH, (NextPacketPtr)>>8);
345  // Move the RX read pointer to the start of the next received packet
346  // This frees the memory we just read out.
347  // However, compensate for the errata point 13, rev B4: enver write an even address!
348  if ((NextPacketPtr - 1 < RXSTART_INIT)
349  || (NextPacketPtr -1 > RXSTOP_INIT)) {
350  enc28j60Write(ERXRDPTL, (RXSTOP_INIT)&0xFF);
351  enc28j60Write(ERXRDPTH, (RXSTOP_INIT)>>8);
352  } else {
353  enc28j60Write(ERXRDPTL, (NextPacketPtr-1)&0xFF);
354  enc28j60Write(ERXRDPTH, (NextPacketPtr-1)>>8);
355  }
356  // decrement the packet counter indicate we are done with this packet
357  enc28j60WriteOp(ENC28J60_BIT_FIELD_SET, ECON2, ECON2_PKTDEC);
358  return(len);
359 }
