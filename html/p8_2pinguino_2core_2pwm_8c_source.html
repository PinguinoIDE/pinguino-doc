50 /* --------------------------------------------------------------------
51  PWM_setFrequency
52  --------------------------------------------------------------------
53  @param: frequency in hertz (range 3kHz .. 12MHz)
54  --------------------------------------------------------------------
55  let's say p = TMR2 Prescale Value
56  PWM Period = [(PR2) + 1] * 4 * TOSC * p
57  so [(PR2) + 1] = PWM Period / (4 * TOSC * p)
58  but PWM Period = 1 / PWM Frequency
59  so [(PR2) + 1] = (1/PWM Frequency) / (4 * 1/FOSC * p)
60  and [(PR2) + 1] = FOSC / (4 * PWM Frequency * p)
61  then [(PR2) + 1] = FOSC / PWM Frequency / 4 / p
62  --------------------------------------------------------------------
63  When TMR2 (TMR4) is equal to PR2 (PR4) the CCPx pin is set
64  ------------------------------------------------------------------*/
65 
66 void PWM_setFrequency(u32 freq)
67 {
68  // PR2+1 calculation
69  _pr2_plus1 = System_getPeripheralFrequency() / freq; // FOSC / (4 * PWM Frequency)
70 
71  // Timer2 prescaler calculation
72  // PR2 max value is 255, so PR2+1 max value is 256
73  // highest prescaler value is 16
74  // 16 * 256 = 4096 so :
75  if (_pr2_plus1 <= 4096) // check if it's not too high
76  {
77  if (_pr2_plus1 <= 256) // no needs of any prescaler
78  {
79  _t2con = 0b00000100; // prescaler is 1, Timer2 On
80  }
81  else if (_pr2_plus1 <= 1024) // needs prescaler 1:4
82  {
83  _pr2_plus1 = _pr2_plus1 >> 2; // divided by 4
84  _t2con = 0b00000101; // prescaler is 4, Timer2 On
85  }
86  else // needs prescaler 1:6
87  {
88  _pr2_plus1 = _pr2_plus1 >> 4; // divided by 16
89  _t2con = 0b00000110; // prescaler is 16, Timer2 On
90  }
91  }
92 }

94 /* --------------------------------------------------------------------
95  PWM_setDutyCycle
96  --------------------------------------------------------------------
97  Set dutycycle with 10-bits resolution, allowing 1024 PWM steps.
98  The 'duty' argument is a (max) 10-bits absolute value for the duty cycle:
99  * duty<1:0> are the 2 LSbits
100  * duty<9:2> are the 8 MSbits
101  Allowed range: 0..1023
102  @param pin: CCPx pin where buzzer is connected (5 or 6)
103  @param duty: 10-bit duty cycle
104  --------------------------------------------------------------------
105  PWM Duty Cycle = (CCPRxL:CCPxCON<5:4>) * Tosc * TMR2 prescaler
106  (CCPRxL:CCPxCON<5:4>) = PWM Duty Cycle / (Tosc * TMR2 prescaler)
107 
108  CCPxCON = 0b00001100 => PWM mode
109  PxA and PxC are active-high;
110  PxB and PxD are active-high
111  ------------------------------------------------------------------*/

120 
121  #if defined(__18f26j53) || defined(__18f46j53) || \
122  defined(__18f27j53) || defined(__18f47j53)

125  CCPTMRS2 = 0;

130  // CCPxCON<5:4> bits.

135  defined(__18f27j53) || defined(__18f47j53)

139  CCPR4L = ( duty >> 2 ) & 0xFF; // 8 LSB
140  CCP4CON |= (duty & 0x300) >> 4; // 2 MSB in <5:4>

214 /* --------------------------------------------------------------------
215  PWM_setPercentDutyCycle
216  --------------------------------------------------------------------
217  Set a percentage duty cycle, allowing max 100 PWM steps.
218  Allowed range: 0..100
219  The duty cycle will be set to the specified percentage of the maximum
220  for the current PWM frequency.
221  Note: The number of available PWM steps can be lower than 100 with
222  (very) high PWM frequencies.
223  ------------------------------------------------------------------*/
224 
225 void PWM_setPercentDutyCycle(u8 pin, u8 percent)
226 {
227  u16 duty;
228 
229  if (percent == 0)
230  duty = 0;
231  else if (percent >= 100)
232  duty = _pr2_plus1 - 1;
233  else
234  duty = percent * (_pr2_plus1 / 4) / 25; // (factor PR2/100)
235 
236  PWM_setDutyCycle(pin, duty << 2);
237 }
