94  * series of bytes from a device on the bus
95  * Input address: 8bit i2c device address

100  */
101 u8 I2C_readFrom(u8 address, u8* data, u8 length, u8 sendStop)
102 {
103  u8 i;
104  u8 resp_sla;
105  // ensure data will fit into buffer
106  if( length > _I2CBUFFERLENGTH_){
107  return 0;
108  }
109 
110  // wait until twi is ready, become master receiver
111  while(i2c_state != I2C_READY){
112  continue;
113  }
114  i2c_state = I2C_MRX;// Master Receiver
115  i2c_sendStop = sendStop;
116  // reset error state (0xFF.. no error occured)
117  i2c_error = 0xFF;
118 
119  // initialize buffer iteration vars
120  _i2c_masterBufferIndex = 0;
121  _i2c_masterBufferLength = length; // This is not intuitive, read on...
122  // On receive, the previously configured ACK/NACK setting is transmitted in
123  // response to the received byte before the interrupt is signalled.
124  // Therefor we must actually set NACK when the _next_ to last byte is
125  // received, causing that NACK to be sent in response to receiving the last
126  // expected byte of data.
127 
128  // build sla+w, slave device address + w bit
129  i2c_slarw = (address) |I2C_READ;
130 
131  if (i2c_inRepStart) {
132  // if we're in the repeated start state, then we've already sent the start,
133  // (@@@ we hope), and the TWI statemachine is just waiting for the address byte.
134  // We need to remove ourselves from the repeated start state before we enable interrupts,
135  // since the ISR is ASYNC, and we could get confused if we hit the ISR before cleaning
136  // up. Also, don't enable the START interrupt. There may be one pending from the
137  // repeated start that we sent outselves, and that would really confuse things.
138  i2c_inRepStart = false;// remember, we're dealing with an ASYNC ISR
139  I2C_restart();
140  resp_sla = I2C_write(i2c_slarw);
141  }
142  else {
143  // send start condition
144  I2C_start();
145  resp_sla = I2C_write(i2c_slarw);
146  }
147  //address sent, ack received
148  // put byte into buffer
149  if (resp_sla) {
150  while (_i2c_masterBufferIndex < _i2c_masterBufferLength)
151  {
152  _i2c_masterBuffer[_i2c_masterBufferIndex++] = I2C_read();
153  // data received, ack sent
154  // ack if more bytes are expected, otherwise nack
155  if(_i2c_masterBufferIndex == _i2c_masterBufferLength){
156  I2C_sendNack();
157  }else{
158  I2C_sendAck();
159  }
160  }
161  }
162 
163 
164  if (_i2c_masterBufferIndex < length)
165  length = _i2c_masterBufferIndex;
166 
167  // copy i2c buffer to data
168  for(i = 0; i < length; ++i){
169  data[i] = _i2c_masterBuffer[i];
170  }
171  if (i2c_sendStop)
172  I2C_stop();
173  else {
174  i2c_inRepStart = true;
175  }
176  i2c_state = I2C_READY;
177  return length;
178 }

182  * Desc attempts to become twi bus master and write a
183  * series of bytes to a device on the bus
184  * Input address: 7bit i2c device address
185  * data: pointer to byte array

225  // if we're in a repeated start, then we've already sent the START

227  //

261  }
