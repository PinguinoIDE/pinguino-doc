146 //Fast way for drive pixels
147 #define fastSetPixel(_r_,_g_,_b_) { \
148  fastWriteHigh(LCD_RS); \
149  fastDelay(); \
150  LCD_DATA_BUS = (((_r_ & 0xF8)<<8) + ((_g_ & 0xFC)<<3) + ((_b_ & 0xF8)>>3)); \
151  fastDelay(); \
152  fastWriteLow(LCD_WR); \
153  fastDelay(); \
154  fastWriteHigh(LCD_WR); \
155  fastDelay(); \
156 }
