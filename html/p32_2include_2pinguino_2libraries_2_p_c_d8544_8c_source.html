289 void PCD8544_refresh()
290 {
291  u8 col, p;
292  u16 t = 0;
293 
294  #if (PCD8544_INTERFACE & PCD8544_SPISW)
295  #if (PCD8544_INTERFACE & PCD8544_PORTB)
296  for(p = 0; p <= DISPLAY_ROWS; p++)
297  {
298  col = 0;
299  PCD8544_command(PCD8544_SETYADDR | p);
300  PCD8544_command(PCD8544_SETXADDR | col);
301  // send data in DDRAM
302  High(PCD8544_DC);
303  Low(PCD8544_SCE);
304  for(; col < DISPLAY_WIDTH; col++)
305  PCD8544_shiftOut(PCD8544_buffer[(DISPLAY_WIDTH*p)+col]);
306  High(PCD8544_SCE);
307  }
308  #else
309  for(p = 0; p <= DISPLAY_ROWS; p++)
310  {
311  col = 0;
312  PCD8544_command(PCD8544_SETYADDR | p);
313  PCD8544_command(PCD8544_SETXADDR | col);
314  // send data in DDRAM
315  digitalwrite(PCD8544._dc, HIGH);
316  if (PCD8544._cs > 0)
317  digitalwrite(PCD8544._cs, LOW);
318  for(; col < DISPLAY_WIDTH; col++)
319  PCD8544_shiftOut(PCD8544_buffer[(DISPLAY_WIDTH*p)+col]);
320  if (PCD8544._cs > 0)
321  digitalwrite(PCD8544._cs, HIGH);
322  }
323  #endif
324  #elif (PCD8544_INTERFACE & PCD8544_SPIHW)
325  for(p = 0; p <= DISPLAY_ROWS; p++)
326  {
327  col = 0;
328  PCD8544_command(PCD8544_SETYADDR | p);
329  PCD8544_command(PCD8544_SETXADDR | col);
330  // send data in DDRAM
331  digitalwrite(PCD8544._dc, HIGH);
332  SPI_select();
333  for(; col < DISPLAY_WIDTH; col++)
334  SPI_write(PCD8544_buffer[t+col]);
335  SPI_deselect();
336  t += DISPLAY_WIDTH;
337  }
338  #endif
339 
340  // no idea why this is necessary but it is to finish the last byte?
341  //PCD8544_command(PCD8544_SETYADDR );
342 
343  #ifdef enablePartialUpdate
344  xUpdateMin = DISPLAY_WIDTH - 1;
345  xUpdateMax = 0;
346  yUpdateMin = DISPLAY_HEIGHT-1;
347  yUpdateMax = 0;
348  #endif
349 }

422 void PCD8544_printChar(u8 c)
423 {
424  u8 b;
425 
426  while (PCD8544.cursor.x >= (PCD8544.screen.width / PCD8544.font.width))
427  {
428  PCD8544.cursor.x -= (PCD8544.screen.width / PCD8544.font.width);
429  PCD8544.cursor.y++;
430  }
431 
432  while (PCD8544.cursor.y > DISPLAY_ROWS)
433  {
434  PCD8544_scrollUp();
435  }
436 
437  switch (c)
438  {
439  case '\n':
440  PCD8544.cursor.y++;
441  break;
442 
443  case '\r':
444  PCD8544.cursor.x = 0;
445  break;
446 
447  case '\t':
448  PCD8544.cursor.x = (PCD8544.cursor.x + 4) % 4;
449  break;
450 
451  default:
452  for (b = 0; b < PCD8544.font.width; b++)
453  {
454  PCD8544_buffer[PCD8544.cursor.x * PCD8544.font.width + PCD8544.cursor.y * PCD8544.screen.width + b] =
455  PCD8544.font.address[2 + (c - 32) * PCD8544.font.width + b];
456  }
457 
458  PCD8544.cursor.x++;
459  }
460 }

479 #if defined(PCD8544PRINTNUMBER) || defined(PCD8544PRINTFLOAT)
480 void PCD8544_printNumber(long value, u8 base)
481 {
482  u8 sign;
483  u8 length;
484 
485  long i;
486  unsigned long v; // absolute value
487 
488  u8 tmp[12];
489  u8 *tp = tmp; // pointer on tmp
490 
491  u8 string[12];
492  u8 *sp = string; // pointer on string
493 
494  if (value==0)
495  {
496  PCD8544_printChar('0');
497  return;
498  }
499 
500  sign = ( (base == 10) && (value < 0) );
501 
502  if (sign)
503  v = -value;
504  else
505  v = (unsigned long)value;
506 
507  //while (v || tp == tmp)
508  while (v)
509  {
510  i = v % base;
511  v = v / base;
512 
513  if (i < 10)
514  *tp++ = i + '0';
515  else
516  *tp++ = i + 'A' - 10;
517  }
518 
519  // start of string
520  if (sign)
521  *sp++ = '-';
522 
523  length = sign + tp - tmp + 1;
524 
525  // backwards writing
526  while (tp > tmp)
527  *sp++ = *--tp;
528 
529  // end of string
530  *sp = 0;
531

536 #if defined(PCD8544PRINTFLOAT)
537 void PCD8544_printFloat(float number, u8 digits)
538 {
539  u8 i, toPrint;
540  u16 int_part;
541  float rounding, remainder;
542 
543  // Handle negative numbers
544  if (number < 0.0)
545  {
546  PCD8544_printChar('-');
547  number = -number;
548  }
549 
550  // Round correctly so that print(1.999, 2) prints as "2.00"
551  rounding = 0.5;
552  for (i=0; i<digits; ++i)
553  rounding /= 10.0;
554 
555  number += rounding;
556 
557  // Extract the integer part of the number and print it
558  int_part = (u16)number;
559  remainder = number - (float)int_part;
560  PCD8544_printNumber(int_part, 10);
561 
562  // Print the decimal point, but only if there are digits beyond
563  if (digits > 0)
564  PCD8544_printChar('.');
565 
566  // Extract digits from the remainder one at a time
567  while (digits-- > 0)
568  {
569  remainder *= 10.0;
570  toPrint = (unsigned int)remainder; //Integer part without use of math.h lib, I think better! (Fazzi)
571  PCD8544_printNumber(toPrint, 10);
572  remainder -= toPrint;
573  }
574 }
