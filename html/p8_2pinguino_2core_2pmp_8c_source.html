375 /*
376  In 16-bit Data mode (MODE16 bit (PMMODEH<2>)),
377  Least Significant Byte (LSB) of data must be presented first
378  It is highly recommended that the BUSY bit be checked prior
379  to any operation that accesses the PMDIN or PMADDR register
380 
381  To perform a write on the parallel port, the user application writes
382  to the PMDIN register (same register as a read operation).
383  This causes the PMP module to first activate the Chip Select lines
384  and the address bus.
385  The write data from the PMDIN register is placed onto the PMD data bus
386  and the write line PMPWR is strobed in Master mode 2, PMRD/PMWR and
387  PMENB lines in Master Mode 1.
388 
389 */
390 
391 void PMP_write(u16 value)
392 {
393  if (_pmp_width == PMP_MODE_16BIT)
394  {
395  PMP_wait(); // wait for PMP to be available
396  PMDIN1L = low8(value);
397  PMP_wait(); // wait for PMP to be available
398  PMDIN1H = high8(value);
399  }
400 
401  else
402  {
403  PMP_wait(); // wait for PMP to be available
404  PMDIN1L = low8(value);
405  }
406 }
407 
408 /*
409  Note that the read data obtained from the PMDIN register
410  is actually the read value from the previous read operation.
411  Therefore, the first user application read will be a dummy read
412  to initiate the first bus read and fill the read register.
413  It is highly recommended that the BUSY bit be checked prior
414  to any operation that accesses the PMDIN or PMADDR register
415 */
416 
417 u16 PMP_read()
418 {
419  u16 dummy;
420 
421  if (_pmp_width == PMP_MODE_16BIT)
422  {
423  PMP_wait(); // wait for PMP to be available
424  dummy = PMDIN1L; // init read cycle, dummy read
425  PMP_wait(); // wait for PMP to be available
426  return (make16(PMDIN1L, PMDIN1H));
427  }
428 
429  else
430  {
431  PMP_wait(); // wait for PMP to be available
432  dummy = PMDIN1L; // dummy read
433  PMP_wait(); // wait for PMP to be available
434  return (PMDIN1L); // real read
435  }
436 }
