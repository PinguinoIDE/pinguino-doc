335  * USB CDC printNumber routine (CDC.printNumber)
336  * added by regis blanchot 14/06/2011

338  * base : see const.h (DEC, BIN, HEXA, OCTO, ...)
339  **********************************************************************/

341 #if defined(CDCPRINTNUMBER) || defined(CDCPRINTFLOAT)
342 void CDCprintNumber(long value, u8 base)
343 {
344  u8 sign;
345  u8 length;
346 
347  long i;
348  unsigned long v; // absolute value
349 
350  u8 tmp[12];
351  u8 *tp = tmp; // pointer on tmp
352 
353  u8 string[12];
354  u8 *sp = string; // pointer on string
355 
356  if (value==0)
357  {
358  CDCputs("0", 1);
359  return;
360  }
361 
362  sign = ( (base == 10) && (value < 0) );
363 
364  if (sign)
365  v = -value;
366  else
367  v = (unsigned long)value;
368 
369  //while (v || tp == tmp)
370  while (v)
371  {
372  i = v % base;
373  v = v / base;
374 
375  if (i < 10)
376  *tp++ = i + '0';
377  else
378  *tp++ = i + 'A' - 10;
379  }
380 
381  // start of string
382  if (sign)
383  *sp++ = '-';
384 
385  length = sign + tp - tmp + 1;
386 
387  // backwards writing
388  while (tp > tmp)
389  *sp++ = *--tp;
390 
391  // end of string
392  *sp = 0;
393 
394  CDCputs(string, length);
395 }

398 /***********************************************************************
399  * USB CDC printFloat routine (CDC.printFloat)
400  * added by regis blanchot 14/06/2011
401  * write a float number on CDC port
402  * base : see const.h (DEC, BIN, HEXA, OCTO, ...)
403  **********************************************************************/
404 
405 #if defined(CDCPRINTFLOAT)
406 void CDCprintFloat(float number, u8 digits)
407 {
408  u8 i, toPrint;
409  u16 int_part;
410  float rounding, remainder;
411 
412  // Handle negative numbers
413  if (number < 0.0)
414  {
415  CDCputs('-', 1);
416  number = -number;
417  }
418 
419  // Round correctly so that print(1.999, 2) prints as "2.00"
420  rounding = 0.5;
421  for (i=0; i<digits; ++i)
422  rounding /= 10.0;
423 
424  number += rounding;
425 
426  // Extract the integer part of the number and print it
427  int_part = (u16)number;
428  remainder = number - (float)int_part;
429  CDCprintNumber(int_part, 10);
430 
431  // Print the decimal point, but only if there are digits beyond
432  if (digits > 0)
433  CDCputs('.', 1);
434 
435  // Extract digits from the remainder one at a time
436  while (digits-- > 0)
437  {
438  remainder *= 10.0;
439  toPrint = (unsigned int)remainder; //Integer part without use of math.h lib, I think better! (Fazzi)
440  CDCprintNumber(toPrint, 10);
441  remainder -= toPrint;
442  }
443 }
444 #endif
