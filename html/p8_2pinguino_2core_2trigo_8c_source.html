38 /* --------------------------------------------------------------------
39  First 90 values (1/4's of the circle) of precomputed sinus table * 256
40  To obtain the other 3/4's of the circle, we use the symmetry of sine wave.
41  Each quadrant is obtained as follows:
42  0 <= x < 90 y = sine[ x ];
43  90 <= x < 180 y = sine[ 180 - x ];
44  180 <= x < 270 y = -sine[ x - 180 ];
45  270 <= x < 360 y = -sine[ 360 - x ];
46  ------------------------------------------------------------------*/
47 
48 // http://devmaster.net/posts/9648/fast-and-accurate-sine-cosine
49 float sine(int i)
50 {
51  float x = 0.01745329 * (float)i; // degree to rad
52  float B = 1.27323954; // 4/pi;
53  float C = -0.40528473; //-4/(pi*pi);
54  float y = B * x + C * x * x;
55  float P = 0.225;
56  y = P * (y * y - y) + y;
57  return y;
58 }
