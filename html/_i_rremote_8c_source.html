293  // The mark output is modulated at the PWM frequency.
294

298  // loopcounter i = time - correction factor

305  while (i) {
306  _1us_
307  i--;
308  }

407 // TIMER3 interrupt code to collect raw data.
408 // Widths of alternating SPACE, MARK are recorded in rawbuf.
409 // Recorded in ticks of 50 microseconds.
410 // rawlen counts the number of entries recorded so far.
411 // First entry is the SPACE between transmissions.
412 // As soon as a SPACE gets long, ready is set, state switches to IDLE, timing of SPACE continues.
413 // As soon as first MARK arrives, gap width is recorded, ready is cleared, and new logging starts
414 
415 #if defined(__IRREMOTE__)
416 void irremote_interrupt() // called from main.c
417 {
418  if (PIR2bits.TMR3IF) // Timer3 interrupt ?
419  {
420  irdata = digitalread(irparams.recvpin);
421 
422  irparams.timer++; // One more 50us tick
423 
424  // Buffer overflow ?
425  if (irparams.rawlen >= RAWBUF)
426  irparams.rcvstate = STATE_STOP;
427 
428  switch(irparams.rcvstate)
429  {
430  // In the middle of a gap
431  case STATE_IDLE:
432  if (irdata == MARK)
433  {
434  // Not big enough to be a gap.
435  if (irparams.timer < GAP_TICKS)
436  irparams.timer = 0;
437 
438  // gap just ended, record duration and start recording transmission
439  else
440  {
441  irparams.rawlen = 0;
442  irparams.rawbuf[irparams.rawlen++] = irparams.timer;
443  irparams.timer = 0;
444  irparams.rcvstate = STATE_MARK;
445  }
446  }
447  break;
448 
449  // timing MARK
450  case STATE_MARK:
451 
452  // MARK ended, record time
453  if (irdata == SPACE)
454  {
455  irparams.rawbuf[irparams.rawlen++] = irparams.timer;
456  irparams.timer = 0;
457  irparams.rcvstate = STATE_SPACE;
458  }
459  break;
460 
461  // timing SPACE
462  case STATE_SPACE:
463 
464  // SPACE just ended, record it
465  if (irdata == MARK)
466  {
467  irparams.rawbuf[irparams.rawlen++] = irparams.timer;
468  irparams.timer = 0;
469  irparams.rcvstate = STATE_MARK;
470  }
471 
472  // SPACE
473  else
474  {
475  // big SPACE, indicates gap between codes
476  // Mark current code as ready for processing
477  // Switch to STOP
478  // Don't reset timer; keep counting space width
479  if (irparams.timer > GAP_TICKS)
480  irparams.rcvstate = STATE_STOP;
481  }
482  break;
483 
484  // waiting, measuring gap
485  case STATE_STOP:
486 
487  // reset gap timer
488  if (irdata == MARK)
489  irparams.timer = 0;
490  break;
491  }
492 
493  if (irparams.blinkflag)
494  {
495  if (irdata == MARK)
496  digitalwrite(USERLED, 1); // turn USERLED on
497  else
498  digitalwrite(USERLED, 0); // turn USERLED off
499  }
500 
501  // Timer3 reset
502  TMR3H = high8(_t3_reload_val);
503  TMR3L = low8(_t3_reload_val);
504 
505  // Timer3 flag reset
506  PIR2bits.TMR3IF = 0;
507  }
508 }
509 #endif

590 #if defined(IRRECV_DECODENEC) || defined(IRRECV_DECODE)
591 // NECs have a repeat only 4 items long
592 u32 IRrecv_decodeNEC(decode_results *results)
593 {
594  u32 data = 0;
595  u16 offset = 1; // Skip first space
596  u16 i;
597  // Initial mark
598  if (!MATCH_MARK(results->rawbuf[offset], NEC_HDR_MARK)) {
599  return ERR;
600  }
601  offset++;
602  // Check for repeat
603  if (irparams.rawlen == 4 &&
604  MATCH_SPACE(results->rawbuf[offset], NEC_RPT_SPACE) &&
605  MATCH_MARK(results->rawbuf[offset+1], NEC_BIT_MARK)) {
606  results->bits = 0;
607  results->value = REPEAT;
608  results->decode_type = NEC;
609  return DECODED;
610  }
611  if (irparams.rawlen < 2 * NEC_BITS + 4) {
612  return ERR;
613  }
614  // Initial space
615  if (!MATCH_SPACE(results->rawbuf[offset], NEC_HDR_SPACE)) {
616  return ERR;
617  }
618  offset++;
619 
620  for (i = 0; i < NEC_BITS; i++) {
621  if (!MATCH_MARK(results->rawbuf[offset], NEC_BIT_MARK)) {
622  return ERR;
623  }
624  offset++;
625  if (MATCH_SPACE(results->rawbuf[offset], NEC_ONE_SPACE)) {
626  data = (data << 1) | 1;
627  }
628  else if (MATCH_SPACE(results->rawbuf[offset], NEC_ZERO_SPACE)) {
629  data <<= 1;
630  }
631  else {
632  return ERR;
633  }
634  offset++;
635  }
636  // Success
637  results->bits = NEC_BITS;
638  results->value = data;
639  results->decode_type = NEC;
640  return DECODED;
641 }

646 {

649 
650  if (irparams.rawlen < 2 * SONY_BITS + 2) return ERR;
651 
652  // Some Sony's deliver repeats fast after first
653  // unfortunately can't spot difference from of repeat from two fast clicks
654  if (results->rawbuf[offset] < SONY_DOUBLE_SPACE_USECS)
655  {
656  // Serial_print("IR Gap found: ");
657  results->bits = 0;
658  results->value = REPEAT;
659  results->decode_type = SANYO;
660  return DECODED;
661  }
662  offset++;
663 
664  // Initial mark

666  offset++;

671  break;
672 
673  offset++;

836 // Gets one undecoded level at a time from the raw buffer.
837 // The RC5/6 decoding is easier if the data is broken into time intervals.
838 // E.g. if the buffer has MARK for 2 time intervals and SPACE for 1,
839 // successive calls to getRClevel will return MARK, MARK, SPACE.
840 // offset and used are updated to keep track of the current position.
841 // t1 is the time interval for a single bit in microseconds.
842 // Returns -1 for error (measured time interval is not a multiple of t1).
843 u16 IRrecv_getRClevel(decode_results *results, u16 *offset, u16 *used, u16 t1)
844 {

846 
847  if (*offset >= results->rawlen) {
848  // After end of recorded buffer, assume SPACE.
849  return SPACE;
850  }
851  width = results->rawbuf[*offset];
852  val = ((*offset) % 2) ? MARK : SPACE;
853  correction = (val == MARK) ? MARK_EXCESS : - MARK_EXCESS;
854 
855  if (MATCH(width, t1 + correction)) {
856  avail = 1;
857  }
858  else if (MATCH(width, 2*t1 + correction)) {
859  avail = 2;
860  }

863  }

866  }

870  *used = 0;
871  (*offset)++;

875  Serial_println("MARK");
876  }
877  else {
878  Serial_println("SPACE");

1085 /* -----------------------------------------------------------------------
1086  * hashdecode - decode an arbitrary IR code.
1087  * Instead of decoding using a standard encoding scheme
1088  * (e.g. Sony, NEC, RC5), the code is hashed to a 32-bit value.
1089  *
1090  * The algorithm: look at the sequence of MARK signals, and see if each one
1091  * is shorter (0), the same length (1), or longer (2) than the previous.
1092  * Do the same with the SPACE signals. Hszh the resulting sequence of 0's,
1093  * 1's, and 2's to a 32-bit value. This will give a unique value for each
1094  * different code (probably), for most code systems.
1095  *
1096  * http://arcfn.com/2010/01/using-arbitrary-remotes-with-arduino.html
1097  */

1100 // 1 if newval is equal, and 2 if newval is longer
1101 // Use a tolerance of 20%
1102 u16 IRrecv_compare(u16 oldval, u16 newval)
1103 {
1104  if (newval < oldval * .8) {
1105  return 0;
1106  }
1107  else if (oldval < newval * .8) {
1108  return 2;
1109  }
1110  else {
1111  return 1;
1112  }
1113 }
