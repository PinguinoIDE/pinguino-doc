251 unsigned findDIR(MFILE *fp)
252 {
253  unsigned eCount; // current entry counter
254  unsigned e; // current entry offset
255  int i, a;
256  MEDIA *mda = fp->mda;
257 
258  // 1. start from the first entry
259  eCount = 0;
260 
261  // load the first sector of root
262  if (!readDIR(fp, eCount))
263  return FAIL;
264 
265  // 2. loop until you reach the end or find the file
266  while (1)
267  {
268  // 2.0 determine the offset in current buffer
269  e = (eCount&0xf) * DIR_ESIZE;
270 
271  // 2.1 read the first char of the file name
272  a = fp->buffer[e + DIR_NAME];
273 
274  // 2.2 terminate if it is empty (end of the list)
275  if (a == DIR_EMPTY)
276  {
277  return NOT_FOUND;
278  } // empty entry
279 
280  // 2.3 skip erased entries if looking for a match
281  if (a != DIR_DEL)
282  {
283  // 2.3.1 if not VOLume or DIR compare the names
284  a = fp->buffer[e + DIR_ATTRIB];
285  if (!(a & (ATT_DIR | ATT_HIDE)) )
286  {
287  // compare file name and extension
288  for (i=DIR_NAME; i<DIR_ATTRIB; i++)
289  {
290  if (fp->buffer[e + i] != fp->name[i])
291  break; // difference found
292  }
293 
294  if (i == DIR_ATTRIB)
295  {
296  // entry found, fill the file structure
297  fp->entry = eCount; // store index
298  fp->time = ReadW(fp->buffer, e+DIR_TIME);
299  fp->date = ReadW(fp->buffer, e+DIR_DATE);
300  fp->size = ReadL(fp->buffer, e+DIR_SIZE);
301  fp->cluster = ReadL(fp->buffer, e+DIR_CLST);
302  return FOUND;
303  }
304  } // not dir nor vol
305  } // not deleted
306 
307  // 2.4 get the next entry
308  eCount++;
309  if ((eCount & 0xf) == 0)
310  { // load a new sector from the Dir
311  if (!readDIR(fp, eCount))
312  return FAIL;
313  }
314 
315  // 2.5. exit the loop if reached the end or error
316  if (eCount >= mda->maxroot)
317  return NOT_FOUND; // last entry reached
318 
319  }// while
320 
321 } // findDIR

388 unsigned readFAT(MFILE *fp, unsigned ccls)
389 {
390  unsigned p, c;
391  LBA l;
392 
393  // get page of current cluster in fat
394  p = ccls >>8; // 256 clusters per sector
395 
396  // check if already cached
397  if (fp->fpage != p)
398  {
399  // load the fat sector containing the cluster
400  l = fp->mda->fat + p;
401  if (!readSECTOR(l, fp->buffer))
402  return FAT_EOF; // failed
403 
404  // note the sector contains a valid FAT page cache
405  fp->fpage = ccls>>8;
406  }
407 
408  // get the next cluster value
409  // cluster = 0xabcd
410  // packed as: 0 | 1 | 2 | 3 |
411  // word p 0 1 | 2 3 | 4 5 | 6 7 |..
412  // cd ab| cd ab| cd ab| cd ab|
413  c = ReadOddW(fp->buffer, ((ccls & 0xFF)<<1));
414 
415  return c;
416 } // readFAT

426 unsigned writeFAT(MFILE *fp, unsigned cls, unsigned v)
427 {
428  int i;
429  unsigned p;
430  LBA l;
431 
432  // load the FAT page containing the requested cluster
433  readFAT(fp, cls);
434 
435  // cluster = 0xabcd
436  // packed as: 0 | 1 | 2 | 3 |
437  // word p 0 1 | 2 3 | 4 5 | 6 7 |..
438  // cd ab| cd ab| cd ab| cd ab|
439 
440  // locate the cluster within the FAT page
441  p = (cls & 0xff)*2; // 2 bytes per cluster
442 
443  // get the next cluster value
444  fp->buffer[p] = v; // lsb
445  fp->buffer[p+1] = (v>>8);// msb
446 
447  // update all FAT copies
448  l = fp->mda->fat + (cls>>8);
449  for (i=0; i<fp->mda->fatcopy; i++, l += fp->mda->fatsize)
450  if (!writeSECTOR(l, fp->buffer))
451  return FAIL;
452 
453  return TRUE;
454 } // writeFAT

505 unsigned newFAT(MFILE * fp)
506 {
507  unsigned i, c = fp->ccls;
508 
509  // sequentially scan through the FAT
510  // looking for an empty cluster
511  do {
512  c++; // check next cluster in FAT
513  // check if reached last cluster in FAT,
514  // re-start from top
515  if (c >= fp->mda->maxcls)
516  c = 0;
517 
518  // check if full circle done, media full
519  if (c == fp->ccls)
520  {
521  FError = FE_MEDIA_FULL;
522  return FAIL;
523  }
524 
525  // look at its value
526  i = readFAT(fp, c);
527 
528  } while (i!=0); // scanning for an empty cluster
529 
530  // mark the new cluster as taken, and last in chain
531  writeFAT(fp, c, FAT_EOF);
532 
533  // if not first cluster, link current cluster to new one
534  if (fp->ccls >0)
535  writeFAT(fp, fp->ccls, c);
536 
537  // update the MFILE structure
538  fp->ccls = c;
539 
540  // invalidate the FAT cache
541  // (since it will soon be overwritten with data)
542  fp->fpage = -1;
543 
544  return TRUE;
545 } // newFAT

588  {
589  FError = FE_MEDIA_NOT_MNTD;
590  return NULL;
591  }
592 
593  // 2. allocate a buffer for the file
594  b = (unsigned char*)malloc(512);
595  if (b == NULL)
596  {
597  FError = FE_MALLOC_FAILED;
598  return NULL;
599  }
600 
601  // 3. allocate a MFILE structure on the heap
602  fp = (MFILE *) malloc(sizeof(MFILE));
603  if (fp == NULL) // report an error
604  {
605  FError = FE_MALLOC_FAILED;
606  free(b);
607  return NULL;
608  }
609 
610  // 4. set pointers to the MEDIA structure and buffer
611  fp->mda = D;
612  fp->buffer = b;
613 
614  // 5. format the filename into name
615  for(i=0; i<8; i++)
616  {
617  // read a char and convert to upper case
618  c = toupper(*filename++);
619  // extension or short name no-extension
620  if ((c == '.') || (c == '\0'))
621  break;
622  else
623  fp->name[i] = c;
624  } // for
625  // if short fill the rest up to 8 with spaces
626  while (i<8) fp->name[i++] = ' ';
627 
628  // 6. if there is an extension
629  if (c != '\0')
630  {
631  for(i=8; i<11; i++)
632  {
633  // read a char and convert to upper case
634  c = toupper(*filename++);
635  if (c == '.')
636  c = toupper(*filename++);
637  if (c == '\0') // short extension
638  break;
639  else
640  fp->name[i] = c;
641  } // for
642  // if short fill the rest up to 3 with spaces
643  while (i<11) fp->name[i++] = ' ';
644  } // if
645 
646  // 7. copy the file mode character (r, w)
647  if ((*mode == 'r')||(*mode == 'w'))
648  fp->mode = *mode;
649  else
650  {
651  FError = FE_INVALID_MODE;
652  goto ExitOpen;
653  }
654 
655  // 8. Search for the file in current directory
656  if ((r = findDIR(fp)) == FAIL)
657  {
658  FError = FE_FIND_ERROR;
659  goto ExitOpen;
660  }
661 
662  // 9. init all counters to the beginning of the file
663  fp->seek = 0; // first byte in file
664  fp->sec = 0; // first sector in the cluster
665  fp->pos = 0; // first byte in sector/cluster
666 
667  // 10. depending on the mode (read or write)
668  if (fp->mode == 'r')
669  { // 10.1 'r' open for reading
670  if (r == NOT_FOUND)
671  {
672  FError = FE_FILE_NOT_FOUND;
673  goto ExitOpen;
674  }
675  else
676  { // found
677  // 10.2 set current cluster pointer on first cluster
678  fp->ccls = fp->cluster;
679 
680  // 10.3 read a sector of data from the file
681  if (!readDATA(fp))
682  {
683  goto ExitOpen;
684  }
685 
686  // 10.4 determine how much data is really inside buffer
687  if (fp->size-fp->seek < 512)
688  fp->top = fp->size - fp->seek;
689  else
690  fp->top = 512;
691  } // found
692  } // 'r'
693  else // 11. open for 'write'
694  {
695  if (r == NOT_FOUND)
696  {
697  // 11.1 allocate a first cluster to it
698  fp->ccls = 0; // indicate brand new file
699  if (newFAT(fp) != TRUE)
700  { // must be media full
701  FError = FE_MEDIA_FULL;
702  goto ExitOpen;
703  }
704  fp->cluster = fp->ccls;
705 
706  // 11.2 create a new entry
707  // search again, for an empty entry this time
708  if ((r = newDIR(fp)) == FAIL)
709  { // report any error
710  FError = FE_IDE_ERROR;
711  goto ExitOpen;
712  }
713  // 11.3 new entry not found
714  if (r == NOT_FOUND)
715  {
716  FError = FE_DIR_FULL;
717  goto ExitOpen;
718  }
719  else // 11.4 new entry identified fp->entry filled
720  {
721  // 11.4.1
722  fp->size = 0;
723 
724  // 11.4.2 determine offset in DIR sector
725  e = (fp->entry & 0xf) * DIR_ESIZE;
726 
727  // 11.4.3 init all fields to 0
728  for (i=0; i<32; i++)
729  fp->buffer[e + i] = 0;
730 
731  // 11.4.4 set date and time
732  fp->date = 0x378A; // Dec 10th, 2007
733  fp->buffer[e + DIR_CDATE] = fp->date;
734  fp->buffer[e + DIR_CDATE+1]= fp->date>>8;
735  fp->buffer[e + DIR_DATE] = fp->date;
736  fp->buffer[e + DIR_DATE+1]= fp->date>>8;
737 
738  fp->time = 0x6000; // 12:00:00 PM
739  fp->buffer[e + DIR_CTIME] = fp->time;
740  fp->buffer[e + DIR_CTIME+1]= fp->time>>8;
741  fp->buffer[e + DIR_TIME] = fp->time+1;
742  fp->buffer[e + DIR_TIME+1]= fp->time>>8;
743 
744  // 11.4.5 set first cluster
745  fp->buffer[e + DIR_CLST] = fp->cluster;
746  fp->buffer[e + DIR_CLST+1]= (fp->cluster>>8);
747 
748  // 11.4.6 set name
749  for (i = 0; i<DIR_ATTRIB; i++)
750  fp->buffer[e + i] = fp->name[i];
751 
752  // 11.4.7 set attrib
753  fp->buffer[e + DIR_ATTRIB] = ATT_ARC;
