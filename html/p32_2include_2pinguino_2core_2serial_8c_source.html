189 /* --------------------------------------------------------------------
190  SerialSetDataRate()
191  ----------------------------------------------------------------------------
192  @param port 1 (UART1) or 2 (UART2)
193  @param baudrate baud rate
194  @return baudrate
195  ----------------------------------------------------------------------------
196  BRGH: High Baud Rate Enable bit
197  if BRGH = 1 = High-Speed mode - 4x baud clock enabled
198  then UxBRG = ((FPB/Desired Baud Rate)/ 4) - 1
199  if BRGH 0 = Standard Speed mode - 16x baud clock enabled
200  then UxBRG = ((FPB/Desired Baud Rate)/16) - 1
201  ------------------------------------------------------------------*/
202 
203 void SerialSetDataRate(u8 port, u32 baudrate)
204 {
205  u8 speed;
206  u32 max, max1, max2;
207  u32 min1, min2;
208  u32 pbclock;
209 
210  pbclock = GetPeripheralClock();
211  max1 = pbclock / 4;
212  min1 = max1 / 65536;
213  max2 = pbclock / 16;
214  min2 = max2 / 65536;
215  if (baudrate > max1) baudrate = max1;
216  if (baudrate < min2) baudrate = min2;
217  max = (min1 + max2) / 2;
218  //if (baudrate > min2 && baudrate < max) speed = UART_ENABLE_STANDARD_SPEED;
219  if (baudrate > max && baudrate < max1) speed = UART_ENABLE_HIGH_SPEED;
220 
221  switch (port)
222  {
223  case UART1:
224  if (speed == UART_ENABLE_HIGH_SPEED)
225  {
226  U1MODEbits.BRGH = UART_ENABLE_HIGH_SPEED;
227  U1BRG = ((pbclock / baudrate) / 4) - 1;
228  }
229  else
230  U1BRG = ((pbclock / baudrate) / 16) - 1;
231  break;
232  case UART2:
233  if (speed == UART_ENABLE_HIGH_SPEED)
234  {
235  U2MODEbits.BRGH = UART_ENABLE_HIGH_SPEED;
236  U2BRG = ((pbclock / baudrate) / 4) - 1;
237  }
238  else
239  U2BRG = ((pbclock / baudrate) / 16) - 1;
240  break;
241 #ifdef ENABLE_UART3
242  case UART3:
243  if (speed == UART_ENABLE_HIGH_SPEED)
244  {
245  U2AMODEbits.BRGH = UART_ENABLE_HIGH_SPEED;
246  U2ABRG = ((pbclock / baudrate) / 4) - 1;
247  }
248  else
249  U2ABRG = ((pbclock / baudrate) / 16) - 1;
250  break;
251 #endif
252 #ifdef ENABLE_UART4
253  case UART4:
254  if (speed == UART_ENABLE_HIGH_SPEED)
255  {
256  U1BMODEbits.BRGH = UART_ENABLE_HIGH_SPEED;
257  U1BBRG = ((pbclock / baudrate) / 4) - 1;
258  }

749 {

801 /***********************************************************************

803  * added by regis blanchot 14/06/2011
804  * write a float number on SERIAL port
805  * base : see const.h (DEC, BIN, HEXA, OCTO, ...)
806  **********************************************************************/

810 {

812  u16 int_part;
813  float rounding, remainder;
814 
815  // Handle negative numbers
816  if (number < 0.0)
817  {
818  SerialPutChar(port, '-');
819  number = -number;
820  }
821 
822  // Round correctly so that print(1.999, 2) prints as "2.00"
823  rounding = 0.5;
824  for (i=0; i<digits; ++i)
825  rounding /= 10.0;
826 
827  number += rounding;
828 
829  // Extract the integer part of the number and print it

834  // Print the decimal point, but only if there are digits beyond
835  if (digits > 0)
836  SerialPutChar(port, '.');
837 
838  // Extract digits from the remainder one at a time
839  while (digits-- > 0)
840  {
841  remainder *= 10.0;
842  toPrint = (unsigned int)remainder; //Integer part without use of math.h lib, I think better! (Fazzi)
843  SerialPrintNumber(port, toPrint, 10);
844  remainder -= toPrint;
845  }
