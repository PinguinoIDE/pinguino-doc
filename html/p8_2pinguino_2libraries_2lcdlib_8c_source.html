146 {

168 #ifdef LCDPRINTFLOAT
169 void lcd_printFloat(float number, u8 digits)
170 {
171  u8 i, toPrint;
172  u16 int_part;
173  float rounding, remainder;
174 
175  // Handle negative numbers
176  if (number < 0.0)
177  {
178  lcd_write('-');
179  number = -number;
180  }
181 
182  // Round correctly so that print(1.999, 2) prints as "2.00"
183  rounding = 0.5;
184  for (i=0; i<digits; ++i)
185  rounding /= 10.0;
186 
187  number += rounding;
188 
189  // Extract the integer part of the number and print it
190  int_part = (u16)number;
191  remainder = number - (float)int_part;
192  lcd_printNumber(int_part, 10);
193 
194  // Print the decimal point, but only if there are digits beyond
195  if (digits > 0)
196  lcd_write('.');
197 
198  // Extract digits from the remainder one at a time
199  while (digits-- > 0)
200  {
201  remainder *= 10.0;
202  toPrint = (unsigned int)remainder; //Integer part without use of math.h lib, I think better! (Fazzi)
203  lcd_printNumber(toPrint, 10);
204  remainder -= toPrint;
205  }
206 }

330 void lcd_begin(u8 lines, u8 dotsize)
331 {
332  if (lines > 1)
333  _displayfunction |= LCD_2LINE;
334 
335  _numlines = lines;
336  _currline = 0;
337 
338  // Some one line displays can select 10 pixel high font
339  if ((dotsize != 0) && (lines == 1))
340  _displayfunction |= LCD_5x10DOTS;
341 
342  Delayms(15); // Wait more than 15 ms after VDD rises to 4.5V
343 
344  // Now we pull both RS and R/W low to begin commands
345  digitalwrite(_rs_pin, LOW);
346  digitalwrite(_enable_pin, LOW);
347 
348  //put the LCD into 4 bit mode
349  if (! (_displayfunction & LCD_8BITMODE) )
350  {
351  // this is according to the hitachi HD44780 datasheet p46, figure 24
352 
353  // we start in 8bit mode, try to set 4 bit mode
354  lcd_write4bits(0x03);
355  Delayms(5); // Wait for more than 4.1 ms
356  // second try
357  lcd_write4bits(0x03);
358  Delayus(150); // Wait more than 100 s
359  // third go!
360  lcd_write4bits(0x03);
361  Delayus(150); // Wait more than 100 s
362  // finally, set to 8-bit interface
363  lcd_write4bits(0x02);
364  }
365  //put the LCD into 8 bit mode
366  else
367  {
368  // this is according to the hitachi HD44780 datasheet p45, figure 23
369 
370  // Send function set command sequence
371  lcd_command(LCD_FUNCTIONSET | _displayfunction);
372  Delayms(5); // Wait for more than 4.1 ms
373 
374  // second try
375  lcd_command(LCD_FUNCTIONSET | _displayfunction);
376  Delayms(5); // Wait for more than 4.1 ms
377  // Delayus(150);
378 
379  // third go
380  lcd_command(LCD_FUNCTIONSET | _displayfunction);
381  }
382 
383  // finally, set # lines, font size, etc.
384  lcd_command(LCD_FUNCTIONSET | _displayfunction);
385 
386  // turn the display on with no cursor or blinking default
387  _displaycontrol = LCD_DISPLAYON | LCD_CURSOROFF | LCD_BLINKOFF;
388  lcd_command(LCD_DISPLAYCONTROL | _displaycontrol);
389 
390  // clear it off
391  lcd_command(LCD_CLEARDISPLAY); // clear display, set cursor position to zero
392  Delayms(2);
393 
394  // Initialize to default text direction (for romance languages)
395  _displaymode = LCD_ENTRYLEFT | LCD_ENTRYSHIFTDECREMENT;
396  // set the entry mode
397  lcd_command(LCD_ENTRYMODESET | _displaymode);
398 }
