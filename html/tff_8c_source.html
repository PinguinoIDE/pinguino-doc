159 CLUST get_cluster ( /* 0,>=2: successful, 1: failed */
160  CLUST clust /* Cluster# to get the link information */
161 )
162 {
163  word wc, bc;
164  dword fatsect;
165  FATFS *fs = FatFs;
166 
167 
168  if (clust >= 2 && clust < fs->max_clust) { /* Valid cluster# */
169  fatsect = fs->fatbase;
170  switch (fs->fs_type) {
171  case FS_FAT12 :
172  bc = (word)clust * 3 / 2;
173  if (!move_window(fatsect + bc / 512U)) break;
174  wc = fs->win[bc % 512U]; bc++;
175  if (!move_window(fatsect + bc / 512U)) break;
176  wc |= (word)fs->win[bc % 512U] << 8;
177  return (clust & 1) ? (wc >> 4) : (wc & 0xFFF);
178 
179  case FS_FAT16 :
180  if (!move_window(fatsect + clust / 256)) break;
181  return LD_WORD(&fs->win[((word)clust * 2) % 512U]);
182 #if _FAT32
183  case FS_FAT32 :
184  if (!move_window(fatsect + clust / 128)) break;
185  return LD_DWORD(&fs->win[((word)clust * 4) % 512U]) & 0x0FFFFFFF;
186 #endif
187  }
188  }
189 
190  return 1; /* Out of cluster range, or an error occured */
191 }

440 /*-----------------------------------------------------------------------*/
441 
442 static
443 char make_dirfile ( /* 1: error - detected an invalid format, '\0'or'/': next character */
444  const char **path, /* Pointer to the file path pointer */
445  char *dirname /* Pointer to directory name buffer {Name(8), Ext(3), NT flag(1)} */
446 )
447 {
448  byte n, t, c, a, b;
449 
450 
451  memset((void *)dirname, ' ', 8+3); /* Fill buffer with spaces */
452  a = 0; b = 0x18; /* NT flag */
453  n = 0; t = 8;
454  for (;;) {
455  c = *(*path)++;
456  if (c == '\0' || c == '/') { /* Reached to end of str or directory separator */
457  if (n == 0) break;
458  dirname[11] = _USE_NTFLAG ? (a & b) : 0;
459  return c;
460  }
461  if (c <= ' ' || c == 0x7F) break; /* Reject invisible chars */
462  if (c == '.') {
463  if (!(a & 1) && n >= 1 && n <= 8) { /* Enter extension part */
464  n = 8; t = 11; continue;
465  }
466  break;
467  }
468  if (_USE_SJIS &&
469  ((c >= 0x81 && c <= 0x9F) || /* Accept S-JIS code */
470  (c >= 0xE0 && c <= 0xFC))) {
471  if (n == 0 && c == 0xE5) /* Change heading \xE5 to \x05 */
472  c = 0x05;
473  a ^= 1; goto md_l2;
474  }
475  if (c == '"') break; /* Reject " */
476  if (c <= ')') goto md_l1; /* Accept ! # $ % & ' ( ) */
477  if (c <= ',') break; /* Reject * + , */
478  if (c <= '9') goto md_l1; /* Accept - 0-9 */
479  if (c <= '?') break; /* Reject : ; < = > ? */
480  if (!(a & 1)) { /* These checks are not applied to S-JIS 2nd byte */
481  if (c == '|') break; /* Reject | */
482  if (c >= '[' && c <= ']') break;/* Reject [ \ ] */
483  if (_USE_NTFLAG && c >= 'A' && c <= 'Z')
484  (t == 8) ? (b &= 0xF7) : (b &= 0xEF);
485  if (c >= 'a' && c <= 'z') { /* Convert to upper case */
486  c -= 0x20;
487  if (_USE_NTFLAG) (t == 8) ? (a |= 0x08) : (a |= 0x10);
488  }
489  }
490  md_l1:
491  a &= 0xFE;
492  md_l2:
493  if (n >= t) break;
494  dirname[n++] = c;
495  }
496  return 1;
497 }
