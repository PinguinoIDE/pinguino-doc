148 char SystemIsMclrReset(void)
149 {
150  // power-on reset ?
151  if (SystemIsPowerOnReset()) return 0;
152  // config. mismatch reset ?
153  if (SystemIsConfigMismatchReset()) return 0;
154  // sleep instruction reset ?
155  if (SystemIsSleepInstructionReset()) return 0;
156  // brown-out reset ?
157  if (SystemIsBrownOutReset()) return 0;
158  // watchdog timeout reset ?
159  if (SystemIsWatchdogTimeoutReset()) return 0;
160  // reset instruction ?
161  if (SystemIsResetInstructionReset()) return 0;
162  // stack full reset ?
163  if (SystemIsStackFullReset()) return 0;
164  // stack underflow reset ?
165  if (SystemIsStackUnderflowReset()) return 0;
166  // then it's a MCLR-pin reset
167  return 1;
168 }
