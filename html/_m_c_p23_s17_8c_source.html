54 //IOCON Flag
55 //Unimplemented: Read as 0.
56 #define MCP23S17_B_NotUse 1
57 //INTPOL: This bit sets the polarity of the INT output pin.
58 //1 = Active-high.
59 //0 = Active-low.
60 #define MCP23S17_B_INTPOL 2
61 //ODR: This bit configures the INT pin as an open-drain output.
62 //1 = Open-drain output (overrides the INTPOL bit).
63 //0 = Active driver output (INTPOL bit sets the polarity).
64 #define MCP23S17_B_ODR 4
65 //HAEN: Hardware Address Enable bit (MCP23S17 only).
66 //1 = Enables the MCP23S17 address pins.
67 //0 = Disables the MCP23S17 address pins.
68 #define MCP23S17_B_HAEN 8
69 //DISSLW: Slew Rate control bit for SDA output.
70 //1 = Slew rate disabled.
71 //0 = Slew rate enabled.
72 #define MCP23S17_B_DISSLW 16
73 //SEQOP: Sequential Operation mode bit.
74 //1 = Sequential operation disabled, address pointer does not increment.
75 //0 = Sequential operation enabled, address pointer increments.
76 #define MCP23S17_B_SEQOP 32
77 //MIRROR: INT Pins Mirror bit
78 //1 = The INT pins are internally connected
79 //0 = The INT pins are not connected. INTA is associated with PortA and INTB is associated with PortB
80 #define MCP23S17_B_MIRROR 64
81 //BANK: Controls how the registers are addressed
82 //1 = The registers associated with each port are separated into different banks
83 //0 = The registers are in the same bank (addresses are sequential)
84 #define MCP23S17_B_BANK 128

113 
114 Write One BYTE in the registers
115 
116 Mcp = Pointer to a user declared MCP23S17 structure
117 Addr = Register Address
118 Val = 8 bit value to write
119 
120 RETURN FALSE on Error
121 ***********************************************************************************/
122 BOOL MCP23S17_write_B(MCP23S17 * Mcp,u8 Addr,u8 Val)
123 {
124  digitalwrite(Mcp->CSPin, LOW);
125  switch (Mcp->SPIChannel)
126  {
127  case 0:
128  SPILoc_Write(&Mcp->SPILoc,((MCP23S17_ADDRESS | (Mcp->SPIAddr&7))<<1)&~1); //Write Into Device
129  SPILoc_Write(&Mcp->SPILoc,Addr);
130  SPILoc_Write(&Mcp->SPILoc,Val);
131  break;
132  case 1:
133  case 2:
134  SPI_write(((MCP23S17_ADDRESS | (Mcp->SPIAddr&7))<<1)&~1); //Write Into Device
135  SPI_write(Addr);
136  SPI_write(Val);
137  break;
138  }
139  digitalwrite(Mcp->CSPin, HIGH);
140  return(TRUE);
141 }

344 /***********************************************************************************
345 BOOL MCP23S17_init(MCP23S17 * Mcp, u8 SPIAddr, u8 CSPin, BOOL UseIntAdd, u8 SPIChannel, u8 LoSCK, u8 LoSDO, u8 LoSDI)
346 
347 Init a MCP23S17 device and structure
348 
349 Mcp = Pointer to a user declared MCP23S17 structure
350 SPIAddr = Internal adress of the MCP23S17 device
351 CSPin = CS Pin of the device
352 UseIntAdd = Enable use of Internal Address (TRUE = YES)
353 SPIChannel = SPI Channel to use (0 = Emulated SPI, 1 = SPI1)
354 LoSCK = SCK Pin of emulated SPI port
355 LoSDO = SDO Pin of emulated SPI port
356 LoSDI = SDI Pin of emulated SPI port
357 
358 RETURN FALSE on Error
359 ***********************************************************************************/
360 BOOL MCP23S17_init(MCP23S17 * Mcp, u8 SPIAddr, u8 CSPin, BOOL UseIntAdd, u8 SPIChannel, u8 LoSCK, u8 LoSDO, u8 LoSDI)
361 {
362  u8 Val;
363 
364  Mcp->SPIAddr = SPIAddr;
365  Mcp->CSPin = CSPin;
366  Mcp->UseIntAdd = UseIntAdd;
367  Mcp->SPIChannel = SPIChannel;
368 
369  // Initialize the chip enable, HIGH
370  digitalwrite(Mcp->CSPin, HIGH);
371  pinmode(Mcp->CSPin, OUTPUT);
372  switch (Mcp->SPIChannel)
373  {
374  case 0:
375  Mcp->SPILoc.CPOL = FALSE; // Clock Polarity
376  Mcp->SPILoc.CPHA = TRUE; // Sample position
377  Mcp->SPILoc.DelayuS = 0; // Delay Clock uS
378  Mcp->SPILoc.LoSCK = LoSCK;
379  Mcp->SPILoc.LoSDO = LoSDO;
380  Mcp->SPILoc.LoSDI = LoSDI;
381  SPILoc_Init (&Mcp->SPILoc);
382  break;
383  case 1:
384  case 2:
385  SPI_init();
386  break;
387  }
388  Mcp->IODIR.W = 0xFFFF;
389  Mcp->GPIO.W = 0;
390  Mcp->GPPU.W = 0;
391  Mcp->IPOL.W = 0;
392  Val = 0;
393  if (UseIntAdd == TRUE)
394  Val |= MCP23S17_B_HAEN;
395  if(MCP23S17_write_B(Mcp,MCP23S17_INTCONA,Val) == FALSE)
396  return(FALSE);
397  if(MCP23S17_write_W(Mcp,MCP23S17_IODIRA,0xFFFF) == FALSE)
398  return(FALSE);
399  return(TRUE);
400 }
