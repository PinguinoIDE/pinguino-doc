132 static
133 int rcvr_datablock( /* 1:OK, 0:Failed */
134 PF_BYTE *buff, /* Data buffer to store received data */
135 UINT btr /* Byte count (must be multiple of 4) */
136 ) {
137  PF_BYTE d;
138  UINT tmr;
139 
140  for (tmr = 1000; tmr; tmr--) { /* Wait for data packet in timeout of 100ms */
141  d = rcvr_spi();
142  if (d != 0xFF)
143  break;
144  DLY_US(100);
145  }
146  if (d != 0xFE)
147  return 0; /* If not valid data token, return with error */
148 
149  do { /* Receive the data block into buffer */
150  rcvr_spi_m(buff++);
151  rcvr_spi_m(buff++);
152  rcvr_spi_m(buff++);
153  rcvr_spi_m(buff++);
154  } while (btr -= 4);
155  rcvr_spi(); /* Discard CRC */
156  rcvr_spi();
157 
158  return 1; /* Return with success */
159 }

198 static PF_BYTE send_cmd(PF_BYTE cmd, /* Command byte */
199 DWORD arg /* Argument */
200 ) {
201  PF_BYTE n, res;
202 
203  if (cmd & 0x80) { /* ACMD<n> is the command sequense of CMD55-CMD<n> */
204  cmd &= 0x7F;
205  res = send_cmd(CMD55, 0);
206  if (res > 1)
207  return res;
208  }
209 
210  /* Select the card and wait for ready */
211  deselect();
212  if (!select())
213  return 0xFF;
214 
215  /* Send command packet */
216  xmit_spi(0x40 | cmd); /* Start + Command index */
217  xmit_spi((PF_BYTE)(arg >> 24)); /* Argument[31..24] */
218  xmit_spi((PF_BYTE)(arg >> 16)); /* Argument[23..16] */
219  xmit_spi((PF_BYTE)(arg >> 8)); /* Argument[15..8] */
220  xmit_spi((PF_BYTE)arg); /* Argument[7..0] */
221  n = 0x01; /* Dummy CRC + Stop */
222  if (cmd == CMD0)
223  n = 0x95; /* Valid CRC for CMD0(0) */
224  if (cmd == CMD8)
225  n = 0x87; /* Valid CRC for CMD8(0x1AA) */
226  xmit_spi(n);
227 
228  /* Receive command response */
229  if (cmd == CMD12)
230  rcvr_spi(); /* Skip a stuff byte when stop reading */
231  n = 10; /* Wait for a valid response in timeout of 10 attempts */
232  do
233  res = rcvr_spi();
234  while ((res & 0x80) && --n);
235 
236  return res; /* Return with the response value */
237 }

249 DSTATUS disk_initialize(PF_BYTE drv /* Physical drive number (0) */
250 ) {
251  PF_BYTE n, ty, cmd, buf[4];
252  UINT tmr;
253  DSTATUS s = 0;
254 
255  if (drv)
256  return STA_NOINIT; /* Supports only single drive */
257  if (Stat & STA_NODISK)
258  return Stat; /* No card in the socket */
259 
260  deselect(); /* Force socket power on */
261  //FCLK_SLOW();
262  for (n = 10; n; n--)
263  rcvr_spi(); /* 80 dummy clocks */
264 
265  CS_H();
266  for (n = 10; n; n--)
267  rcvr_spi(); /* 80 dummy clocks */
268 
269  ty = 0;
270  if (send_cmd(CMD0, 0) == 1) { /* Enter Idle state */
271  if (send_cmd(CMD8, 0x1AA) == 1) { /* SDv2? */
272  for (n = 0; n < 4; n++)
273  buf[n] = rcvr_spi(); /* Get trailing return value of R7 resp */
274  if (buf[2] == 0x01 && buf[3] == 0xAA) { /* The card can work at vdd range of 2.7-3.6V */
275  for (tmr = 1000; tmr; tmr--) { /* Wait for leaving idle state (ACMD41 with HCS bit) */
276  if (send_cmd(ACMD41, 1UL << 30) == 0)
277  break;
278  DLY_US(1000);
279  }
280  if (tmr && send_cmd(CMD58, 0) == 0) { /* Check CCS bit in the OCR */
281  for (n = 0; n < 4; n++)
282  buf[n] = rcvr_spi();
283  ty = (buf[0] & 0x40) ? CT_SD2 | CT_BLOCK : CT_SD2; /* SDv2 */
284  }
285  }
286  } else { /* SDv1 or MMCv3 */
287  if (send_cmd(ACMD41, 0) <= 1) {
288  ty = CT_SD1;
289  cmd = ACMD41; /* SDv1 */
290  } else {
291  ty = CT_MMC;
292  cmd = CMD1; /* MMCv3 */
293  }
294  for (tmr = 1000; tmr; tmr--) { /* Wait for leaving idle state */
295  if (send_cmd(ACMD41, 0) == 0)
296  break;
297  DLY_US(1000);
298  }
299  if (!tmr || send_cmd(CMD16, 512) != 0) /* Set R/W block length to 512 */
300  ty = 0;
301  }
302  }
303  CardType = ty;
304  if (ty) {/* Initialization succeded */
305  s &= ~STA_NOINIT;
306  // 6. increase speed
307  SPI2CON = 0; // disable the SPI2 module
308  SPI2BRG = 0; // maximum possible baud rate = Fpb/2
309  SPI2CON = 0x8120; // re-enable the SPI2 module
310  } else {
311  /* Initialization failed */
312  s |= STA_NOINIT;
313  }
314  Stat = s;
315 
316  deselect();
317 
318  return Stat;
319 }

336 DRESULT disk_read(PF_BYTE drv, /* Physical drive nmuber (0) */
337 PF_BYTE *buff, /* Pointer to the data buffer to store read data */
338 DWORD sector, /* Start sector number (LBA) */
339 PF_BYTE count /* Sector count (1..255) */
340 ) {
341  if (drv || !count)
342  return RES_PARERR;
343  if (Stat & STA_NOINIT)
344  return RES_NOTRDY;
345 
346  if (!(CardType & CT_BLOCK))
347  sector *= 512; /* Convert to byte address if needed */
348 
349  if (count == 1) { /* Single block read */
350  if ((send_cmd(CMD17, sector) == 0) /* READ_SINGLE_BLOCK */
351  && rcvr_datablock(buff, 512))
352  count = 0;
353  } else { /* Multiple block read */
354  if (send_cmd(CMD18, sector) == 0) { /* READ_MULTIPLE_BLOCK */
355  do {
356  if (!rcvr_datablock(buff, 512))
357  break;
358  buff += 512;
359  } while (--count);
360  send_cmd(CMD12, 0); /* STOP_TRANSMISSION */
361  }
362  }
363  deselect();
364 
365  return count ? RES_ERROR : RES_OK;
366 }

415 DRESULT disk_ioctl(PF_BYTE drv, /* Physical drive number (0) */
416 PF_BYTE ctrl, /* Control code */
417 void *buff /* Buffer to send/receive data block */
418 ) {
419  DRESULT res;
420  PF_BYTE n, csd[16], *ptr = buff;
421  DWORD csize;
422 
423  if (drv)
424  return RES_PARERR;
425  if (Stat & STA_NOINIT)
426  return RES_NOTRDY;
427 
428  res = RES_ERROR;
429  switch (ctrl) {
430  case CTRL_SYNC: /* Flush dirty buffer if present */
431  if (select()) {
432  deselect();
433  res = RES_OK;
434  }
435  break;
436 
437  case GET_SECTOR_COUNT: /* Get number of sectors on the disk (WORD) */
438  if ((send_cmd(CMD9, 0) == 0) && rcvr_datablock(csd, 16)) {
439  if ((csd[0] >> 6) == 1) { /* SDv2? */
440  csize = csd[9] + ((WORD) csd[8] << 8) + 1;
441  *(DWORD*) buff = (DWORD) csize << 10;
442  } else { /* SDv1 or MMCv2 */
443  n = (csd[5] & 15) + ((csd[10] & 128) >> 7)
444  + ((csd[9] & 3) << 1) + 2;
445  csize = (csd[8] >> 6) + ((WORD) csd[7] << 2) + ((WORD) (csd[6]
446  & 3) << 10) + 1;
447  *(DWORD*) buff = (DWORD) csize << (n - 9);
448  }
449  res = RES_OK;
450  }
451  break;
452 
453  case GET_SECTOR_SIZE: /* Get sectors on the disk (WORD) */
454  *(WORD*) buff = 512;
455  res = RES_OK;
456  break;
457 
458  case GET_BLOCK_SIZE: /* Get erase block size in unit of sectors (DWORD) */
459  if (CardType & CT_SD2) { /* SDv2? */
460  if (send_cmd(ACMD13, 0) == 0) { /* Read SD status */
461  rcvr_spi();
462  if (rcvr_datablock(csd, 16)) { /* Read partial block */
463  for (n = 64 - 16; n; n--)
464  rcvr_spi(); /* Purge trailing data */
465  *(DWORD*) buff = 16UL << (csd[10] >> 4);
466  res = RES_OK;
467  }
468  }
469  } else { /* SDv1 or MMCv3 */
470  if ((send_cmd(CMD9, 0) == 0) && rcvr_datablock(csd, 16)) { /* Read CSD */
471  if (CardType & CT_SD1) { /* SDv1 */
472  *(DWORD*) buff = (((csd[10] & 63) << 1) + ((WORD) (csd[11]
473  & 128) >> 7) + 1) << ((csd[13] >> 6) - 1);
474  } else { /* MMCv3 */
475  *(DWORD*) buff = ((WORD) ((csd[10] & 124) >> 2) + 1)
476  * (((csd[11] & 3) << 3) + ((csd[11] & 224) >> 5)
477  + 1);
478  }
479  res = RES_OK;
480  }
481  }
482  break;
483 
484  case MMC_GET_TYPE: /* Get card type flags (1 byte) */
485  *ptr = CardType;
486  res = RES_OK;
487  break;
488 
489  case MMC_GET_CSD: /* Receive CSD as a data block (16 bytes) */
490  if ((send_cmd(CMD9, 0) == 0) /* READ_CSD */
491  && rcvr_datablock(buff, 16))
492  res = RES_OK;
493  break;
494 
495  case MMC_GET_CID: /* Receive CID as a data block (16 bytes) */
496  if ((send_cmd(CMD10, 0) == 0) /* READ_CID */
497  && rcvr_datablock(buff, 16))
498  res = RES_OK;
499  break;
500 
501  case MMC_GET_OCR: /* Receive OCR as an R3 resp (4 bytes) */
502  if (send_cmd(CMD58, 0) == 0) { /* READ_OCR */
503  for (n = 0; n < 4; n++)
504  *((PF_BYTE*) buff + n) = rcvr_spi();
505  res = RES_OK;
506  }
507  break;
508 
509  case MMC_GET_SDSTAT: /* Receive SD status as a data block (64 bytes) */
510  if (send_cmd(ACMD13, 0) == 0) { /* SD_STATUS */
511  rcvr_spi();
512  if (rcvr_datablock(buff, 64))
513  res = RES_OK;
514  }
515  break;
516 
517  default:
518  res = RES_PARERR;
519  }
520 
521  deselect();
522 
523  return res;
524 }

531 void disk_timerproc(void) {
532  static WORD pv;
533  WORD p;
534  PF_BYTE s;
535  UINT n;
536 
537  n = Timer1; /* 1000Hz decrement timer */
538  if (n)
539  Timer1 = --n;
540  n = Timer2;
541  if (n)
542  Timer2 = --n;
543 
544  p = pv;
545  pv = getCD() & getWP(); /* Sample socket switch */
546 
547  if (p == pv) { /* Have contacts stabled? */
548  s = Stat;
549 
550  if (p & getWP()) /* WP is H (write protected) */
551  s |= STA_PROTECT;
552  else
553  /* WP is L (write enabled) */
554  s &= ~STA_PROTECT;
555 
556  if (p & getCD()) /* INS = H (Socket empty) */
557  s |= (STA_NODISK | STA_NOINIT);
558  else
559  /* INS = L (Card inserted) */
560  s &= ~STA_NODISK;
561 
562  Stat = s;
563  }

568 /*---------------------------------------------------------*/
569 /* This is a real time clock service to be called from */
570 /* FatFs module. Any valid time must be returned even if */

574 /* The current time is returned packed into a DWORD
575  (32 bit) value. The bit fields are as follows:
576  bits 31:25 Year from 1980 (0..127)
577  bits 24:21 Month (1..12)
578  bits 20:16 Day in month (1..31)
579  bits 15:11 Hour (0..23)
580  bits 10:05 Minute (0..59)
581  bits 04:00 Second / 2 (0..29) */
582 
583 DWORD get_fattime(void) {
584  DWORD tmr = 0;
585 
586 // Pre-processor commands added so only use RTCC if the board is known
587 // to support the RTCC ** Added 07 May 2012
588 
589 // For boards known to support the RTCC library
590 
591 #if defined (PIC32_PINGUINO) || defined (PIC32_PINGUINO_OTG)
592  rtccTime pTm, cTm;
593  rtccDate pDt, cDt;
594 
595  RTCC_GetTimeDate(&pTm, &pDt); // get time and date from RTC
596  // assumes RTC has been set and is running
597  // OK - could be expanded to check that RTC
598  // is running and that a valid value is
599  // being returned by the RTC
600  cTm = RTCC_ConvertTime(&pTm); // convert time from bcd to decimal format
601  cDt = RTCC_ConvertDate(&pDt); // convert date from bcd to decimal format
602 
603  /* Pack date and time into a DWORD variable */
604 // tmr = (((DWORD) pDt.year - 80)) | ((DWORD) pDt.mon) | ((DWORD) pDt.mday)
605 // | (WORD) (pTm.hour) | (WORD) (pTm.min) | (WORD) (pTm.sec);
606  tmr = cDt.year + 20;
607 
608  // Correction according to a newer rtcc library, with diff struct members.
609 
610  tmr = (tmr << 4) | cDt.month; // shifts left 4 bits and adds monthth
611  tmr = (tmr << 5) | cDt.dayofmonth; // shifts left 5 bits and adds m.day
612  tmr = (tmr << 5) | cTm.hours; // shifts left 5 bits and adds hour
613  tmr = (tmr << 6) | cTm.minutes; // shift left 6 bits and adds minutes
614  tmr = (tmr << 5) | (cTm.seconds/2); // shifts left 5 bits and adds seconds/2
615 
616 // For other boards use a fixed date and time of 01 Jan 2012 12:00:00
617 #else
618  tmr = 12 + 20;
619  tmr = (tmr << 4) | 1; // shifts left 4 bits and adds month
620  tmr = (tmr << 5) | 1; // shifts left 5 bits and adds m.day
621  tmr = (tmr << 5) | 12; // shifts left 5 bits and adds hour
622  tmr = (tmr << 6) | 0; // shift left 6 bits and adds minutes
623  tmr = (tmr << 5) | (0/2); // shifts left 5 bits and adds seconds/2
624 #endif
625 
626  return tmr;
627 }
