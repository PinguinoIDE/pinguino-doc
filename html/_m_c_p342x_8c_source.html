304 /***********************************************************************************
305 BOOL MCP342x_loop(MCP342x * Mcp)
306 
307 MCP342x device management loop
308 
309 Mcp = Pointer to a user declared MCP342x structure
310 
311 RETURN
312  3 = New conversion started
313  2 = Data not ready
314  1 = Nothing to do
315 -1 = Not valid pointer
316 -2 = Error reading I2C data
317 -3 = Conversion timeout
318 -4 = No channel found
319 -5 = Error writing I2C data
320 ***********************************************************************************/
321 int MCP342x_loop(MCP342x * Mcp)
322  {
323  int Valo;
324  u8 *Vb = (u8 *)&Valo;
325  u8 Ctn, Found,ValTx;
326 
327  // No valid structure return error
328  if (Mcp == NULL)
329  return (-1);
330 
331  if ( (Mcp->On == FALSE) || (millis() < Mcp->MilliSec) )
332  return (1);
333 
334  //Test if sampling started
335  if (Mcp->InSample == TRUE)
336  {
337  //Read Data from I2C port
338  if (MCP342x_read_Data(Mcp) == FALSE)
339  return(-2);
340  //Test if data not available
341  if ((Mcp->FrameRx[0] & MCP342x_BUSY) != 0)
342  {
343  //Inc error count
344  Mcp->CtErr++;
345  //if too many error abort sampling
346  if (Mcp->CtErr > 10)
347  {
348  Mcp->ActCh = 0;
349  Mcp->MilliSec = millis() + 2;
350  Mcp->InSample = FALSE;
351  return(-3);
352  }
353  else
354  {
355  //else wait 50mS
356  Mcp->MilliSec = millis() + 50;
357  return(2);
358  }
359  }
360  else
361  {
362  //Data available
363  Mcp->CtErr = 0;
364  //propagate sign
365  Vb[3] = Mcp->FrameRx[3] & 0X80 ? 0XFF : 0;
366  //check for 16 or 18 bit data
367  if ((Mcp->FrameRx[0] & MCP342x_18_BIT) == MCP342x_18_BIT)
368  {
369  //transfer 18 bit data
370  Vb[0] = Mcp->FrameRx[1];
371  Vb[1] = Mcp->FrameRx[2];
372  Vb[2] = Mcp->FrameRx[3];
373  }
374  else
375  {
376  //transfer 16 bit data
377  Vb[0] = Mcp->FrameRx[2];
378  Vb[1] = Mcp->FrameRx[3];
379  Vb[2] = Vb[3];
380  }
381  Mcp->Ch[Mcp->ActCh-1].Points = Valo;
382  //calculate nV ((Point * nV/point) / Gain)
383  switch (Mcp->FrameRx[0] & MCP342x_18_BIT)
384  {
385  case MCP342x_12_BIT:
386  Valo *= MCP342x_12_BIT_NV;
387  break;
388  case MCP342x_14_BIT:
389  Valo *= MCP342x_14_BIT_NV;
390  break;
391  case MCP342x_16_BIT:
392  Valo *= MCP342x_16_BIT_NV;
393  break;
394  case MCP342x_18_BIT:
395  Valo *= MCP342x_18_BIT_NV;
396  break;
397  }
398 
399  switch (Mcp->FrameRx[0] & MCP342x_256mVFS)
400  {
401  case MCP342x_2048mVFS:
402  Valo /= 1;
403  break;
404  case MCP342x_1024mVFS:
405  Valo /= 2;
406  break;
407  case MCP342x_512mVFS:
408  Valo /= 4;
409  break;
410  case MCP342x_256mVFS:
411  Valo /= 8;
412  break;
413  }
414  Mcp->Ch[Mcp->ActCh-1].nV = Valo;
415  Mcp->InSample = FALSE;
416  }
417  }
418 
419  //Test if not in sampling
420  if (Mcp->InSample == FALSE)
421  {
422  Found = 0;
423  //If channel not assigned or wrong channel or channel stopped
424  if ( (Mcp->ActCh == 0) || (Mcp->ActCh > Mcp->NumCh) || (Mcp->Ch[Mcp->ActCh-1].On == FALSE))
425  {
426  //find first valid channell
427  for (Ctn = 1 ; Ctn <= Mcp->NumCh ; Ctn++)
428  if (Mcp->Ch[Ctn-1].On == TRUE)
429  {
430  Found = Ctn;
431  break;
432  }
433  }
434  else
435  {
436  //Find Next valid channell
437  Ctn = Mcp->ActCh+1;
438  while (Found == 0)
439  {
440  if (Ctn > Mcp->NumCh)
441  Ctn = 1;
442  if (Mcp->Ch[Ctn-1].On == TRUE)
443  {
444  Found = Ctn;
445  break;
446  }
447  if (Ctn == Mcp->ActCh)
448  {
449  if (Mcp->Ch[Ctn-1].On == TRUE)
450  Found = Ctn;
451  break;
452  }
453  Ctn++;
454  }
455  }
456  //Test if valid channel not found
457  if (Found == 0)
458  {
459  //Stop any sampling and return error
460  Mcp->On = FALSE;
461  Mcp->ActCh = 0;
462  Mcp->MilliSec = millis()+50;
463  return (-4);
464  }
465  else
466  {
467  //set actual channel
468  Mcp->ActCh = Found;
469  //Set channel parameters
470  switch (Mcp->ActCh)
471  {
472  case 1:
473  ValTx = MCP342x_START | MCP342x_CHANNEL_1 | Mcp->Ch[0].Res | Mcp->Ch[0].Gain;
474  break;
475  case 2:
476  ValTx = MCP342x_START | MCP342x_CHANNEL_2 | Mcp->Ch[1].Res | Mcp->Ch[1].Gain;
477  break;
478  case 3:
479  ValTx = MCP342x_START | MCP342x_CHANNEL_3 | Mcp->Ch[2].Res | Mcp->Ch[2].Gain;
480  break;
481  case 4:
482  ValTx = MCP342x_START | MCP342x_CHANNEL_4 | Mcp->Ch[3].Res | Mcp->Ch[3].Gain;
483  break;
484  }
485  //Start a new sampling
486  if (MCP342x_write(Mcp,ValTx) == FALSE)
487  {
488  Mcp->ActCh = 0;
489  Mcp->MilliSec = millis()+70;
490  return (-5);
491  }
492 
493  //Calculate and set the appropriate conversion delay
494  switch (Mcp->Ch[Mcp->ActCh-1].Res & MCP342x_18_BIT)
495  {
496  case MCP342x_12_BIT:
497  Mcp->MilliSec = millis()+MCP342x_12_BIT_MS;
498  break;
499  case MCP342x_14_BIT:
500  Mcp->MilliSec = millis()+MCP342x_14_BIT_MS;
501  break;
502  case MCP342x_16_BIT:
503  Mcp->MilliSec = millis()+MCP342x_16_BIT_MS;
504  break;
505  case MCP342x_18_BIT:
506  Mcp->MilliSec = millis()+MCP342x_18_BIT_MS;
507  break;
508  }
509  //set sampling in progress
510  Mcp->InSample = TRUE;
511  return(3);
512  }
513  }
514  }
