181 u8 BT_init(u8 uart_port, u32 baud_rate)
182 {
183  SerialConfigure(uart_port, UART_ENABLE, UART_RX_TX_ENABLED, baud_rate);
184 //BT_echoOff(uart_port);
185  BT_setCommandMode(uart_port);
186  BT_restore(uart_port);
187  BT_setDeviceName(uart_port, "Pinguino");
188 // Pass through, DCE, enable escape sequence, disable entering in command moode with DTR/DSR, enable LED
189 //BT_send(uart_port,"AT+BTCFG=33\r\n");
190  BT_setAutoConnection(uart_port);
191  BT_setUARTSpeed(uart_port, baud_rate);
192  BT_disableSecurity(uart_port);
193  BT_writeFlash(uart_port);
194  BT_start(uart_port);
195 
196  return ( BT_ok(uart_port) );
197 }
