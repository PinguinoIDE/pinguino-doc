164 void SetAnalog(u8 pin)
165 {
166  #if defined(PIC32_PINGUINO) || defined(PIC32_PINGUINO_OTG)
167  if ((pin==4)||(pin==18)) TRISDbits.TRISD9=1; // analog input 18 is shared with I2C
168  if ((pin==5)||(pin==19)) TRISDbits.TRISD10=1; // analog input 19 is shared with I2C
169  #endif
170  #if defined(PIC32_PINGUINO_220)||defined(PINGUINO32MX250)||defined(PINGUINO32MX220)
171  #if defined(PINGUINO32MX250)||defined(PINGUINO32MX220)
172  ANSELBSET=__analogmask[pin];
173  TRISBSET=__analogmask[pin];
174  #else
175  switch (__portanalogmask[pin])
176  {
177  case 1: ANSELBSET=__analogmask[pin];
178  TRISBSET=__analogmask[pin];

223 u16 analogRead(u8 pin)
224 {
225 
226  AD1CON1bits.ADON = 0; // stop analogic converter
227  if (IsDigital(pin)) SetAnalog(pin); // set analog
228  AD1CHS=(__bufmask[pin]/4)<<16; // select channel
229  AD1CON1bits.ADON = 1; // start analogic converter
230  AD1CON1bits.SAMP=1; // start sampling
231  while (!AD1CON1bits.DONE); // wait for conversion
232  return(ADC1BUF0+(8*(AD1CON2bits.BUFS&0x01))); // return result
233 }
