233 /* ----------------------------------------------------------------------------
234  ---------- Ecriture d'un quartet (mode 4 bits) dans le LCD
235  ----------------------------------------------------------------------------
236  Envoie d'un quartet vers les pins :
237  - D4 a D7 du LCD
238  - P4 a P7 du PCF8574
239  NB : quartet est en fait un 8 bits dont seuls les 4 bits de poids fort nous interessent
240  @param quartet = 4 bits a envoyer au LCD
241  @param mode = LCD Command (LCD_CMD) or Data (LCD_DATA) mode
242  --------------------------------------------------------------------------*/
243 
244 static void lcdi2c_send4(u8 quartet, u8 mode)
245 {
246  // D7 D6 D5 D4 EN RW RS BL
247  PCF8574_data.val = quartet; // x x x x 0 0 0 0
248  LCD_EN = LOW; // x x x x 0 0 0 0
249  LCD_RW = LCD_WRITE; // x x x x 0 0 0 0
250  LCD_RS = mode; // x x x x 0 0 0/1 0
251  LCD_BL = gBacklight; // x x x x 0 0 0/1 0/1
252 
254 
255  I2C_start(); // send start condition
256 
257  //I2C_writechar(PCF8574_address | I2C_WRITE);
258  I2C_write((PCF8574_address << 1) | I2C_WRITE);
259 
260  LCD_EN = HIGH;
261  I2C_write(PCF8574_data.val);
262  // E Pulse Width > 300ns
263 
264  LCD_EN = LOW;
265  I2C_write(PCF8574_data.val);
266  // E Enable Cycle > (300 + 200) = 500ns
267 
268  I2C_stop(); // send stop confition
269 }

271 /* ----------------------------------------------------------------------------
272  ---------- Ecriture d'un octet dans le LCD en mode 4 bits
273  ----------------------------------------------------------------------------
274  Les données sont écrites en envoyant séquentiellement :
275  1/ les quatre bits de poids fort
276  2/ les quatre bits de poids faible
277  NB : les poids sont stockes dans les quatre bits de poids fort
278  qui correspondent aux pins D4 a D7 du LCD ou du PCF8574
279  @param octet = octet a envoyer au LCD
280  @param mode = LCD Command (LCD_CMD) or Data (LCD_DATA) mode
281  --------------------------------------------------------------------------*/
282 
283 static void lcdi2c_send8(u8 octet, u8 mode)
284 {
285  lcdi2c_send4(octet & LCD_MASK, mode); // send upper 4 bits
286  lcdi2c_send4((octet << 4) & LCD_MASK, mode); // send lower 4 bits
287  //Delayus(46); // Wait for instruction excution time (more than 46us)
288 }
