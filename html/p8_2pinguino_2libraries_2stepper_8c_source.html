16  FIRST RELEASE: 17-05-2011
17  LAST RELEASE: 19-12-2013
18  --------------------------------------------------------------------
19  When wiring multiple stepper motors to a microcontroller,
20  you quickly run out of output pins, with each motor requiring 4 connections.
21 
22  By making use of the fact that at any time two of the four motor
23  coils are the inverse of the other two, the number of
24  control connections can be reduced from 4 to 2.
25 
26  A slightly modified circuit around a Darlington transistor array or an L293 H-bridge
27  connects to only 2 microcontroler pins, inverts the signals received,
28  and delivers the 4 (2 plus 2 inverted ones) output signals required
29  for driving a stepper motor.
30 
31  The sequence of wavestep control signals for 4 control wires is as follows:
32 
33  Step A B C D
34  1 1 0 0 0
35  2 0 0 1 0
36  3 0 1 0 0
37  4 0 0 0 1
38 
39  The sequence of fullstep control signals for 4 control wires is as follows:
40 
41  Step A B C D
42  1 1 0 1 0
43  2 0 1 1 0
44  3 0 1 0 1
45  4 1 0 0 1
46 
47  The sequence of halfstep control signals for 4 control wires is as follows:
48 
49  Step A B C D
50  1 1 0 1 0
51  2 0 0 1 0
52  3 0 1 1 0
53  4 0 1 0 0
54  5 0 1 0 1
55  6 0 0 0 1
56  7 1 0 0 1
57  8 1 0 0 0
58 
59  The sequence of controls signals for 2 control wires is as follows :
60  (columns C1 and C2 from above):
61 
62  Step A B
63  1 0 1
64  2 1 1
65  3 1 0
66  4 0 0
67 
68  The circuits can be found at
69

311 {
312  //int delay_ms_per_move; // delay for the whole move
313  int delay_us_per_step; // delay between 2 steps
314  int prescaler = 0; // 0 = 1:1 default prescale value

316  // 2 = 1:64 prescale value
317  // 3 = 1:256 prescale value
318 
319  // delay between 2 steps (1 mn = 60 * 1000 * 1000 us)
320  delay_us_per_step = 60 * 1000 * 1000 / revolutionPerMinute / this_number_of_steps;
321  //delay_ms_per_move = delay_us_per_step * steps_to_move / 1000;
322  // number of CPU cycles between 2 steps
323  cycles_per_step = delay_us_per_step * ( System_getPeripheralFrequency() / 1000 / 1000 );
324 
325  while (cycles_per_step > 0xFFFF)
326  {
327  prescaler += 1;
328  if (prescaler > 2)
329  {
330  prescaler = 3;
331  cycles_per_step /= 4;
332  }
333  else
334  {
335  cycles_per_step /= 8;
336  }
337  }

339  // Timer3 config.
340  INTCONbits.GIEH = 0; // enable global HP interrupts
341  INTCONbits.GIEL = 0; // enable global LP interrupts
342  TMR3H = high8(0xFFFF - cycles_per_step);
343  TMR3L = low8(0xFFFF - cycles_per_step);
344  T3CON = prescaler << 4; // prescaler, internal peripheral clock
345  IPR2bits.TMR3IP = 1; // interrupt has high priority
346  PIR2bits.TMR3IF = 0; // reset interrupt flag

384 void Stepper_stepMotor(int thisStep)
385 {
386  int index;
387  int current_microstep;
388  int t, inc, dec;
389 
390  // wave-step
391  if (this_number_of_microsteps == 0)
392  {
393  // step the motor to step number 0, 1, 2, or 3:
394  t = wave_table [ (thisStep % 4) ];
395 
396  // Winding A
397  digitalwrite( this_motor_pin_A, t >> 0 & 1); // bit 0
398  digitalwrite( this_motor_pin_B, t >> 1 & 1); // bit 1
399 
400  // Winding B
401  if (this_pin_count == 4)
402  {
403  digitalwrite( this_motor_pin_C, t >> 2 & 1); // bit 2
404  digitalwrite( this_motor_pin_D, t >> 3 & 1); // bit 3
405  }
406  }
407 
408  // full-step
409  else if (this_number_of_microsteps == 1)
410  {
411  // step the motor to step number 0, 1, 2, or 3:
412  t = steps_table [ (thisStep % 4) * 2 ];
413 
414  // Winding A
415  digitalwrite( this_motor_pin_A, t >> 0 & 1); // bit 0
416  digitalwrite( this_motor_pin_B, t >> 1 & 1); // bit 1
417 
418  // Winding B
419  if (this_pin_count == 4)
420  {
421  digitalwrite( this_motor_pin_C, t >> 2 & 1); // bit 2
422  digitalwrite( this_motor_pin_D, t >> 3 & 1); // bit 3
423  }
424  }
425 
426  // half-step
427  else if (this_number_of_microsteps == 2)
428  {
429  // step the motor to step number 0, 1, 2, 3, 4, 5, 6 or 7
430  t = steps_table [ thisStep % 8 ];
431 
432  // Winding A
433  digitalwrite( this_motor_pin_A, t >> 0 & 1); // bit 0
434  digitalwrite( this_motor_pin_B, t >> 1 & 1); // bit 1
435 
436  // Winding B
437  if (this_pin_count == 4)
438  {
439  digitalwrite( this_motor_pin_C, t >> 2 & 1); // bit 2
440  digitalwrite( this_motor_pin_D, t >> 3 & 1); // bit 3
441  }
442  }
443 
444  // microstepping (4, 8, 16 or 32 microsteps / step)
445  #ifdef __MICROSTEPPING__
446  else if (this_number_of_microsteps > 2)
447  {
455  // from 0 to total number of microsteps - 1
456  current_microstep = ( thisStep % this_number_of_microsteps );// + 1;
457 
458  // table index is function of total microsteps number
459  // this_jump = (TABLE_DEPTH / this_number_of_microsteps);
460  index = current_microstep * this_jump;
461 
462  // from 0 to 90 degrees (from 0 to 1023)
463  // increase current in next active coil
464  inc = sine_table[ index & INDEX_MASK ];
465 
466  // from 90 to 0 degrees (from 1023 to 0)
467  // decrease current in active coil
468  dec = sine_table[ INDEX_MASK - (index & INDEX_MASK) ];
469 
470  switch ( (thisStep / this_number_of_microsteps) % 4 )
471  {
472  case 0: // from A to C
473  analogwrite( this_motor_pin_A, (u16) dec );
474  analogwrite( this_motor_pin_B, (u16) 0 );
475  analogwrite( this_motor_pin_C, (u16) inc );
476  analogwrite( this_motor_pin_D, (u16) 0 );
477  break;
478 
479  case 1: // from C to B
480  analogwrite( this_motor_pin_A, (u16) 0 );
481  analogwrite( this_motor_pin_B, (u16) inc );
482  analogwrite( this_motor_pin_C, (u16) dec );
483  analogwrite( this_motor_pin_D, (u16) 0 );
484  break;
485 
486  case 2: // from B to D
487  analogwrite( this_motor_pin_A, (u16) 0 );
488  analogwrite( this_motor_pin_B, (u16) dec );
489  analogwrite( this_motor_pin_C, (u16) 0 );
490  analogwrite( this_motor_pin_D, (u16) inc );
491  break;
492 
493  case 3: // from D to A
494  analogwrite( this_motor_pin_A, (u16) inc );
495  analogwrite( this_motor_pin_B, (u16) 0 );
496  analogwrite( this_motor_pin_C, (u16) 0 );
497  analogwrite( this_motor_pin_D, (u16) dec );
498  break;
499  }
500  }
