386 BT_RESPONSE BT_connect(u8 uart_port, u8 * bdaddr)
387 {
388  BT_RESPONSE response;
389 
390  BT_sendCommand(uart_port,"AT+BTPAR=1,%s\r", bdaddr); // Pair with remote device
391  response = BT_getExtendedResponse(uart_port);
392  if (response.code == BT_ERROR) return response;
393  BT_sendCommand(uart_port,"AT+BTCLT=%s,1\r", bdaddr); // on Port 1
394  response = BT_getExtendedResponse(uart_port);
395  // the device will enter data mode if OK
396  return response;
