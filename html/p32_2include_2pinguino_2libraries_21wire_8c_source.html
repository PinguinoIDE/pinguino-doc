65 /* ----------------------------------------------------------------------------
66  ---------- Initiates the one wire bus
67  ----------------------------------------------------------------------------
68  Performs a reset on the one-wire bus and returns the presence detect.
69  Reset is 2*480us long, presence checked another 60us later.
70  returns 0=presence, 1=no part
71  --------------------------------------------------------------------------*/
72 
73  u8 OneWireReset(u8 DQpin)
74  {
75  u8 presence;
76 
77  OneWireLow(DQpin); // pull DQ line low
78  Delayus(480); // leave it low for min. 480us
79  OneWireHigh(DQpin); // allow line to return high
80  Delayus(60); // takes 15 to 60 us
81  presence = digitalread(DQpin); // get presence signal
82  Delayus(420); // wait for end of timeslot (960-480-60=420)
83  return (presence);
84  }
