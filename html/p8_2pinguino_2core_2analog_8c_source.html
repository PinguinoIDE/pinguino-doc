134 u16 analogread(u8 channel)
135 {
136  u16 result=0;
137  // #if defined(PINGUINO4550) || defined(PICUNO_EQUO)
138  // ADCON1=0x07;
139  // #else
140  // ADCON1=0x0A;
141  // #endif
142 
143  #ifdef PICUNO_EQUO
144 
145  if(channel>=14 && channel<=16)
146  ADCON0=(channel-14) << 2;
147  else if(channel>=17 && channel<=19)
148  ADCON0=(channel-13) << 2;
149 
150  // RB - March 2014 - Enable individual Analog channel selection
151  // Has to be done for other Pinguino
152 
153  #elif defined(PINGUINO47J53A)
154 
155  if (channel > 15)
156  return 0;
157 
158  if (channel >= 8 && channel <= 15)
159  channel = channel - 8; // A0=8 to A7=15
160 
161  if (channel < 5)
162  TRISA |= 1 << channel; // channel as INPUT
163 
164  if (channel >= 5 && channel <= 7)
165  TRISE |= 1 << (channel - 5);// channel as INPUT
166 
167  ANCON0 |= 1 << channel; // channel enabled
168 
169  ADCON0 = channel << 2; // A0=0 to A7=7
170 
171  #else
172 
173  if(channel>=13 && channel<=20)
174  ADCON0=(channel-13) << 2; // A0 = 13, ..., A4 = 17
175  else if(channel<=5)
176  ADCON0 = channel << 2; // A0 = 0, ..., A4 = 4
177 
178  #endif
179 
180  ADCON0bits.ADON=1; // A/D Converter module is enabled
181 
182  for (result=1;result<10;result++) // Acquisition time
183  __asm NOP __endasm;
184 
185  ADCON0bits.GO=1; // Start A/D Conversion
186 
187  while (ADCON0bits.GO); // Wait for conversion stop
188 
189  result = ADRESH << 8;
190  result += ADRESL;
191 
192  ADCON0bits.ADON = 0; // A/D Converter module is disabled
193 
194  return(result);
195 }

199 /* --------------------------------------------------------------------
200  analogWrite
201 
202  analogWrite() uses PWM module.
203  The maximum PWM resolution (bits) for a given PWM frequency is given by the equation :
204 
205  PWM Resolution (max) = Log ( Fosc / Fpwm ) / Log ( 2 )
206 
207  If 10-bit (max) resolution is needed, then Fpwm must be <= Fosc / 1024.
208  For ex. if Fosc = 48 MHz then Fpwm must be <= 46875 Hz
209 
210  If Fosc = 48 MHz and Fpwm = 46875 Hz and Prescaler = 1, then :
211  PWM Period = 1 / Fpwm = [(PR2) + 1] • 4 • TOSC •(TMR2 Prescale Value)
212  PR2 = ( Fosc / Fpwm / 4 / TMR2 Prescale Value ) - 1
213  PR2 = 255
214  ------------------------------------------------------------------*/
