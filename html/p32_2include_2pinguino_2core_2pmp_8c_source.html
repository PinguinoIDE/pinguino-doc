236  // In Master mode 1, read and write strobe are combined into a

341 /*
342  In 16-bit Data mode (MODE16 bit (PMMODE<10>) = 1),
343  the write to the PMDIN register causes
344  PMDIN<15:0> to appear on the data bus, (PMD<15:0>).
345  In 8-bit Data mode (MODE16 bit (PMMODE<10>) = 0),
346  the write to the PMDIN register causes
347  PMDIN<7:0> to appear on the data bus, PMD<7:0>.
348  The upper 8 bits, PMD<15:8>, are ignored.
349 */
350 
351 void PMP_write(u16 value)
352 {
353  PMP_wait(); // wait for PMP to be available
354  PMDIN = value;
355 }
