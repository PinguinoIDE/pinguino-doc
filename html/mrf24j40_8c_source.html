335 void mrf24j40_handle_isr() {
336 
337 uns8 intstat;
338 
339  // first we need to get a copy of intstat to find out what happened
340  intstat = mrf24j40_short_addr_read(INTSTAT);
341 
342  // mrf24j40 intstat register is cleared on read
343  if (test_bit(intstat, INTSTAT_RXIF)) {
344  #ifndef __PIC32MX__
345  serial_print_str("R");
346  #endif
347  // handle receive
348  mrf24j40_receive_callback();
349  }
350  if (test_bit(intstat, INTSTAT_TXNIF)) {
351  #ifndef __PIC32MX__
352  serial_print_str("Tx complete txstat=0x");
353  #endif
354  uns8 stat = mrf24j40_short_addr_read(TXSTAT);
355  mrf24j40_transmit_callback(stat & 0b00000001, // success = 0
356  stat >> 6, // retries
357  test_bit(stat, TXSTAT_CCAFAIL)); // due to cca failure?
358  #ifndef __PIC32MX__
359  serial_print_int_hex(stat);
360  serial_print_nl();
361  #endif
362  }

366 uns8 mrf24j40_receive(uns8 *data, uns8 bytes_to_receive) {
367 
368 uns8 frame_length;
369 uns16 frame_pos;
370 uns8 buffer_count;
371 /*
372 1. Receive RXIF interrupt.
373 2. Disable host microcontroller interrupts.
374 3. Set RXDECINV = 1; disable receiving packets off air.
375 4. Read address, 0x300; get RXFIFO frame length value.
376 5. Read RXFIFO addresses, 0x301 through (0x300 + Frame Length + 2); read packet data plus LQI and RSSI.
377 6. Clear RXDECINV = 0; enable receiving packets.
378 7. Enable host microcontroller interrupts.
379 */
380  #ifndef __PIC32MX__
381  serial_putc('a');
382  #endif
383  // Disable reading packets off air
384  mrf24j40_short_addr_write(BBREG1, 1 << BBREG1_RXDECINV);
385  #ifndef __PIC32MX__
386  serial_putc('b');
387  #endif
388 
389  frame_pos = 0x300;
390  frame_length = mrf24j40_long_addr_read(frame_pos++);
391  buffer_count = 0;
392  #ifndef __PIC32MX__
393  serial_putc('c');
394  #endif
395 
396  while ((buffer_count <= bytes_to_receive) && (buffer_count <= frame_length + 2)) {
397  //0x301 through (0x300 + Frame Length + 2 ); read packet data plus LQI and RSSI.
398  data[buffer_count++] = mrf24j40_long_addr_read(frame_pos++);
399  }
400  #ifndef __PIC32MX__
401  serial_putc('d');
402  #endif
403 
404  // Re-enable reading packets off air
405  mrf24j40_short_addr_write(BBREG1, 0);
406  #ifndef __PIC32MX__
407  serial_putc('e');
408  #endif
409 
410  return buffer_count - 1;
411 
412 }

414 void mrf24j40_transmit_to_extended_address(uns8 frame_type, uns16 dest_pan_id, uns8 *dest_extended_address,
415  uns8 *data, uns8 data_length, uns8 ack) {
416 
417  // See notes below on frame control bytes format:
418  uns8 fc_msb = 0b11001100; // 64 bit dest (10,11) 64 bit src (14,15)
419  uns8 fc_lsb = 0b00000000 | frame_type; // pan id compression=0, data
420  #ifdef __PIC32MX__
421  u8 count=0;
422  #endif
423 
424  if (ack) {
425  set_bit(fc_lsb, 5); // ack bit
426  }
427 
428  data_sequence_number++;
429 
430  uns8 header_length = 3+8+8+2+2;
431  uns8 frame_length = header_length + data_length;
432 
433  // Write out data to mrf
434 
435  mrf24j40_long_addr_write(0x00, header_length);
436  mrf24j40_long_addr_write(0x01, frame_length);
437 
438  mrf24j40_long_addr_write(0x02, fc_lsb);
439  mrf24j40_long_addr_write(0x03, fc_msb);
440  mrf24j40_long_addr_write(0x04, data_sequence_number);
441 
442  mrf24j40_long_addr_write(0x05, dest_pan_id & 0xff); // dest pan id LSB
443  mrf24j40_long_addr_write(0x06, dest_pan_id >> 8); // MSB
444 
445  mrf24j40_long_addr_write(0x07, dest_extended_address[7]); // LSB
446  mrf24j40_long_addr_write(0x08, dest_extended_address[6]);
447  mrf24j40_long_addr_write(0x09, dest_extended_address[5]);
448  mrf24j40_long_addr_write(0x0a, dest_extended_address[4]);
449  mrf24j40_long_addr_write(0x0b, dest_extended_address[3]);
450  mrf24j40_long_addr_write(0x0c, dest_extended_address[2]);
451  mrf24j40_long_addr_write(0x0d, dest_extended_address[1]);
452  mrf24j40_long_addr_write(0x0e, dest_extended_address[0]); // MSB
453 
454  mrf24j40_long_addr_write(0x0f, pan_id & 0xff); // src pan id LSB
455  mrf24j40_long_addr_write(0x10, pan_id >> 8); // MSB
456 
457 
458  mrf24j40_long_addr_write(0x11, extended_address[7]); // LSB
459  mrf24j40_long_addr_write(0x12, extended_address[6]);
460  mrf24j40_long_addr_write(0x13, extended_address[5]);
461  mrf24j40_long_addr_write(0x14, extended_address[4]);
462  mrf24j40_long_addr_write(0x15, extended_address[3]);
463  mrf24j40_long_addr_write(0x16, extended_address[2]);
464  mrf24j40_long_addr_write(0x17, extended_address[1]);
465  mrf24j40_long_addr_write(0x18, extended_address[0]); // MSB
466  #ifndef __PIC32MX__
467  for (uns8 count=0; count < data_length; count++) {
468  mrf24j40_long_addr_write(count+header_length+2, data[count]);
469  #endif
470  #ifdef __PIC32MX__
471  for (count=0; count < data_length; count++) {
472  mrf24j40_long_addr_write(count+header_length+2, data[count]);
473  #endif
474  }
475 
476  uns8 txncon = mrf24j40_short_addr_read(TXNCON);
477 
478  set_bit(txncon, TXNCON_TXNTRIG);
479  if (ack) {
480  set_bit(txncon, TXNCON_TXNACKREQ);
481  }
482  mrf24j40_short_addr_write(TXNCON, txncon);
