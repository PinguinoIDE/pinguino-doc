194 
195  #error "Processor Not Yet Supported. Please, Take Contact with Developpers."
196 
197  #endif
198 
199  if (wpointer!=RXBUFFERLENGTH-1) // if not last place in buffer
200  newwp=wpointer+1; // place=place+1
201 
202  else
203  newwp=1; // else place=1
204 
205  if (rpointer!=newwp) // if read pointer!=write pointer
206  rx[wpointer++]=caractere; // store received char

419 /***********************************************************************
420  * USB SERIAL printFloat routine (SERIAL.printFloat)
421  * added by regis blanchot 14/06/2011
422  * write a float number on SERIAL port
423  * base : see const.h (DEC, BIN, HEXA, OCTO, ...)

426 #if defined(SERIALPRINTFLOAT)
427 void Serial_printFloat(float number, u8 digits)
428 {
429  u8 i, toPrint;
430  u16 int_part;
431  float rounding, remainder;
432 
433  // Handle negative numbers
434  if (number < 0.0)
435  {
436  Serial_puts('-', 1);
437  number = -number;
438  }
439 
440  // Round correctly so that print(1.999, 2) prints as "2.00"
441  rounding = 0.5;
442  for (i=0; i<digits; ++i)
443  rounding /= 10.0;
444 
445  number += rounding;
446 
447  // Extract the integer part of the number and print it
448  int_part = (u16)number;
449  remainder = number - (float)int_part;
450  Serial_printNumber(int_part, 10);
451 
452  // Print the decimal point, but only if there are digits beyond
453  if (digits > 0)
454  Serial_puts('.', 1);
455 
456  // Extract digits from the remainder one at a time
457  while (digits-- > 0)
458  {
459  remainder *= 10.0;
460  toPrint = (unsigned int)remainder; //Integer part without use of math.h lib, I think better! (Fazzi)
461  Serial_printNumber(toPrint, 10);
462  remainder -= toPrint;
463  }
464 }
