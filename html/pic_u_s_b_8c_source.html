193  // Process GET_STATUS
194 static void GetStatus(void) {
195  // Mask off the Recipient bits
196  u8 recipient = SetupPacket.bmRequestType & 0x1F;
197 #ifdef DEBUG_PRINT
198  printf("GetStatus\r\n");
199 #endif
200  controlTransferBuffer[0] = 0;
201  controlTransferBuffer[1] = 0;
202 
203  // See where the request goes
204  if (recipient == 0x00) {
205  // Device
206  requestHandled = 1;
207  // Set bits for self powered device and remote wakeup.
208  if (selfPowered)
209  controlTransferBuffer[0] |= 0x01;
210  if (remoteWakeup)
211  controlTransferBuffer[0] |= 0x02;
212  }
213  else if (recipient == 0x01) {
214  // Interface
215  requestHandled = 1;
216  }
217  else if (recipient == 0x02) {
218  // Endpoint
219  u8 endpointNum = SetupPacket.wIndex0 & 0x0F;
220  u8 endpointDir = SetupPacket.wIndex0 & 0x80;
221  requestHandled = 1;
222  // Endpoint descriptors are 8 bytes long, with each in and out taking 4 bytes
223  // within the endpoint. (See PIC datasheet.)
224  inPtr = (u8 *)&EP_OUT_BD(0) + (endpointNum * 8);
225  if (endpointDir)
226  inPtr += 4;
227  if(*inPtr & BDS_BSTALL)
228  controlTransferBuffer[0] = 0x01;
229  }
230 
231  if (requestHandled) {
232  outPtr = (u8 *)&controlTransferBuffer;
233  wCount = 2;
234  }
235 }

256  }
257  else if (recipient == 0x02) {
258  // Endpoint
259  u8 endpointNum = SetupPacket.wIndex0 & 0x0F;
260  u8 endpointDir = SetupPacket.wIndex0 & 0x80;

265  // within the endpoint. (See PIC datasheet.)

267  if (endpointDir)
268  inPtr += 4;
269 
270  if(SetupPacket.bRequest == SET_FEATURE)
271  *inPtr = 0x84;
272  else {
273  if(endpointDir == 1)
274  *inPtr = 0x00;
275  else

283 void ProcessStandardRequest(void) {
284  u8 request = SetupPacket.bRequest;
285 
286  if((SetupPacket.bmRequestType & 0x60) != 0x00)
287  // Not a standard request - don't process here. Class or Vendor
288  // requests have to be handled seperately.
289  return;
290 
291  if (request == SET_ADDRESS) {
292  // Set the address of the device. All future requests
293  // will come to that address. Can't actually set UADDR
294  // to the new address yet because the rest of the SET_ADDRESS
295  // transaction uses address 0.
296 #ifdef DEBUG_PRINT
297  printf("SET_ADDRESS: %uhx\r\n", SetupPacket.wValue0);
298 #endif
299  requestHandled = 1;
300  deviceState = ADDRESS;
301  deviceAddress = SetupPacket.wValue0;
302  }
303  else if (request == GET_DESCRIPTOR) {
304  GetDescriptor();
305  }
306  else if (request == SET_CONFIGURATION) {
307 #ifdef DEBUG_PRINT
308  printf("SET_CONFIGURATION\r\n");
309 #endif
310  requestHandled = 1;
311  currentConfiguration = SetupPacket.wValue0;
312  // TBD: ensure the new configuration value is one that
313  // exists in the descriptor.
314  if (currentConfiguration == 0)
315  // If configuration value is zero, device is put in
316  // address state (USB 2.0 - 9.4.7)
317  deviceState = ADDRESS;
318  else {
319  // Set the configuration.
320  deviceState = CONFIGURED;
321 
322  // Initialize the endpoints for all interfaces
323 #ifdef USB_USE_HID
324  HIDInitEndpoint();
325 #endif
326 #ifdef USB_USE_UART
327  UARTInitEndpoint();
328 #endif
329 #ifdef USB_USE_CDC
330  CDCInitEndpoint();
331 #endif
332 #ifdef USB_USE_BULK
333  BULKInitEndpoint();
334 #endif
335 
336  // TBD: Add initialization code here for any additional
337  // interfaces beyond the one used for the HID
338  }
339  }
340  else if (request == GET_CONFIGURATION) {
341 #ifdef DEBUG_PRINT
342  printf("GET_CONFIGURATION\r\n");
343 #endif
344  requestHandled = 1;
345  outPtr = (u8*)&currentConfiguration;
346  wCount = 1;
347  }
348  else if (request == GET_STATUS) {
349  GetStatus();
350  }
351  else if ((request == CLEAR_FEATURE) ||
352  (request == SET_FEATURE)) {
353  SetFeature();
354  }
355  else if (request == GET_INTERFACE) {
356  // No support for alternate interfaces. Send
357  // zero back to the host.
358 #ifdef DEBUG_PRINT
359  printf("GET_INTERFACE\r\n");
360 #endif
361  requestHandled = 1;
362  controlTransferBuffer[0] = 0;
363  outPtr = (u8 *)&controlTransferBuffer;
364  wCount = 1;
365  }
366  else if (request == SET_INTERFACE) {
367  // No support for alternate interfaces - just ignore.
368 #ifdef DEBUG_PRINT
369  printf("SET_INTERFACE\r\n");
370 #endif
371  requestHandled = 1;
372  }
373  else if (request == SET_DESCRIPTOR) {
374 #ifdef DEBUG_PRINT
375  printf("SET_DESCRIPTOR\r\n");
376 #endif
377  }
378  else if (request == SYNCH_FRAME) {
379 #ifdef DEBUG_PRINT
380  printf("SYNCH_FRAME\r\n");
381 #endif
382  }
383  else {
384 #ifdef DEBUG_PRINT
385  printf("Default Std Request\r\n");
386 #endif
387  }
388 }

394 void InDataStage(unsigned char ep) {
395 #if !USE_MEMCPY
396  u8 i;
397 #endif
398  word bufferSize;
399 
400  // Determine how many bytes are going to the host
401  if(wCount < EP0_BUFFER_SIZE)
402  bufferSize = wCount;
403  else
404  bufferSize = EP0_BUFFER_SIZE;
405 
406 #ifdef DEBUG_PRINT
407  // printf("USBCtrlTrfTxService: %d\r\n", bufferSize);
408 #endif
409 
410  // Load the high two bits of the u8 count into BC8:BC9
411  // Clear BC8 and BC9
412  EP_IN_BD(ep).Stat.uc &= ~(BDS_BC8 | BDS_BC9);
413  EP_IN_BD(ep).Stat.uc |= (u8)((bufferSize & 0x0300) >> 8);
414  EP_IN_BD(ep).Cnt = (u8)(bufferSize & 0xFF);
415  EP_IN_BD(ep).ADDR = PTR16(&controlTransferBuffer);
416 
417  // Update the number of bytes that still need to be sent. Getting
418  // all the data back to the host can take multiple transactions, so
419  // we need to track how far along we are.
420  wCount = wCount - bufferSize;
421 
422  // Move data to the USB output buffer from wherever it sits now.
423  inPtr = (u8 *)&controlTransferBuffer;
424 
425 #if USE_MEMCPY
426  memcpy(inPtr, outPtr, bufferSize);
427 #else
428  for (i=0;i<bufferSize;i++) {
429 #ifdef DEBUG_PRINT
430  //printf("0x%x = 0x%x (0x%uhx) ", PTR16(ramPtr), PTR16(romPtr), *romPtr);
431 #endif
432  *inPtr++ = *outPtr++;
433  }
434 #endif
435 }

477 void SetupStage(void) {
478  // Note: Microchip says to turn off the UOWN bit on the IN direction as
479  // soon as possible after detecting that a SETUP has been received.
480  EP_IN_BD(0).Stat.uc &= ~BDS_UOWN;
481  EP_OUT_BD(0).Stat.uc &= ~BDS_UOWN;
482 
483  // Initialize the transfer process
484  ctrlTransferStage = SETUP_STAGE;
485  requestHandled = 0; // Default is that request hasn't been handled
486  HIDPostProcess = 0; // Assume standard request until know otherwise
487  wCount = 0; // No bytes transferred
488 
489  // See if this is a standard (as definded in USB chapter 9) request
490  ProcessStandardRequest();
491 
492  // only Process CDC or HID if recipient is an interface
493  // See if the HID class can do something with it.
494 #ifdef USB_USE_HID
495  if ((SetupPacket.bmRequestType & USB_RECIP_MASK) == USB_RECIP_INTERFACE)
496  ProcessHIDRequest();
497 #endif
498 
499 #ifdef USB_USE_CDC
500  if ((SetupPacket.bmRequestType & USB_RECIP_MASK) == USB_RECIP_INTERFACE)
501  ProcessCDCRequest();
502 #endif
503  // TBD: Add handlers for any other classes/interfaces in the device
504  if (!requestHandled) {
505  // If this service wasn't handled then stall endpoint 0
506  EP_OUT_BD(0).Cnt = EP0_BUFFER_SIZE;
507  EP_OUT_BD(0).ADDR = PTR16(&SetupPacket);
508  EP_OUT_BD(0).Stat.uc = BDS_UOWN | BDS_BSTALL;
509  EP_IN_BD(0).Stat.uc = BDS_UOWN | BDS_BSTALL;
510  }
511  else if (SetupPacket.bmRequestType & 0x80) {
512  // Device-to-host
513  if(SetupPacket.wLength < wCount)
514  wCount = SetupPacket.wLength;
515  InDataStage(0);
516  ctrlTransferStage = DATA_IN_STAGE;
517  // Reset the out buffer descriptor for endpoint 0
518  EP_OUT_BD(0).Cnt = EP0_BUFFER_SIZE;
519  EP_OUT_BD(0).ADDR = PTR16(&SetupPacket);
520  EP_OUT_BD(0).Stat.uc = BDS_UOWN;
521 
522  // Set the in buffer descriptor on endpoint 0 to send data
523  EP_IN_BD(0).ADDR = PTR16(&controlTransferBuffer);
524  // Give to SIE, DATA1 packet, enable data toggle checks
525  EP_IN_BD(0).Stat.uc = BDS_UOWN | BDS_DTS | BDS_DTSEN;
526  }
527  else {
528  // Host-to-device
529  ctrlTransferStage = DATA_OUT_STAGE;
530 
531  // Clear the input buffer descriptor
532  EP_IN_BD(0).Cnt = 0;
533  EP_IN_BD(0).Stat.uc = BDS_UOWN | BDS_DTS | BDS_DTSEN;
534 
535  // Set the out buffer descriptor on endpoint 0 to receive data
536  EP_OUT_BD(0).Cnt = EP0_BUFFER_SIZE;
537  EP_OUT_BD(0).ADDR = PTR16(&controlTransferBuffer);
538  // Give to SIE, DATA1 packet, enable data toggle checks
539  EP_OUT_BD(0).Stat.uc = BDS_UOWN | BDS_DTS | BDS_DTSEN;
540  }
541 
542  // Enable SIE token and packet processing
543  UCONbits.PKTDIS = 0;
544 }

559  // This is the starting point for processing a Control Transfer. The code directly
560  // follows the sequence of transactions described in the USB spec chapter 5. The
561  // only Control Pipe in this firmware is the Default Control Pipe (endpoint 0).
562  // Control messages that have a different destination will be discarded.
563 void ProcessControlTransfer(void) {
564 #ifdef DEBUG_PRINT
565  printf("Pct\n\r");
566 #endif
567  if (USTATbits.DIR == OUT) {
568  // Endpoint 0:out
569  // Pull PID from middle of BD0STAT
570  u8 PID = (EP_OUT_BD(0).Stat.uc & 0x3C) >> 2;
571  if (PID == 0x0D)
572  // SETUP PID - a transaction is starting
573  SetupStage();
574  else if (ctrlTransferStage == DATA_OUT_STAGE) {
575  // Complete the data stage so that all information has
576  // passed from host to device before servicing it.
577  OutDataStage(0);
578 
579 #ifdef USB_USE_HID
580  if (HIDPostProcess) {
581  // Determine which report is being set.
582  u8 reportID = SetupPacket.wValue0;
583 
584  // Find out if an Output or Feature report has arrived on the control pipe.
585  // Get the report type from the Setup packet.
586  if (SetupPacket.wValue1 == 0x02) {
587  // Output report
588  SetOutputReport(reportID);
589  }
590  else if (SetupPacket.wValue1 == 0x03) {
591  // Feature report
592  SetFeatureReport(reportID);
593  }
594  else {
595  // Unknown report type
596  }
597  }
598 #endif
599 
600  // Turn control over to the SIE and toggle the data bit
601  if(EP_OUT_BD(0).Stat.DTS)
602  EP_OUT_BD(0).Stat.uc = BDS_UOWN | BDS_DTSEN;
603  else
604  EP_OUT_BD(0).Stat.uc = BDS_UOWN | BDS_DTS | BDS_DTSEN;
605  }
606  else {
607  // Prepare for the Setup stage of a control transfer
608  WaitForSetupStage();
609  }
610  }
611  else if(USTATbits.DIR == IN) {
612  // Endpoint 0:in
613  if ((UADDR == 0) && (deviceState == ADDRESS)) {
614  // TBD: ensure that the new address matches the value of
615  // "deviceAddress" (which came in through a SET_ADDRESS).
616  UADDR = SetupPacket.wValue0;
617 #ifdef DEBUG_PRINT
618  printf("UADDR = 0x%x\r\n", (word)UADDR);
619 #endif
620  if(UADDR == 0)
621  // If we get a reset after a SET_ADDRESS, then we need
622  // to drop back to the Default state.
623  deviceState = DEFAULT_STATUS;
624  }
625 
626  if (ctrlTransferStage == DATA_IN_STAGE) {
627  // Start (or continue) transmitting data
628  InDataStage(0);
629 
630  // Turn control over to the SIE and toggle the data bit
631  if(EP_IN_BD(0).Stat.DTS)
632  EP_IN_BD(0).Stat.uc = BDS_UOWN | BDS_DTSEN;
633  else
634  EP_IN_BD(0).Stat.uc = BDS_UOWN | BDS_DTS | BDS_DTSEN;
635  }
636  else {
637  // Prepare for the Setup stage of a control transfer
638  WaitForSetupStage();
639  }
640  }
641  else {
642 #ifdef DEBUG_PRINT
643  printf("IN on EP %d\r\n", (USTAT >> 3) & 0x0f);
644  printf("USTAT = 0x%uhx\r\n", USTAT);
645 #endif
646  }
647 }

820  // Main entry point for USB tasks. Checks interrupts, then checks for transactions.
821 void ProcessUSBTransactions(void)
822 {
823  // See if the device is connected yet.
824  if(deviceState == DETACHED)
825  return;
826 
827  // If the USB became active then wake up from suspend
828  if(UIRbits.ACTVIF && UIEbits.ACTVIE)
829  UnSuspend();
830 
831  // If we are supposed to be suspended, then don't try performing any
832  // processing.
833  if(UCONbits.SUSPND == 1)
834  return;
835 
836  // Process a bus reset
837  if (UIRbits.URSTIF && UIEbits.URSTIE)
838  BusReset();
839 
840  // No bus activity for a while - suspend the firmware
841  if (UIRbits.IDLEIF && UIEbits.IDLEIE)
842  Suspend();
843 
844 
845  if (UIRbits.SOFIF && UIEbits.SOFIE)
846  StartOfFrame();
847 
848  if (UIRbits.STALLIF && UIEbits.STALLIE)
849  Stall();

851  // TBD: See where the error came from.
852  // Clear errors
853  if (UIRbits.UERRIF && UIEbits.UERRIE)
854  UIRbits.UERRIF = 0;
855 
856  // Unless we have been reset by the host, no need to keep processing
857  if (deviceState < DEFAULT_STATUS)
858  return;
859 
860  // A transaction has finished. Try default processing on endpoint 0.
861  if(UIRbits.TRNIF && UIEbits.TRNIE)
862  {
863  ProcessControlTransfer();
864  // Turn off interrupt
865  UIRbits.TRNIF = 0;
866  }
