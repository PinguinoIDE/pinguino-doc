81 void EEPROM_write8(u8 address, u8 mydata)
82 {
83  EEADR = address;
84  EEDATA = mydata;
85  // start write sequence as described in datasheet, page 91
86  EECON1bits.EEPGD = 0;
87  EECON1bits.CFGS = 0;
88  EECON1bits.WREN = 1; // enable writes to data EEPROM
89  INTCONbits.GIE = 0; // disable interrupts
90  EECON2 = 0x55;
91  EECON2 = 0x0AA;
92  EECON1bits.WR = 1; // start writing
93  while (EECON1bits.WR)
94  {
95  Nop();
96  }
97  if (EECON1bits.WRERR)
98  {
99  #ifdef DEBUG
100  serial_printf("Writing Error !");
101  #endif
102  }
103  EECON1bits.WREN = 0;
104  INTCONbits.GIE = 1; // enable interrupts
105 }
