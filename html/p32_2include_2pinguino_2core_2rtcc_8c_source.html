305  Enable the secondary oscillator (SOSC) for the RTCC
306  To allow the RTCC to be clocked by an external 32.768 kHz crystal, the SOSCEN bit
307  (OSCCON<1>) must be set (refer to Register 6-1 in Section 6. “Oscillators” (DS61112)) in the
308  "PIC32 Family Reference Manual”. This is the only bit outside of the RTCC module with which
309  the user must be concerned for enabling the RTCC.
310  ----------------------------------------------------------------------------
311  01-01-2012 Mark Harper function renamed from RTCC_StartSOSC to RTCC_SOSCenable
312  ---------------------------------------------------------------------------*/
313 
314 void RTCC_SOSCenable(void)
315 {
316  SystemUnlock();
317  OSCCONbits.SOSCEN = 1;
318  SystemLock();
319  // Wait for stable SOCSC oscillator output before enabling the RTCC.
320  // This typically requires a 32 ms delay between enabling the SOSC and enabling the RTCC.
321  // 1st option
322  // Delayms(50);
323  // 2nd option
324  // Wait for the SOSC to run
325  while (RTCC_GetSOSCstatus() != RTCC_SOSC_ON);

455 /* -----------------------------------------------------------------------------
456  adjust RTCC from +511 to -512 clock pulses every one minute
457  0111111111 = Maximum positive adjustment, adds 511 RTC clock pulses every one minute
458  1000000000 = Minimum negative adjustment, subtracts 512 clock pulses every one minute
459  10 dec 2011 : bug fixed thanks to Mark Harper

461 
462 void RTCC_SetCalibration(int cal)
463 {
464  rtccTime t0;
465 
466  if (cal < -512) cal = -512;
467  if (cal > 511) cal = 511;
468  if (RTCCON & 0x8000) // if RTCC is ON
469  {
470  t0 = RTCC_GetTime();
471  if ((t0.seconds & 0xFF) == 00) // we're at second 00, wait auto-adjust to be performed
472  while(!(RTCCON & 0x2)); // wait until second half...
473  }
474  RTCC_SetWriteEnable();
475  RTCCONCLR = 0x03FF0000; // clear the calibration
476  RTCCONbits.CAL = cal;
477  RTCC_SetWriteDisable();
478 }
