55 void LedControl_init(u8 dataPin, u8 clkPin, u8 csPin, u8 numDevices)
56 {
57  u8 i;
58 
59  LEDCONTROL_SPI_MOSI = dataPin;
60  LEDCONTROL_SPI_CLK = clkPin;
61  LEDCONTROL_SPI_CS = csPin;
62  if(numDevices<=0 || numDevices>8 )
63  numDevices=8;
64  maxDevices=numDevices;
65  pinmode(LEDCONTROL_SPI_MOSI,OUTPUT);
66  pinmode(LEDCONTROL_SPI_CLK,OUTPUT);
67  pinmode(LEDCONTROL_SPI_CS,OUTPUT);
68  digitalwrite(LEDCONTROL_SPI_CS,HIGH);
69  LEDCONTROL_SPI_MOSI=dataPin;
70  for(i=0;i<64;i++)
71  status[i]=0x00;
72  for(i=0;i<maxDevices;i++)
73  {
74  LedControl_spiTransfer(i,OP_DISPLAYTEST,0);
75  //scanlimit is set to max on startup
76  LedControl_setScanLimit(i,7);
77  //decode is done in source
78  LedControl_spiTransfer(i,OP_DECODEMODE,0);
79  LedControl_clearDisplay(i);
80  //we go u8o shutdown-mode on startup
81  LedControl_shutdown(i,true);
82  }
83 }

234 void shiftOut(u8 dataPin, u8 clockPin, u8 bitOrder, u8 val)
235 {
236  u8 i;
237  u8 bitMask;
238 
239  for (i = 0; i < 8; i++)
240  {
241  if (bitOrder == LSBFIRST)
242  bitMask = (1 << i);
243  else
244  bitMask = (1 << (7 - i));
245 
246  digitalwrite(dataPin, (val & bitMask) ? HIGH : LOW);
247  digitalwrite(clockPin, HIGH);
248  digitalwrite(clockPin, LOW);
249  }
250 }

252 void LedControl_spiTransfer(u8 matrix, volatile u8 opcode, volatile u8 data)
253 {
254  //Create an array with the data to shift out
255  u8 offset=matrix*2;
256  u8 maxbytes=maxDevices*2;
257  u8 i;
258 
259  for(i=0;i<maxbytes;i++)
260  spidata[i]=(u8)0;
261  //put our device data u8o the array
262  spidata[offset+1]=opcode;
263  spidata[offset]=data;
264  //enable the line
265  digitalwrite(LEDCONTROL_SPI_CS,LOW);
266  //Now shift out the data
267  for(i=maxbytes;i>0;i--)
268  shiftOut(LEDCONTROL_SPI_MOSI,LEDCONTROL_SPI_CLK,MSBFIRST,spidata[i-1]);
269  //latch the data onto the display
270  digitalwrite(LEDCONTROL_SPI_CS,HIGH);
271 }
