23 /*
24  SimpleModbusSlave allows you to communicate
25  to any slave using the Modbus RTU protocol.
26 
27  This implementation DOES NOT fully comply with the Modbus specifications.
28 
29  Specifically the frame time out have not been implemented according
30  to Modbus standards. The code does however combine the check for
31  inter character time out and frame time out by incorporating a maximum
32  time out allowable when reading from the message stream.
33 
34  SimpleModbusSlave implements an unsigned int return value on a call to modbus_update().
35  This value is the total error count since the slave started. It's useful for fault finding.
36 
37  This code is for a Modbus slave implementing functions 3 and 16
38  function 3: Reads the binary contents of holding registers (4X references)
39  function 16: Presets values into a sequence of holding registers (4X references)
40 
41  All the functions share the same register array.
42 
43  Note:
44  The Arduino serial ring buffer is 128 bytes or 64 registers.
45  Most of the time you will connect the arduino to a master via serial
46  using a MAX485 or similar.
47 
48  In a function 3 request the master will attempt to read from your
49  slave and since 5 bytes is already used for ID, FUNCTION, NO OF BYTES
50  and two BYTES CRC the master can only request 122 bytes or 61 registers.
51 
52  In a function 16 request the master will attempt to write to your
53  slave and since a 9 bytes is already used for ID, FUNCTION, ADDRESS,
54  NO OF REGISTERS, NO OF BYTES and two BYTES CRC the master can only write
55  118 bytes or 59 registers.
56 
57  Using a USB to Serial converter the maximum bytes you can send is
58  limited to its internal buffer which differs between manufactures.
59 
60  The functions included here have been derived from the
61  Modbus Specifications and Implementation Guides
62 
63  http://www.modbus.org/docs/Modbus_over_serial_line_V1_02.pdf
64  http://www.modbus.org/docs/Modbus_Application_Protocol_V1_1b.pdf
65  http://www.modbus.org/docs/PI_MBUS_300.pdf
66 */

280 u16 modbuss_update(MODBUSS_DATA * ModBusS_Data)
281 {
282  char caract;
283  unsigned char buffer;
284  unsigned char overflow;
285  unsigned char id;
286  u16 crc;
287  u16 startingAddress;
288  u16 no_of_registers;
289  u16 maxData;
290  unsigned char index;
291  unsigned char address;
292  u16 crc16;
293  unsigned char noOfBytes;
294  unsigned char responseFrameSize;
295  u16 temp;
296 
297  if (modbuss_SerialClearRxError(ModBusS_Data) == FALSE)
298  return;
299 
300  if (modbuss_SerialAvailable(ModBusS_Data))
301  {
302  buffer = 0;
303  overflow = 0;
304 
305  while (modbuss_SerialAvailable(ModBusS_Data))
306  {
307  // The maximum number of bytes is limited to the serial buffer size of 128 bytes
308  // If more bytes is received than the MODBUSS_BUFFER_SIZE the overflow flag will be set and the
309  // serial buffer will be red untill all the data is cleared from the receive buffer.
310  if (overflow)
311  modbuss_SerialRead(ModBusS_Data);
312  else
313  {
314  caract = modbuss_SerialRead(ModBusS_Data);
315  if (buffer == MODBUSS_BUFFER_SIZE)
316  overflow = 1;
317  else
318  {
319  ModBusS_Data->frame[buffer] = caract;
320  buffer++;
321  }
322  }
323  Delayus(ModBusS_Data->T1_5); // inter character time out
324  }
325 
326  // If an overflow occurred increment the errorCount
327  // variable and return to the main sketch without
328  // responding to the request i.e. force a timeout
329  if (overflow)
330  return ModBusS_Data->errorCount++;
331 
332  // The minimum request packet is 8 bytes for function 3 & 16
333  if (buffer > 7)
334  {
335  id = ModBusS_Data->frame[0];
336  ModBusS_Data->broadcastFlag = 0;
337  if (id == 0)
338  ModBusS_Data->broadcastFlag = 1;
339 
340  if (id == ModBusS_Data->slaveID || ModBusS_Data->broadcastFlag) // if the recieved ID matches the ModBusS_Data->slaveID or broadcasting id (0), continue
341  {
342  crc = ((ModBusS_Data->frame[buffer - 2] << 8) | ModBusS_Data->frame[buffer - 1]); // combine the crc Low & High bytes
343  if (modbuss_calculateCRC(ModBusS_Data,buffer - 2) == crc) // if the calculated crc matches the recieved crc continue
344  {
345  ModBusS_Data->function = ModBusS_Data->frame[1];
346  startingAddress = ((ModBusS_Data->frame[2] << 8) | ModBusS_Data->frame[3]); // combine the starting address bytes
347  no_of_registers = ((ModBusS_Data->frame[4] << 8) | ModBusS_Data->frame[5]); // combine the number of register bytes
348  maxData = startingAddress + no_of_registers;
349 
350  // broadcasting is not supported for function 3
351  if (!ModBusS_Data->broadcastFlag && (ModBusS_Data->function == 3))
352  {
353  if (startingAddress < ModBusS_Data->holdingRegsSize) // check exception 2 ILLEGAL DATA ADDRESS
354  {
355  if (maxData <= ModBusS_Data->holdingRegsSize) // check exception 3 ILLEGAL DATA VALUE
356  {
357  noOfBytes = no_of_registers * 2;
358  // ID, function, noOfBytes, (dataLo + dataHi)*number of registers,
359  // crcLo, crcHi
360  responseFrameSize = 5 + noOfBytes;
361  ModBusS_Data->frame[0] = ModBusS_Data->slaveID;
362  ModBusS_Data->frame[1] = ModBusS_Data->function;
363  ModBusS_Data->frame[2] = noOfBytes;
364  address = 3; // PDU starts at the 4th byte
365 
366  for (index = startingAddress; index < maxData; index++)
367  {
368  temp = ModBusS_Data->regs[index];
369  ModBusS_Data->frame[address] = temp >> 8; // split the register into 2 bytes
370  address++;
371  ModBusS_Data->frame[address] = temp & 0xFF;
372  address++;
373  }
374 
375  crc16 = modbuss_calculateCRC(ModBusS_Data,responseFrameSize - 2);
376  ModBusS_Data->frame[responseFrameSize - 2] = crc16 >> 8; // split crc into 2 bytes
377  ModBusS_Data->frame[responseFrameSize - 1] = crc16 & 0xFF;
378  modbuss_sendPacket(ModBusS_Data,responseFrameSize);
379  }
380  else
381  modbuss_exceptionResponse(ModBusS_Data,3); // exception 3 ILLEGAL DATA VALUE
382  }
383  else
384  modbuss_exceptionResponse(ModBusS_Data,2); // exception 2 ILLEGAL DATA ADDRESS
385  }
386  else if (ModBusS_Data->function == 16)
387  {
388  // Check if the recieved number of bytes matches the calculated bytes
389  // minus the request bytes.
390  // id + function + (2 * address bytes) + (2 * no of register bytes) +
391  // byte count + (2 * CRC bytes) = 9 bytes
392  if (ModBusS_Data->frame[6] == (buffer - 9))
393  {
394  if (startingAddress < ModBusS_Data->holdingRegsSize) // check exception 2 ILLEGAL DATA ADDRESS
395  {
396  if (maxData <= ModBusS_Data->holdingRegsSize) // check exception 3 ILLEGAL DATA VALUE
397  {
398  address = 7; // start at the 8th byte in the ModBusS_Data->frame
399  for (index = startingAddress; index < maxData; index++)
400  {
401  ModBusS_Data->regs[index] = ((ModBusS_Data->frame[address] << 8) | ModBusS_Data->frame[address + 1]);
402  address += 2;
403  }
404  // only the first 6 bytes are used for CRC calculation
405  crc16 = modbuss_calculateCRC(ModBusS_Data,6);
406  ModBusS_Data->frame[6] = crc16 >> 8; // split crc into 2 bytes
407  ModBusS_Data->frame[7] = crc16 & 0xFF;
408 
409  // a function 16 response is an echo of the first 6 bytes from
410  // the request + 2 crc bytes
411  if (!ModBusS_Data->broadcastFlag) // don't respond if it's a broadcast message
412  modbuss_sendPacket(ModBusS_Data,8);
413  }
414  else
415  modbuss_exceptionResponse(ModBusS_Data,3); // exception 3 ILLEGAL DATA VALUE
416  }
417  else
418  modbuss_exceptionResponse(ModBusS_Data,2); // exception 2 ILLEGAL DATA ADDRESS
419  }
420  else
421  ModBusS_Data->errorCount++; // corrupted packet
422  }
423  else
424  modbuss_exceptionResponse(ModBusS_Data,1); // exception 1 ILLEGAL FUNCTION
425  }
426  else // checksum failed
427  ModBusS_Data->errorCount++;
428  } // incorrect id
429  }
430  else
431  if (buffer > 0 && buffer < 8)
432  ModBusS_Data->errorCount++; // corrupted packet
433  }
434  return ModBusS_Data->errorCount;
435 }

438 void modbuss_exceptionResponse(MODBUSS_DATA * ModBusS_Data,unsigned char exception)
439 {
440  u16 crc16;
441  // each call to modbuss_exceptionResponse() will increment the errorCount
442  ModBusS_Data->errorCount++;
443  if (!ModBusS_Data->broadcastFlag) // don't respond if its a broadcast message
444  {
445  ModBusS_Data->frame[0] = ModBusS_Data->slaveID;
446  ModBusS_Data->frame[1] = (ModBusS_Data->function | 0x80); // set MSB bit high, informs the master of an exception
447  ModBusS_Data->frame[2] = exception;
448  crc16 = modbuss_calculateCRC(ModBusS_Data,3); // ID, function|0x80, exception code
449  ModBusS_Data->frame[3] = crc16 >> 8;
450  ModBusS_Data->frame[4] = crc16 & 0xFF;
451  // exception response is always 5 bytes
452  // ID, function + 0x80, exception code, 2 bytes crc
453  modbuss_sendPacket(ModBusS_Data,5);
454  }
455 }

458 void modbuss_configure(MODBUSS_DATA * ModBusS_Data,char port,long baud, unsigned char _slaveID,
459  unsigned char _TxEnablePin, u16 _holdingRegsSize,
460  u16* _regs)
461 {
462  ModBusS_Data->port = port;
463  ModBusS_Data->slaveID = _slaveID;
464  modbuss_SerialBegin(ModBusS_Data,baud);
465  ModBusS_Data->holdingRegsSize = _holdingRegsSize;
466  ModBusS_Data->regs = _regs;
467  ModBusS_Data->TxEnablePin = _TxEnablePin;
468  if (ModBusS_Data->TxEnablePin > 0)
469  {
470  pinmode(ModBusS_Data->TxEnablePin, OUTPUT);
471  digitalwrite(ModBusS_Data->TxEnablePin, LOW);
472  }
473  ModBusS_Data->errorCount = 0; // initialize errorCount
474 
475  // Modbus states that a baud rate higher than 19200 must use a fixed 750 us
476  // for inter character time out and 1.75 ms for a ModBusS_Data->frame delay for baud rates
477  // below 19200 the timing is more critical and has to be calculated.
478  // E.g. 9600 baud in a 10 bit packet is 960 characters per second
479  // In milliseconds this will be 960characters per 1000ms. So for 1 character
480  // 1000ms/960characters is 1.04167ms per character and finally modbus states
481  // an inter-character must be 1.5T or 1.5 times longer than a character. Thus
482  // 1.5T = 1.04167ms * 1.5 = 1.5625ms. A ModBusS_Data->frame delay is 3.5T.
483 
484  if (baud > 19200)
485  {
486  ModBusS_Data->T1_5 = 750;
487  ModBusS_Data->T3_5 = 1750;
488  }
489  else
490  {
491  ModBusS_Data->T1_5 = 15000000/baud; // 1T * 1.5 = T1.5
492  ModBusS_Data->T3_5 = 35000000/baud; // 1T * 3.5 = T3.5
493  }
494 }

496 u16 modbuss_calculateCRC(MODBUSS_DATA * ModBusS_Data,unsigned char bufferSize)
497 {
498  u16 temp, temp2, flag;
499  unsigned char i;
500  unsigned char j;
501  temp = 0xFFFF;
502  for (i = 0; i < bufferSize; i++)
503  {
504  temp = temp ^ ModBusS_Data->frame[i];
505  for (j = 1; j <= 8; j++)
506  {
507  flag = temp & 0x0001;
508  temp >>= 1;
509  if (flag)
510  temp ^= 0xA001;
511  }
512  }
513  // Reverse byte order.
514  temp2 = temp >> 8;
515  temp = (temp << 8) | temp2;
516  temp &= 0xFFFF;
517  // the returned value is already swapped
518  // crcLo byte is first & crcHi byte is last
519  return temp;
520 }

523 {
524  unsigned char i;

529  modbuss_SerialWrite(ModBusS_Data,ModBusS_Data->frame[i]);
530 
531  modbuss_SerialFlush(ModBusS_Data);
