119  ---------- DS18B20Read()
120  ----------------------------------------------------------------------------
121  * Description: reads the ds18x20 device on the 1-wire bus and returns the temperature

124  resolution = 9 to 12 bit resolution

128  u8 DS18B20Read(u8 pin, u8 rom, u8 resolution, DS18B20_Temperature * t)
129  {
130  u8 res, busy = LOW;
131  u8 temp_lsb, temp_msb;
132  u16 temp;
133 
134  switch (resolution)
135  {
136  case RES12BIT: res = 0b01100000; break; // 12-bit resolution
137  case RES11BIT: res = 0b01000000; break; // 11-bit resolution
138  case RES10BIT: res = 0b00100000; break; // 10-bit resolution
139  case RES9BIT: res = 0b00000000; break; // 9-bit resolution
140  default: res = 0b00000000; break; // 9-bit resolution
142  }
143 
144  if (!DS18B20Configure(pin, rom, 0, 0, res)) return FALSE; // no alarm
145 
146  if (OneWireReset(pin)) return FALSE;
147 
148  if (rom == SKIPROM)
149  {
150  // Skip ROM, address all devices
151  OneWireWrite(pin, SKIPROM);
152  }
153  else
154  {
155  // Talk to a particular device
156  if (!DS18B20MatchRom(pin, rom)) return FALSE;
157  }
158 
159  OneWireWrite(pin, CONVERT_T); // Start temperature conversion
160 
161  while (busy == LOW) // Wait while busy ( = bus is low)
162  busy = OneWireRead(pin);
163 
164  if (OneWireReset(pin)) return FALSE;
165 
166  if (rom == SKIPROM)
167  {
168  // Skip ROM, address all devices
169  OneWireWrite(pin, SKIPROM);
170  }
171  else
172  {
173  // Talk to a particular device
174  if (!DS18B20MatchRom(pin, rom)) return FALSE;
175  }
176 
177  OneWireWrite(pin, READ_SCRATCHPAD);// Read scratchpad
178 
179  temp_lsb = OneWireRead(pin); // byte 0 of scratchpad : temperature lsb
180  temp_msb = OneWireRead(pin); // byte 1 of scratchpad : temperature msb
181 
182  if (OneWireReset(pin)) return FALSE;
183 
184  // Calculation
185  // ---------------------------------------------------------------------
186  // Temperature Register Format
187  // BIT7 BIT6 BIT5 BIT4 BIT3 BIT2 BIT1 BIT0
188  // LS BYTE 2^3 2^2 2^1 2^0 2^-1 2^-2 2^-3 2^-4
189  // BIT15 BIT14 BIT13 BIT12 BIT11 BIT10 BIT9 BIT8
190  // MS BYTE S S S S S 2^6 2^5 2^4
191  // S = SIGN
192 
193  temp = temp_msb;
194  temp = (temp << 8) + temp_lsb; // combine msb & lsb into 16 bit variable
195 
196  if (temp_msb & 0b11111000) // test if sign is set, i.e. negative
197  {
198  t->sign = 1;
199  temp = (temp ^ 0xFFFF) + 1; // 2's complement conversion
200  }
201  else
202  {
203  t->sign = 0;
204  }
205 
206  t->integer = (temp >> 4) & 0x7F; // fractional part is removed, leaving only integer part
207 
208 /*
209  t->fraction = 0; // fractional part
210  if (BitRead(temp, 0)) t->fraction += 625;
211  if (BitRead(temp, 1)) t->fraction += 1250;
212  if (BitRead(temp, 2)) t->fraction += 2500;
213  if (BitRead(temp, 3)) t->fraction += 5000;
214 */
215  t->fraction = (temp & 0x0F) * 625;
216  t->fraction /= 100; // two digits after decimal
217 
218  return TRUE;
219  }

254 /* ----------------------------------------------------------------------------
255  ---------- Address a specific slave device on a multidrop or single-drop bus
256  ----------------------------------------------------------------------------
257  * Arguments: pin = pin number where one wire bus is connected.
258  rom = index of the sensor
259  * Description: reads and returns a byte of data from the device.
260  ----------------------------------------------------------------------------
261  The match ROM command followed by a 64-bit ROM code sequence allows the bus
262  master to address a specific slave device on a multidrop or single-drop bus.
263  Only the slave that exactly matches the 64-bit ROM code sequence will respond
264  to the function command issued by the master; all other slaves on the bus
265  will wait for a reset pulse.
266  --------------------------------------------------------------------------*/
267 
268  u8 DS18B20MatchRom(u8 pin, u8 rom)
269  {
270  u8 i;
271  if (OneWireReset(pin)) return FALSE;
272  OneWireWrite(pin, MATCHROM); // Match Rom
273  for (i = 0; i < 8; i++) // Send the Address ROM Code.
274  OneWireWrite(pin, DS18B20Rom[rom][i]);
275  return TRUE;
276  }

347 /* ----------------------------------------------------------------------------
348  ---------- Next
349  ----------------------------------------------------------------------------
350  * Arguments: pin number where one wire bus is connected.
351  * Description: searches for the next device on the 1-wire bus. If
352  there are no more devices on the 1-wire then false is returned.
353  --------------------------------------------------------------------------*/
354 
355  u8 DS18B20GetNext(u8 pin)
356  {
357  u8 m = 1; // ROM Bit index
358  u8 n = 0; // ROM Byte index
359  u8 k = 1; // bit mask
360  u8 x = 0;
361  u8 discrepMarker = 0; // discrepancy marker
362  u8 g; // Output bit
363  u8 nxt; // return value
364  int flag;
365 
366  nxt = FALSE; // set the next flag to false
367  dowcrc = 0; // reset the dowcrc
368 
369  flag = OneWireReset(pin); // reset the 1-wire
370  if(flag||doneFlag) // no parts -> return false
371  {
372  lastDiscrep = 0; // reset the search
373  return FALSE;
374  }
375  // send SearchROM command for all eight bytes
376  OneWireWrite(pin, SEARCHROM);
377  do {
378  x = 0;
379  if(OneWireReadBit(pin) == 1) x = 2;
380  //myDelay_10us(12);
381  if(OneWireReadBit(pin) == 1 ) x |= 1;
382  if(x == 3)
383  break;
384  else
385  {
386  if(x > 0) // all devices coupled have 0 or 1
387  g = x >> 1; // bit write value for search
388  else
389  {
390  // if this discrepancy is before the last
391  // discrepancy on a previous Next then pick
392  // the same as last time
393  if(m < lastDiscrep)
394  g = ( (ROM[n] & k) > 0);
395  else // if equal to last pick 1
396  g = (m == lastDiscrep); // if not then pick 0
397  // if 0 was picked then record position with mask k
398  if (g == 0) discrepMarker = m;
399  }
400  if (g == 1) // isolate bit in ROM[n] with mask k
401  ROM[n] |= k;
402  else
403  ROM[n] &= ~k;
404  OneWireWriteBit(pin, g); // ROM search write
405  m++; // increment bit counter m
406  k = k << 1; // and shift the bit mask k
407  if(k == 0) // if the mask is 0 then go to new ROM
408  { // byte n and reset mask
409  DS18B20_crc(ROM[n]); // accumulate the CRC
410  n++;
411  k++;
412  }
413  }
414  } while (n < 8); // loop until through all ROM bytes 0-7
415  if(m < 65 || dowcrc) // if search was unsuccessful then
416  lastDiscrep=0; // reset the last discrepancy to 0
417  else // search was successful, so set lastDiscrep, lastOne, nxt
418  {
419  lastDiscrep = discrepMarker;
420  doneFlag = (lastDiscrep == 0);
421  nxt = TRUE; // indicates search is not complete yet, more parts remain
422  }
423  return nxt;
424  }

427  ---------- CRC
428  ----------------------------------------------------------------------------
429  * Arguments: x
430  * Global: dowcrc - global crc stored here
431  * Description: Update the CRC for transmitted and received data using
432  8-bit CRC equivalent polynomial function : x^8 + x^5 + x + 1
433  * 1rst method
434  dowcrc = dscrc_table[dowcrc^x];
435  return dowcrc;
436  * 2nd method
437  int i = (x ^ dowcrc) & 0xff;
438  dowcrc = 0;
439  if(i & 1)
440  dowcrc ^= 0x5e;
441  if(i & 2)
442  dowcrc ^= 0xbc;
443  if(i & 4)
444  dowcrc ^= 0x61;
445  if(i & 8)
446  dowcrc ^= 0xc2;
