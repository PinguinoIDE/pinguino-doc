127 static u16 System_readFlashMemory(u32 address)
128 {
129  u8 h8,l8;
130 
131  TBLPTRU=address>>16;
132  TBLPTRH=address>>8;
133  TBLPTRL=address;
134  __asm tblrd*+ __endasm;
135  l8=TABLAT;

137  h8=TABLAT;

141 /* ----------------------------------------------------------------------------
142  Calculates the CPU frequency.
143 
144  - if PLL is enabled
145  * CPU Freq. = 48MHz / CPUDIV
146  * Incoming Freq. = 4 * PLLDIV (unused) (x3 or x4 on 25k50)
147  - if PLL is disabled
148  * if OSCCONbits.SCS == 0, Incoming Freq. = External Oscillator
149  CPU Freq. = Incoming Freq. / CPUDIV
150  Inc. Freq. is unknown, must be defined by user (for ex. #define CRYSTAL = 8000000L)
151  * if OSCCONbits.SCS == 1, Incoming Freq. = Timer1
152  CPU Freq. = Incoming Freq.
153  how to get freq. from Timer 1 ?
154  * if OSCCONbits.SCS >= 2, Incoming Freq. = Internal Oscillator
155  CPU Freq. = Incoming Freq.
156  Inc. Freq. = IRCF bits
157 
158  TODO : 18f14k22
159  --------------------------------------------------------------------------*/

161 u32 System_getCpuFrequency()
162 {
163  u8 CPUDIV;
164 
165  #if defined(__18f2455) || defined(__18f4455) || \
166  defined(__18f2550) || defined(__18f4550)
167 
168  u8 fosc, CPUDIV_XTAL;
169 
170  #endif
171 
172  switch (OSCCONbits.SCS)
173  {
174  case 0: // primary osc. (internal or external / CPUDIV)
175 
176  #if defined(__18f25k50) || defined(__18f45k50)
177 
178  if (OSCCON2bits.PLLEN)
179  {
180  return 48000000L;
181  }
182  else
183  {
184  return CRYSTAL;
185  }
186 
187  #endif
188 
189  // CPUDIV ?
190 
191  #if defined(__18f2455) || defined(__18f4455) || \
192  defined(__18f2550) || defined(__18f4550)
193 
194  CPUDIV = ( System_readFlashMemory(__CONFIG1L) & 0b00011000 ) >> 3;
195  CPUDIV_XTAL = cpudiv_xtal[CPUDIV];
196  fosc = System_readFlashMemory(__CONFIG1H) & 0b00001111;
197 
198  #else // xxJ5x
199 
200  CPUDIV = System_readFlashMemory(__CONFIG1H) & 0b00000011;
201 
202  #endif
203 
204  CPUDIV = cpudiv[CPUDIV];
205 
206  // PLL ?
207 
208  #if defined(__18f2455) || defined(__18f4455) || \
209  defined(__18f2550) || defined(__18f4550)
210 
211  fosc >>=1;
212  if( (fosc==0) || (fosc==2) || (fosc==6) )
213  {
214  return CRYSTAL / CPUDIV_XTAL;
215  }
216  else
217  {
218  return 96000000UL / CPUDIV;
219  }
220 
221  #elif defined(__18f26j50) || defined(__18f46j50) || \
222  defined(__18f26j53) || defined(__18f46j53) || \
223  defined(__18f27j53) || defined(__18f47j53)
224 
225  if (OSCTUNEbits.PLLEN)
226  {
227  return 48000000UL / CPUDIV;
228  }
229  else
230  {
231  return CRYSTAL / CPUDIV;
232  }
233 
234  #endif
235 
236  case 1: // secondary osc. (timer 1, most of the time 32768 Hz)
237  return 32768;
238 
239  case 2: // reserved or postscaled internal clock
