526 // bresenham's algorithm - thx wikpedia
527 void ST7565_drawLine(u8 x0, u8 y0, u8 x1, u8 y1, u8 color)
528 {
529  u8 dx, dy, steep = abs(y1 - y0) > abs(x1 - x0);
530  u8 err, ystep;
531  if (steep)
532  {
533  swap(x0, y0);
534  swap(x1, y1);
535  }
536 
537  if (x0 > x1)
538  {
539  swap(x0, x1);
540  swap(y0, y1);
541  }
542 
543  // much faster to put the test here, since we've already sorted the points
544  updateBoundingBox(x0, y0, x1, y1);
545 
546  dx = x1 - x0;
547  dy = abs(y1 - y0);
548 
549  err = dx / 2;
550 
551  if (y0 < y1)
552  {
553  ystep = 1;
554  } else {
555  ystep = -1;}
556 
557  for (; x0<=x1; x0++)
558  {
559  if (steep)
560  {
561  ST7565_my_setPixel(y0, x0, color);
562  } else {
563  ST7565_my_setPixel(x0, y0, color);
564  }
565  err -= dy;
566  if (err < 0)
567  {
568  y0 += ystep;
569  err += dx;
570  }
571  }
572 }
