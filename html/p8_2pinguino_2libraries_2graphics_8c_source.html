106 void drawLine(u16 x0, u16 y0, u16 x1, u16 y1)
107 {
108  s16 steep, t;
109  s16 deltax, deltay, error;
110  s16 x, y;
111  s16 ystep;
112 
113  // simple clipping
114 /*
115  if (( x0 < 0) || (x0 > HRES)) return;
116  if (( x1 < 0) || (x1 > HRES)) return;
117  if (( y0 < 0) || (y0 > VRES)) return;
118  if (( y1 < 0) || (y1 > VRES)) return;
119 */
120 
121  steep = abs(y1 - y0) > abs(x1 - x0);
122 
123  if (steep)
124  {
125  swap(x0, y0);
126  swap(x1, y1);
127  }
128 
129  if (x0 > x1)
130  {
131  swap(x0, x1);
132  swap(y0, y1);
133  }
134 
135  deltax = x1 - x0;
136  deltay = abs(y1 - y0);
137 
138  error = 0;
139  y = y0;
140 
141  if (y0 < y1) ystep = 1; else ystep = -1;
142 
143  for (x = x0; x < x1; x++)
144  {
145  if (steep)
146  drawPixel(y,x);
147  else
148  drawPixel(x,y);
149 
150  error += deltay;
151 
152  if ( (error<<1) >= deltax)
153  {
154  y += ystep;
155  error -= deltax;
156  }
157  }
158 }
